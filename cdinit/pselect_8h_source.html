218 // If events are pending, process an unspecified number of them.
219 // If no events are pending, wait until one event is received and
220 // process this event (and possibly any other events received
221 // simultaneously).
222 // If processing an event removes a watch, there is a possibility
223 // that the watched event will still be reported (if it has
224 // occurred) before pull_events() returns.
225 //
226 // do_wait - if false, returns immediately if no events are
227 // pending.
228 void pull_events(bool do_wait) noexcept
229 {
230 struct timespec ts;
231 struct timespec *wait_ts = nullptr;
232
233 Base::lock.lock();
234
235 // Check whether any timers are pending, and what the next timeout is.
236 this->process_monotonic_timers(do_wait, ts, wait_ts);
237
238 fd_set read_set_c;
239 fd_set write_set_c;
240 fd_set err_set;
241
242 read_set_c = read_set;
243 write_set_c = write_set;
244 err_set = read_set;
245
246 const sigset_t &active_sigmask = this->get_active_sigmask();
247
248 // We want "poll_sigmask" to have unmasked both the signals that were previously unmasked, and
249 // the signals that we need to see because they are being watched.
250 this->sigmaskf(SIG_UNBLOCK, nullptr, &poll_sigmask);
251
252 // This is horrible, but hopefully will be optimised well. POSIX gives no way to combine signal
253 // sets other than this.
254 for (int i = 1; i < NSIG; i++) {
255 if (!sigismember(&active_sigmask, i)) {
256 sigdelset(&poll_sigmask, i);
257 }
258 }
259 int nfds = max_fd + 1;
260 Base::lock.unlock();
261
262 // using sigjmp/longjmp is ugly, but there is no other way. If a signal that we're watching is
263 // received during polling, it will longjmp back to here:
264 if (sigsetjmp(this->get_sigreceive_jmpbuf(), 1) != 0) {
265 this->process_signal(poll_sigmask);
266 do_wait = false;
267 }
268
269 if (!do_wait) {
270 ts.tv_sec = 0;
271 ts.tv_nsec = 0;
272 wait_ts = &ts;
273 }
274
275 std::atomic_signal_fence(std::memory_order::memory_order_release);
276
277 int r = pselect(nfds, &read_set_c, &write_set_c, &err_set, wait_ts, &poll_sigmask);
278
279 if (r == -1 || r == 0) {
280 // signal or no events
281 if (r == 0) {
282 if (! do_wait) {
283 // At least on Mac OS, pselect doesn't seem to give us a pending signal
284 // if we have a zero timeout. Force detection using sigmask:
285 sigset_t origmask;
286 this->sigmaskf(SIG_SETMASK, &poll_sigmask, &origmask);
287 this->sigmaskf(SIG_SETMASK, &origmask, nullptr);
288 }
289 else {
290 // timeout:
291 Base::lock.lock();
292 this->process_monotonic_timers();
293 Base::lock.unlock();
294 }
295 }
296 return;
297 }
298
299 process_events(&read_set_c, &write_set_c, &err_set);
300 }
