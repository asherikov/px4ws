417// Read a setting/variable name; return empty string if no valid name
418//
419// If env is set, dashes/dots are not allowed within names. They are not typically allowed by shells
420// and they interfere with substitution patterns.
421inline string read_config_name(string_iterator & i, string_iterator end, bool env = false, bool *num = nullptr) noexcept
422{
423 using std::locale;
424 using std::ctype;
425 using std::use_facet;
426
427 // To avoid the horror of locales, we'll use the classic facet only, to identify digits, control
428 // characters and punctuation. (Unless something is totally crazy, we are talking about ASCII or
429 // a superset of it, but using the facet allows us to avoid that assumption). However, we're only
430 // working with "narrow" char type so accuracy is limited. In general, that's not going to matter
431 // much, but may allow certain unicode punctuation characters to be used as part of a name for example.
432 const ctype<char> & facet = use_facet<ctype<char>>(locale::classic());
433
434 string rval;
435
436 // For environment lookups, integers are valid names (particularly for argument)
437 if (env && facet.is(ctype<char>::digit, *i)) {
438 while (facet.is(ctype<char>::digit, *i)) {
439 rval += *i;
440 ++i;
441 }
442 if (num) {
443 *num = true;
444 }
445 return rval;
446 }
447
448 // Don't allow empty name, numeric digit, or dash/dot at start of setting name
449 if (i == end || (*i == '-' || *i == '.' || facet.is(ctype<char>::digit, *i))) {
450 return {};
451 }
452
453 // Within the setting name, allow dash and dot unless parsing envvar name
454 // also allow any non-control, non-punctuation non-space character.
455 while (i != end && (((*i == '-' || *i == '.') && !env) || *i == '_'
456 || (!facet.is(ctype<char>::cntrl, *i) && !facet.is(ctype<char>::punct, *i)
457 && !facet.is(ctype<char>::space, *i)))) {
458 rval += *i;
459 ++i;
460 }
461 return rval;
462}

464// Read a setting value, with an assignment or plus-assignment (append).
465//
466// In general a setting value is a single-line string. It may contain multiple parts
467// separated by white space (which is normally collapsed). A hash mark - # - denotes
468// the end of the value and the beginning of a comment (it should be preceded by
469// whitespace).
470//
471// Part of a value may be quoted using double quote marks, which prevents collapse
472// of whitespace and interpretation of most special characters (the quote marks will
473// not be considered part of the value). A backslash can precede a character (such
474// as '#' or '"' or another backslash) to remove its special meaning. Newline
475// characters are not allowed in values and cannot be quoted.
476//
477// This function expects the string to be in an ASCII-compatible encoding (the "classic" locale).
478//
479// Throws service_description_exc (with service name unset) on error.
480//
481// Parameters:
482// setting_val - the (current) setting value, to be modified by the call
483// operation - the operation- assign or append
484// input_pos - the file and line number on which the setting appears (starts), used for error
485// reporting
486// i - reference to string iterator through the line (updated to end of setting on
487// return)
488// end - iterator at end of line (not including newline character if any)
489// part_positions - list of pair<unsigned,unsigned> to which the position of each setting value
490// part will be added as [start,end). May be null.
491inline void read_setting_value(std::string &setting_val, setting_op_t operation,
492 file_pos_ref input_pos, string_iterator &i, string_iterator end,
493 std::list<std::pair<unsigned,unsigned>> *part_positions = nullptr,
494 char delimiter = ' ')
495{
496 using std::locale;
497 using std::isspace;
498
499 unsigned &line_num = input_pos.get_line_num_ref();
500
501 i = skipwsln(i, end, line_num);
502
503 if (operation == setting_op_t::PLUSASSIGN) {
504 // Ensure that values are correctly delimited. This is usually only for debugging
505 // niceness as for commands where this is mostly used the offsets actually delimit
506 // the components, but e.g. for capabilities (comma-separated) it matters more.
507 if (!setting_val.empty()) {
508 setting_val += delimiter;
509 }
510 }
511 else {
512 setting_val.clear();
513 if (part_positions != nullptr) {
514 part_positions->clear();
515 }
516 }
517
518 bool new_part = true;
519 int part_start;
520
521 while (i != end) {
522 char c = *i;
523 if (c == '\"') {
524 if (new_part) {
525 part_start = setting_val.length();
526 new_part = false;
527 }
528 // quoted string
529 ++i;
530 while (i != end) {
531 c = *i;
532 if (c == '\"') break;
533 else if (c == '\\') {
534 // A backslash escapes the following character.
535 c = *(++i);
536 setting_val += c;
537 }
538 else {
539 setting_val += c;
540 }
541 ++i;
542 }
543 if (i == end) {
544 // String wasn't terminated
545 throw service_description_exc(input_pos, "unterminated quoted string");
546 }
547 }
548 else if (c == '\\') {
549 if (new_part) {
550 part_start = setting_val.length();
551 new_part = false;
552 }
553 // A backslash escapes the next character
554 setting_val += *(++i);
555 }
556 else if (isspace(c, locale::classic())) {
557 if (!new_part && part_positions != nullptr) {
558 part_positions->emplace_back(part_start, setting_val.length());
559 new_part = true;
560 }
561 i = skipcomment(i, end, line_num);
562 if (i == end) break;
563 setting_val += ' '; // collapse ws to a single space
564 continue;
565 }
566 else if (c == '#') {
567 // Possibly intended a comment; we require leading whitespace to reduce occurrence of accidental
568 // comments in setting values.
569 throw service_description_exc(input_pos, "hashmark (`#') comment must be separated from setting value by whitespace");
570 }
571 else {
572 if (new_part) {
573 part_start = setting_val.length();
574 new_part = false;
575 }
576 setting_val += c;
577 }
578 ++i;
579 }
580
581 // Got to end:
582 if (!new_part && part_positions != nullptr) {
583 part_positions->emplace_back(part_start, setting_val.length());
584 }
585}

659// Parse a userid parameter which may be a numeric user ID or a username. If a name, the
660// userid is looked up via the system user database (getpwnam() function). In this case,
661// the associated group is stored in the location specified by the group_p parameter if
662// it is not null.
663inline uid_t parse_uid_param(file_pos_ref input_pos, const std::string &param,
664 const std::string &service_name, const char *setting_name, gid_t *group_p)
665{
666 const char * uid_err_msg = "specified user id contains invalid numeric characters "
667 "or is outside allowed range.";
668
669 // Could be a name or a numeric id. But we should assume numeric first, just in case
670 // a user manages to give themselves a username that parses as a number.
671 std::size_t ind = 0;
672 try {
673 // POSIX does not specify whether uid_t is a signed or unsigned type, but regardless
674 // is is probably safe to assume that valid values are positive. We'll also assert
675 // that the value range fits within "unsigned long long" since it seems unlikely
676 // that would ever not be the case.
677 static_assert((uintmax_t)std::numeric_limits<uid_t>::max()
678 <= (uintmax_t)std::numeric_limits<unsigned long long>::max(), "uid_t is too large");
679 unsigned long long v = std::stoull(param, &ind, 0);
680 if (v > static_cast<unsigned long long>(std::numeric_limits<uid_t>::max())
681 || ind != param.length()) {
682 throw service_description_exc(service_name, std::string(setting_name) + ": " + uid_err_msg, input_pos);
683 }
684 return v;
685 }
686 catch (std::out_of_range &exc) {
687 throw service_description_exc(service_name, uid_err_msg, input_pos);
688 }
689 catch (std::invalid_argument &exc) {
690 // Ok, so it doesn't look like a number: proceed...
691 }
692
693 errno = 0;
694 struct passwd * pwent = getpwnam(param.c_str());
695 if (pwent == nullptr) {
696 // Maybe an error, maybe just no entry.
697 if (errno == 0) {
698 throw service_description_exc(service_name, std::string(setting_name) + ": specified user \"" + param
699 + "\" does not exist in system database.", input_pos);
700 }
701 else {
702 throw service_description_exc(service_name, std::string("error accessing user database: ")
703 + strerror(errno), input_pos);
704 }
705 }
706
707 if (group_p) {
708 *group_p = pwent->pw_gid;
709 }
710
711 return pwent->pw_uid;
712}

719
720 // Could be a name or a numeric id. But we should assume numeric first, just in case
721 // a user manages to give themselves a username that parses as a number.
722 std::size_t ind = 0;
723 try {
724 // POSIX does not specify whether uid_t is an signed or unsigned, but regardless
725 // is is probably safe to assume that valid values are positive. We'll also assume
726 // that the value range fits with "unsigned long long" since it seems unlikely
727 // that would ever not be the case.
728 static_assert((uintmax_t)std::numeric_limits<gid_t>::max()
729 <= (uintmax_t)std::numeric_limits<unsigned long long>::max(), "gid_t is too large");

737 }
738 catch (std::out_of_range &exc) {
739 throw service_description_exc(service_name, std::string(setting_name) + ": " + gid_err_msg,
740 input_pos);
741 }
742 catch (std::invalid_argument &exc) {
743 // Ok, so it doesn't look like a number: proceed...
744 }
745
746 errno = 0;
747 struct group * grent = getgrnam(param.c_str());
748 if (grent == nullptr) {
749 // Maybe an error, maybe just no entry.
750 if (errno == 0) {
751 throw service_description_exc(service_name, std::string(setting_name) + ": specified group \""
752 + param + "\" does not exist in system database.", input_pos);
753 }
754 else {
755 throw service_description_exc(service_name, std::string("error accessing group database: ")
756 + strerror(errno), input_pos);
757 }
758 }

760 return grent->gr_gid;

781// Parse a time, specified as a decimal number of seconds (with optional fractional component after decimal
782// point or decimal comma).
783inline void parse_timespec(file_pos_ref input_pos, const std::string &paramval,
784 const std::string &servicename, const char * paramname, timespec &ts)
785{
786 decltype(ts.tv_sec) isec = 0;
787 decltype(ts.tv_nsec) insec = 0;
788 auto max_secs = std::numeric_limits<decltype(isec)>::max() / 10;
789 auto len = paramval.length();
790 decltype(len) i;
791 for (i = 0; i < len; i++) {
792 char ch = paramval[i];
793 if (ch == '.' || ch == ',') {
794 i++;
795 break;
796 }
797 if (ch < '0' || ch > '9') {
798 throw service_description_exc(servicename, std::string("bad value for ") + paramname, input_pos);
799 }
800 // check for overflow
801 if (isec >= max_secs) {
802 throw service_description_exc(servicename, std::string("too-large value for ") + paramname, input_pos);
803 }
804 isec *= 10;
805 isec += ch - '0';
806 }
807 decltype(insec) insec_m = 100000000; // 10^8
808 for ( ; i < len; i++) {
809 char ch = paramval[i];
810 if (ch < '0' || ch > '9') {
811 throw service_description_exc(servicename, std::string("bad value for ") + paramname, input_pos);
812 }
813 insec += (ch - '0') * insec_m;
814 insec_m /= 10;
815 }
816 ts.tv_sec = isec;
817 ts.tv_nsec = insec;
818}

880// Parse resource limits setting (can specify both hard and soft limit).
881inline void parse_rlimit(const std::string &line, file_pos_ref input_pos,
882 const std::string &service_name, const char *param_name, service_rlimits &rlimit)
883{
884 // Examples:
885 // 4:5 - soft:hard limits both set
886 // 4:- soft set, hard set to unlimited
887 // 4: soft set, hard limit unchanged
888 // 4 soft and hard limit set to same limit
889
890 if (line.empty()) {
891 throw service_description_exc(service_name, std::string(param_name) + ": bad value.", input_pos);
892 }
893
894 const char *cline = line.c_str();
895 rlimit.hard_set = rlimit.soft_set = false;
896
897 try {
898 const char * index = cline;
899 errno = 0;
900 if (cline[0] != ':') {
901 rlimit.soft_set = true;
902 if (cline[0] == '-') {
903 rlimit.limits.rlim_cur = RLIM_INFINITY;
904 index = cline + 1;
905 }
906 else {
907 errno = 0;
908 char *nindex;
909 unsigned long long limit = std::strtoull(cline, &nindex, 0);
910 index = nindex;
911 if (errno == ERANGE || limit > std::numeric_limits<rlim_t>::max()) throw std::out_of_range("");
912 if (index == cline) throw std::invalid_argument("");
913 rlimit.limits.rlim_cur = limit;
914 }
915
916 if (*index == 0) {
917 rlimit.hard_set = true;
918 rlimit.limits.rlim_max = rlimit.limits.rlim_cur;
919 return;
920 }
921
922 if (*index != ':') {
923 throw service_description_exc(service_name, std::string(param_name) + ": bad value.", input_pos);
924 }
925 }
926
927 index++;
928 if (*index == 0) return;
929
930 rlimit.hard_set = true;
931
932 if (*index == '-') {
933 rlimit.limits.rlim_max = RLIM_INFINITY;
934 if (index[1] != 0) {
935 throw service_description_exc(service_name, std::string(param_name) + ": bad value.", input_pos);
936 }
937 }
938 else {
939 const char *hard_start = index;
940 char *nindex;
941 errno = 0;
942 unsigned long long limit = std::strtoull(hard_start, &nindex, 0);
943 index = nindex;
944 if (errno == ERANGE || limit > std::numeric_limits<rlim_t>::max()) throw std::out_of_range("");
945 if (index == hard_start) throw std::invalid_argument("");
946 rlimit.limits.rlim_max = limit;
947 }
948 }
949 catch (std::invalid_argument &exc) {
950 throw service_description_exc(service_name, std::string(param_name) + ": bad value.", input_pos);
951 }
952 catch (std::out_of_range &exc) {
953 throw service_description_exc(service_name, std::string(param_name) + ": too-large value.", input_pos);
954 }
955}

962// Process an opened service file, line by line.
963// Parameters:
964// name - the service name
965// service_input - the service file input stream (stack)
966// process_line_func - a function to process settings, of the form:
967// void(string &line, file_pos_ref position, string &setting, string_iterator i,
968// string_iterator end)
969// Called with:
970// line - the complete line, excluding newline character
971// position - the input position (file_pos_ref)
972// setting - the setting name, from the beginning of the line
973// op - the operation (setting_op_t)
974// i - iterator at the beginning of the setting value
975// end - iterator marking the end of the line
976// argval - the service argument, or nullptr if none
977// resolve_var - a functor to resolve variable values.
978// Accepts: const std::string & - the variable name.
979// Returns char * - the resolved value.
980// Throws:
981// service load exceptions or I/O exceptions if enabled on stream.
982template <typename T, typename resolve_var_t>
983void process_service_file(string name, file_input_stack &service_input, T process_line_func,
984 const char *argval, const resolve_var_t &resolve_var)
985{
986 string line;
987
988 while (service_input.getline(line)) {
989
990 unsigned line_num = service_input.current_line();
991
992 if (line.empty()) continue;
993 while (line.back() == '\\') {
994 // Check for double-backslash, i.e. backslash escaped with backslash. We actually need
995 // to check for any even number.
996 bool is_odd_number = true;
997 auto r = std::next(line.rbegin());
998 auto re = line.rend();
999 while (r != re) {
1000 if (*r != '\\') break;
1001 is_odd_number = !is_odd_number;
1002 ++r;
1003 }
1004
1005 if (!is_odd_number) break;
1006
1007 string nextline;
1008 string::iterator j;
1009 string::iterator endnext;
1010
1011 line.back() = '\n';
1012 if (!service_input.getline_same_file(nextline)) {
1013 throw service_description_exc(service_input,
1014 "end-of-file follows backslash escape character (`\\')");
1015 }
1016
1017 j = nextline.begin();
1018 endnext = nextline.end();
1019 j = skipws(j, endnext);
1020 if (j == nextline.begin()) {
1021 throw service_description_exc(service_input,
1022 "line following line-continuation backslash (`\\') "
1023 "does not begin with whitespace character");
1024 }
1025 line.append(nextline);
1026 }
1027
1028 string::iterator i = line.begin();
1029 string::iterator end = line.end();
1030
1031 i = skipwsln(i, end, line_num);
1032 if (i != end) {
1033 if (*i == '#') continue; // comment without setting
1034
1035 if (*i == '@') {
1036 // meta command
1037 ++i;
1038 string meta_cmd = read_config_name(i, end);
1039 bool is_include_opt = (meta_cmd == "include-opt");
1040 if (is_include_opt || meta_cmd == "include") {
1041 // @include-opt or @include
1042 file_pos_ref input_pos { service_input.current_file_name(), line_num };
1043 std::string include_name = read_include_path(name, meta_cmd, input_pos, i, end, argval, resolve_var);
1044
1045 std::ifstream file(include_name);
1046 file.exceptions(std::ios::badbit);
1047 if (file) {
1048 service_input.push(include_name, std::move(file));
1049 }
1050 else {
1051 if (!is_include_opt || errno != ENOENT) {
1052 throw service_load_exc(name, include_name + ": cannot open: " + strerror(errno));
1053 }
1054 }
1055 }
1056 else {
1057 file_pos_ref input_pos { service_input.current_file_name(), line_num };
1058 throw service_description_exc(name, "bad meta command", input_pos);
1059 }
1060
1061 continue;
1062 }
1063
1064 string setting = read_config_name(i, end);
1065 i = skipwsln(i, end, line_num);
1066
1067 setting_op_t setting_op;
1068
1069 bool bad_line = false;
1070 if (setting.empty()) {
1071 bad_line = true;
1072 }
1073 else if (*i == '=') {
1074 setting_op = setting_op_t::ASSIGN;
1075 }
1076 else if (*i == ':') {
1077 setting_op = setting_op_t::COLON;
1078 }
1079 else if (*i == '+') {
1080 ++i;
1081 if (i != end && *i == '=') {
1082 // "+=" is allowed
1083 setting_op = setting_op_t::PLUSASSIGN;
1084 }
1085 else {
1086 bad_line = true;
1087 }
1088 }
1089 else {
1090 bad_line = true;
1091 }
1092
1093 if (bad_line) {
1094 throw service_description_exc(name, "badly formed line.", service_input);
1095 }
1096
1097 i = skipwsln(++i, end, line_num);
1098
1099 file_pos_ref fpr { service_input.current_file_name(), line_num };
1100 process_line_func(line, fpr, setting, setting_op, i, end);
1101 }
1102 }

1119// Substitute variable references in a value with their values. Specified offsets must give
1120// the location of separate arguments after word splitting and are adjusted appropriately.
1121// If you simply wish to substitute all variables in the given string, pass an offsets list
1122// containing one pair with the string's bounds (0, size). '$$' resolves to a single '$'.
1123// Parameters:
1124// setting_name - the name of the setting from which the value comes (used for errors)
1125// line (by ref.) - the complete setting value line (on call), with variable references
1126// replaced on return.
1127// offsets (by ref.) - the position as [start,end) for each token in the line (must be
1128// valid on call, is updated on return).
1129// var_resolve - a functor to resolve variable values.
1130// Accepts: const std::string & - the variable name.
1131// Returns char * - the resolved value.
1132// argval - the service argument value, if any (nullptr if none).
1133// Throws:
1134// service_description_exc - if a $-substitution is ill-formed
1135// std::length_error - if string length limit exceeded or command line is too long
1136// std::bad_alloc - on allocation failure
1137template <typename T>
1138static void value_var_subst(const char *setting_name, std::string &line,
1139 std::list<std::pair<unsigned,unsigned>> &offsets, const T &var_resolve, const char *argval)
1140{
1141 auto dindx = line.find('$');
1142 if (dindx == string::npos) {
1143 return;
1144 }
1145
1146 if (line.length() > (size_t)std::numeric_limits<int>::max()) {
1147 // (avoid potential for overflow later)
1148 throw std::length_error("string too long");
1149 }
1150
1151 auto i = offsets.begin();
1152 unsigned xpos = 0; // position to copy from in original line
1153 std::string r_line;
1154 int offadj = 0;
1155
1156 while (i != offsets.end()) {
1157
1158 i->first += offadj; // don't adjust end yet
1159
1160 // inhibit_collapse is set if we process anything which may be empty but shouldn't collapse
1161 // to "no argument"
1162 bool inhibit_collapse = false;
1163 bool do_collapse = false;
1164
1165 while (i->second > dindx) {
1166 r_line.append(line, xpos, dindx - xpos); // copy unmatched part
1167 if (line[dindx + 1] == '$') {
1168 // double dollar, collapse to single
1169 r_line += '$';
1170 xpos = dindx + 2;
1171 --offadj;
1172 }
1173 else {
1174 // variable
1175 auto token_end = std::next(line.begin(), i->second);
1176 auto spos = dindx + 1;
1177 bool wsplit = line[spos] == '/';
1178 if (wsplit) ++spos;
1179 bool brace = line[spos] == '{';
1180 if (brace) ++spos;
1181 auto j = std::next(line.begin(), spos);
1182 // may be a service argument
1183 bool is_arg = false;
1184 // read environment variable name
1185 string name = read_config_name(j, token_end, true, &is_arg);
1186 if (name.empty()) {
1187 throw service_description_exc(setting_name, "invalid/missing variable name after '$'");
1188 }
1189 else if (is_arg && name != "1") {
1190 // only one arg is supported
1191 throw service_description_exc(setting_name, "only one service argument may be present");
1192 }
1193 char altmode = '\0';
1194 bool colon = false;
1195 auto altbeg = j, altend = j;
1196 if (brace) {
1197 /* ${foo+val}, ${foo-val}, ${foo:+val}, ${foo:-val} */
1198 if (*j == ':') {
1199 colon = true;
1200 ++j;
1201 if (*j != '+' && *j != '-') {
1202 throw service_description_exc(setting_name, "invalid syntax in variable substitution");
1203 }
1204 }
1205 if (*j == '+' || *j == '-') {
1206 altmode = *j;
1207 altbeg = ++j;
1208 while (j != token_end && *j != '}') {
1209 ++j;
1210 }
1211 altend = j;
1212 }
1213 if (*j != '}') {
1214 throw service_description_exc(setting_name, "unmatched '{' in variable substitution");
1215 }
1216 ++j;
1217 }
1218 size_t line_len_before = r_line.size();
1219 string_view resolved_vw;
1220 auto *resolved = is_arg ? argval : var_resolve(name);
1221 if (resolved) {
1222 resolved_vw = resolved;
1223 }
1224 /* apply shell-like substitutions */
1225 if (altmode == '-') {
1226 if (!resolved || (colon && !*resolved)) {
1227 resolved_vw = {line.c_str() + (altbeg - line.begin()), (size_t)(altend - altbeg)};
1228 }
1229 }
1230 else if (altmode == '+') {
1231 if (resolved && (!colon || *resolved)) {
1232 resolved_vw = {line.c_str() + (altbeg - line.begin()), (size_t)(altend - altbeg)};
1233 }
1234 }
1235 else if (is_arg && !argval) {
1236 // $1 and ${1} is special in that it must be set or it is an error
1237 // however, we want the more complex syntaxes for conditional substitution
1238 throw service_description_exc(setting_name, "missing value in argument substitution");
1239 }
1240
1241 xpos = j - line.begin();
1242 int name_len = xpos - dindx;
1243 offadj -= name_len;
1244
1245 if (!wsplit) {
1246 inhibit_collapse = true;
1247 do_collapse = false;
1248 if (!resolved_vw.empty()) {
1249 r_line.append(resolved_vw.data(), resolved_vw.length());
1250 }
1251 }
1252 else {
1253 // Must perform word splitting. Find first whitespace:
1254 auto r_vw_beg = resolved_vw.data();
1255 auto r_vw_end = r_vw_beg + resolved_vw.length();
1256 const char *wsp = findws(r_vw_beg, r_vw_end);
1257
1258 // If we have whitespace, append up to that whitespace and then split:
1259 while (wsp != r_vw_end) {
1260 if (wsp != r_vw_beg) {
1261 r_line.append(r_vw_beg, wsp - r_vw_beg);
1262 }
1263
1264 auto orig_i_second = i->second;
1265
1266 size_t line_len_after = r_line.size();
1267 if (i->first == line_len_after && !inhibit_collapse) {
1268 // whitespace at the start of the word; just trim it
1269 goto next_section;
1270 }
1271
1272 // Break here:
1273 i->second = r_line.length();
1274
1275 r_line += ' ';
1276 ++line_len_after;
1277
1278 if (line_len_after > (size_t)std::numeric_limits<int>::max()) {
1279 // (avoid potential overflow)
1280 throw service_description_exc(setting_name, "value too long (after substitution)");
1281 }
1282
1283 // Create new argument from split:
1284 i = offsets.insert(std::next(i), {r_line.length(), orig_i_second});
1285 offadj += (int)line_len_after - (int)line_len_before;
1286 line_len_before = r_line.size();
1287
1288 // Find the next break, if any:
1289 next_section:
1290 r_vw_beg = skipws(wsp, r_vw_end);
1291 wsp = findws(r_vw_beg, r_vw_end);
1292 inhibit_collapse = false;
1293 }
1294
1295 if (r_vw_beg != r_vw_end) {
1296 r_line.append(r_vw_beg, r_vw_end - r_vw_beg);
1297 }
1298
1299 do_collapse = !inhibit_collapse;
1300 }
1301
1302 size_t line_len_after = r_line.size();
1303
1304 if (line_len_after > (size_t)std::numeric_limits<int>::max()) {
1305 // (avoid potential overflow)
1306 throw service_description_exc(setting_name, "value too long (after substitution)");
1307 }
1308
1309 offadj += (int)line_len_after - (int)line_len_before;
1310 }
1311
1312 dindx = line.find('$', xpos);
1313 }
1314
1315 i->second += offadj;
1316
1317 if (do_collapse && i->first == i->second) {
1318 i = offsets.erase(i);
1319 }
1320 else {
1321 ++i;
1322 }
1323
1324 while (i != offsets.end() && i->second < dindx) {
1325 i->first += offadj;
1326 i->second += offadj;
1327 ++i;
1328 }
1329 }
1330
1331 r_line.append(line, xpos); // copy final unmatched part
1332 line = std::move(r_line);
1333}

1364template <typename resolve_var_t>
1365inline string read_include_path(string const &svcname, string const &meta_cmd, file_pos_ref input_pos,
1366 string_iterator &i, string_iterator end, const char *argval, const resolve_var_t &resolve_var)
1367{
1368 string rval;
1369 std::list<std::pair<unsigned,unsigned>> parts;
1370
1371 read_setting_value(rval, setting_op_t::ASSIGN, input_pos, i, end, &parts);
1372 if (parts.size() != 1) {
1373 throw service_description_exc(svcname, "'@" + meta_cmd + "' requires a single argument", input_pos);
1374 }
1375
1376 std::list<std::pair<unsigned,unsigned>> offsets;
1377 offsets.emplace_back(0, rval.size());
1378 value_var_subst(meta_cmd.c_str(), rval, offsets, resolve_var, argval);
1379 return rval;
1380}

1471 // Finalise settings (after processing all setting lines), perform some basic sanity checks
1472 // and optionally some additional lint checks.
1473 //
1474 // Template parameters:
1475 // propagate_sde - whether to propagate service description errors (if false, they are
1476 // reported via report_err)
1477 // (remaining template parameters should be inferred)
1478 // Parameters:
1479 // report_error - functor to report any errors
1480 // service_arg - service argument, if any (may be null)
1481 // envmap - environment variables
1482 // report_line - functor to report lint (default: don't report)
1483 // var_subst - functor to resolve environment variable values
1484 // Throws:
1485 // service_description_exc, bad_alloc
1486 //
1487 // Note: we have the do_report_lint parameter to prevent code (and strings) being emitted for
1488 // lint checks even when the dummy_lint function is used. (Ideally the compiler would optimise
1489 // them away).
1490 //
1491 template <bool propagate_sde = false, typename T, typename U, typename V,
1492 bool do_report_lint = !std::is_same<U, decltype(dummy_lint)>::value>

1494 const V &var_subst)

1497 || service_type == service_type_t::SCRIPTED) {
1498 if (command.empty()) {
1499 report_error("'command' setting not specified.");
1500 }
1501 }
1502
1503 if (do_report_lint && (service_type == service_type_t::INTERNAL
1504 || service_type == service_type_t::TRIGGERED)) {
1505 if (!command.empty()) {
1506 report_lint("'command' specified, but ignored for the specified (or default) service type.");
1507 }
1508 if (!stop_command.empty()) {
1509 report_lint("'stop-command' specified, but ignored for the specified (or default) service type.");
1510 }
1511 if (!working_dir.empty()) {
1512 report_lint("'working-dir' specified, but ignored for the specified (or default) service type.");
1513 }
1514 #if SUPPORT_CGROUPS
1515 if (!run_in_cgroup.empty()) {
1516 report_lint("'run-in-cgroup' specified, but ignored for the specified (or default) service type.");
1517 }
1518 #endif
1519 #if SUPPORT_CAPABILITIES
1520 if (capabilities.get()) {
1521 report_lint("'capabilities' specified, but ignored for the specified (or default) service type.");
1522 }
1523 if (secbits.get()) {
1524 report_lint("'securebits' specified, but ignored for the specified (or default) service type.");
1525 }
1526 #endif
1527 if (run_as_uid != (uid_t)-1) {
1528 report_lint("'run-as' specified, but ignored for the specified (or default) service type.");
1529 }
1530 if (!socket_path.empty()) {
1531 report_lint("'socket-listen' specified, but ignored for the specified (or default) service type'.");
1532 }
1533 #if USE_UTMPX
1534 if (inittab_id[0] != 0 || inittab_line[0] != 0) {
1535 report_lint("'inittab_line' or 'inittab_id' specified, but ignored for the specified (or default) service type.");
1536 }
1537 #endif
1538 if (onstart_flags.signal_process_only || onstart_flags.start_interruptible) {
1539 report_lint("signal options were specified, but ignored for the specified (or default) service type.");
1540 }
1541 if (onstart_flags.pass_cs_fd) {
1542 report_lint("option 'pass_cs_fd' was specified, but ignored for the specified (or default) service type.");
1543 }
1544 if (onstart_flags.skippable) {
1545 report_lint("option 'skippable' was specified, but ignored for the specified (or default) service type.");
1546 }
1547 #if SUPPORT_CAPABILITIES
1548 if (onstart_flags.no_new_privs) {
1549 report_lint("option 'no_new_privs' was specified, but ignored for the specified (or default) service type.");
1550 }
1551 #endif
1552 if (log_type != log_type_id::NONE) {
1553 report_lint("option 'log_type' was specified, but ignored for the specified (or default) service type.");
1554 }
1555 if (nice_is_set) {
1556 report_lint("option 'nice' was specified, but ignored for the specified (or default) service type.");

1558 #if SUPPORT_IOPRIO
1559 if (ioprio >= 0) {
1560 report_lint("option 'ioprio' was specified, but ignored for the specified (or default) service type.");

1577 }

1579 if (service_type == service_type_t::BGPROCESS) {

1600 str_offsets.emplace_back(0, setting_value.size());

1622 if (logfile_gid == (gid_t)-1) logfile_gid = getgid();

1624 // (if we do support supplementary groups, run_as_gid==-1 means "use the user groups including
1625 // supplementary groups" whereas run_as_gid==X means "use group X with no supplementary groups").
1626#if USE_INITGROUPS
1627 if (run_as_uid_gid == (gid_t)-1 && run_as_gid == (gid_t)-1) {
1628 // run-as-UID specified by number; set run_as_gid to inhibit initialisation of
1629 // supplementary groups
1630 run_as_gid = getgid();
1631 }
1632#else // !USE_INITGROUPS
1633 if (run_as_gid == (gid_t)-1) run_as_gid = run_as_uid_gid;
1634#endif
1635
1636 if (log_type != log_type_id::LOGFILE) {
1637 logfile.clear();
1638 }
1639
1640 if (!value(service_type).is_in(service_type_t::PROCESS, service_type_t::BGPROCESS)) {

1644 }

1648 // and optionally some additional lint checks. Resolve variables from provided env_map.
1649 // See finalise() above.
1650 //
1651 // Throws:
1652 // service_description_exc
1653 //
1654 template <bool propagate_sde = false, typename T, typename U = decltype(dummy_lint),
1655 typename V = decltype(resolve_env_var),
1656 bool do_report_lint = !std::is_same<U, decltype(dummy_lint)>::value>
1657 void finalise(T &report_error, environment::env_map const &envmap, const char *argval,
1658 U &report_lint = dummy_lint, V &var_subst = resolve_env_var)
1659 {
1660 auto do_var_subst = [&](const std::string &name) {
1661 return var_subst(name, envmap);
1662 };
1663 finalise<propagate_sde, T, U, decltype(do_var_subst), do_report_lint>(report_error,
1664 argval, report_lint, do_var_subst);
1665 }
1666};

1668// Process a service description line. In general, parse the setting value and record the parsed
1669// value in a service settings wrapper object. Errors will be reported via service_description_exc
1670// exception.
1671//
1672// Type parameters:
1673// settings_wrapper : wrapper for service settings
1674// load_service_t : type of load service function/lambda (see below)
1675// process_dep_dir_t : type of process_dep_dir function/lambda (see below)
1676//
1677// Parameters:
1678// settings : wrapper object for service settings
1679// name : name of the service being processed
1680// service_arg : service argument, if any (may be null)
1681// line : the current line of the service description file
1682// input_pos : the current input position (for error reporting)
1683// setting : the name of the setting (from the beginning of line)
1684// setting_op : the operator specified after the setting name
1685// i : iterator at beginning of setting value (including whitespace)
1686// end : iterator at end of line
1687// load_service : function to load a service
1688// arguments: const char *service_name
1689// return: a value that can be used (with a dependency type) to construct a
1690// dependency in the 'depends' vector within the 'settings' object
1691// process_dep_dir : function to process a dependency directory
1692// arguments: decltype(settings.depends) &dependencies
1693// const string &waitsford - directory as specified in parameter
1694// dependency_type dep_type - type of dependency to add
1695// Throws:
1696// service_description_exc, std::bad_alloc, std::length_error (string too long; unlikely)
1697template <typename settings_wrapper,
1698 typename load_service_t,
1699 typename process_dep_dir_t,
1700 typename lookup_var_t = decltype(null_resolve_env_var)>
1701void process_service_line(settings_wrapper &settings, ::string_view name, const char *service_arg,
1702 string &line, file_pos_ref input_pos, string &setting, setting_op_t setting_op,
1703 string::iterator &i, string::iterator &end, load_service_t load_service,
1704 process_dep_dir_t process_dep_dir, const lookup_var_t &lookup_var = null_resolve_env_var)
1705{
1706 // find the setting:
1707 setting_details *details = all_settings;
1708 while (setting != details->setting_str) {
1709 details++;
1710 if (details->setting_id == setting_id_t::LAST) {
1711 throw service_description_exc(name, "unknown setting: '" + setting + "'.", input_pos);
1712 }
1713 }
1714
1715 if (setting_op == setting_op_t::PLUSASSIGN && !details->supp_plus_assign) {
1716 throw service_description_exc(name, "cannot use '+=' with setting '" + setting + "'", input_pos);
1717 }
1718
1719 switch (details->setting_id) {
1720 case setting_id_t::COMMAND:
1721 read_setting_value(settings.command, setting_op, input_pos, i, end, &settings.command_offsets);
1722 break;
1723 case setting_id_t::WORKING_DIR:
1724 settings.working_dir = read_setting_value(input_pos, i, end, nullptr);
1725 break;
1726 case setting_id_t::ENV_FILE:
1727 settings.env_file = read_value_resolved(setting.c_str(), input_pos, i, end,
1728 service_arg, lookup_var);
1729 break;
1730 #if SUPPORT_CGROUPS
1731 case setting_id_t::RUN_IN_CGROUP:
1732 settings.run_in_cgroup = read_setting_value(input_pos, i, end, nullptr);
1733 break;
1734 #endif
1735 #if SUPPORT_CAPABILITIES
1736 case setting_id_t::CAPABILITIES:
1737 {
1738 std::string capabilities_str;
1739 read_setting_value(capabilities_str, setting_op, input_pos, i, end, nullptr, ',');
1740
1741 cap_iab_wrapper cap_iab(capabilities_str);
1742 if (!cap_iab.get()) {
1743 throw service_description_exc(name, "invalid capabilities: " + capabilities_str,
1744 "capabilities", input_pos);
1745 }
1746 settings.capabilities = std::move(cap_iab);
1747 break;
1748 }
1749 case setting_id_t::SECUREBITS:
1750 {
1751 std::list<std::pair<unsigned,unsigned>> indices;
1752 string onstart_cmds = read_setting_value(input_pos, i, end, &indices);
1753 // plain assignment will clear, while append will add more
1754 if (setting_op != setting_op_t::PLUSASSIGN) {
1755 settings.secbits.clear();
1756 }
1757 for (auto indexpair : indices) {
1758 ::string_view secbit_txt { onstart_cmds.data() + indexpair.first,
1759 indexpair.second - indexpair.first };
1760 if (secbit_txt == "keep-caps") {
1761 settings.secbits.set(SECBIT_KEEP_CAPS);
1762 }
1763 else if (secbit_txt == "keep-caps-locked") {
1764 settings.secbits.set(SECBIT_KEEP_CAPS_LOCKED);
1765 }
1766 else if (secbit_txt == "no-setuid-fixup") {
1767 settings.secbits.set(SECBIT_NO_SETUID_FIXUP);
1768 }
1769 else if (secbit_txt == "no-setuid-fixup-locked") {
1770 settings.secbits.set(SECBIT_NO_SETUID_FIXUP_LOCKED);
1771 }
1772 else if (secbit_txt == "noroot") {
1773 settings.secbits.set(SECBIT_NOROOT);
1774 }
1775 else if (secbit_txt == "noroot-locked") {
1776 settings.secbits.set(SECBIT_NOROOT_LOCKED);
1777 }
1778 else {
1779 throw service_description_exc(name, "unknown securebits flag: " + secbit_txt,
1780 "securebits", input_pos);
1781 }
1782 }
1783 break;
1784 }
1785 #endif
1786 case setting_id_t::NICE:
1787 {
1788 string nice_str = read_setting_value(input_pos, i, end);
1789 settings.nice_is_set = true;
1790 settings.nice = (int)parse_snum_param(input_pos, nice_str, name,
1791 std::numeric_limits<int>::min() / 2, std::numeric_limits<int>::max() / 2);
1792 break;
1793 }
1794 #if SUPPORT_IOPRIO
1795 case setting_id_t::IOPRIO:
1796 {
1797 string ioprio_str = read_setting_value(input_pos, i, end);
1798 if (ioprio_str == "none") {
1799 settings.ioprio = 0;
1800 }
1801 else if (starts_with(ioprio_str, "realtime:")) {
1802 auto nval = parse_unum_param(input_pos, ioprio_str.substr(9 /* len 'realtime:' */), name, 7);
1803 settings.ioprio = IOPRIO_PRIO_VALUE(IOPRIO_CLASS_RT, nval);
1804 }
1805 else if (starts_with(ioprio_str, "best-effort:")) {
1806 auto nval = parse_unum_param(input_pos, ioprio_str.substr(12 /* len 'best-effort:' */), name, 7);
1807 settings.ioprio = IOPRIO_PRIO_VALUE(IOPRIO_CLASS_BE, nval);
1808 }
1809 else if (ioprio_str == "idle") {
1810 settings.ioprio = 3 << 13;
1811 settings.ioprio = IOPRIO_PRIO_VALUE(IOPRIO_CLASS_IDLE, 0);
1812 }
1813 else {
1814 throw service_description_exc(name, "invalid value for ioprio: " + ioprio_str,
1815 "ioprio", input_pos);
1816 }
1817 break;
1818 }
1819 #endif
1820 #if SUPPORT_OOM_ADJ
1821 case setting_id_t::OOM_SCORE_ADJ:
1822 {
1823 string oom_adj_str = read_setting_value(input_pos, i, end);
1824 settings.oom_adj_is_set = true;
1825 settings.oom_adj = (int)parse_snum_param(input_pos, oom_adj_str, name, -1000, 1000);
1826 break;
1827 }
1828 #endif
1829 case setting_id_t::SOCKET_LISTEN:
1830 settings.socket_path = read_setting_value(input_pos, i, end, nullptr);
1831 break;
1832 case setting_id_t::SOCKET_PERMISSIONS:
1833 {
1834 string sock_perm_str = read_setting_value(input_pos, i, end, nullptr);
1835 settings.socket_perms = parse_perms(input_pos, sock_perm_str, name, "socket-permissions");
1836 break;
1837 }
1838 case setting_id_t::SOCKET_UID:
1839 {
1840 string sock_uid_s = read_setting_value(input_pos, i, end, nullptr);
1841 settings.socket_uid = parse_uid_param(input_pos, sock_uid_s, name, "socket-uid",
1842 &settings.socket_uid_gid);
1843 break;
1844 }
1845 case setting_id_t::SOCKET_GID:
1846 {
1847 string sock_gid_s = read_setting_value(input_pos, i, end, nullptr);
1848 settings.socket_gid = parse_gid_param(input_pos, sock_gid_s, name, "socket-gid");
1849 break;
1850 }
1851 case setting_id_t::STOP_COMMAND:
1852 read_setting_value(settings.stop_command, setting_op, input_pos, i, end,
1853 &settings.stop_command_offsets);
1854 break;
1855 case setting_id_t::PID_FILE:
1856 settings.pid_file = read_setting_value(input_pos, i, end);
1857 break;
1858 case setting_id_t::DEPENDS_ON:
1859 {
1860 string dependency_name = read_value_resolved(setting.c_str(), input_pos, i, end,
1861 service_arg, lookup_var);
1862 settings.depends.emplace_back(load_service(dependency_name.c_str()),
1863 dependency_type::REGULAR);
1864 break;
1865 }
1866 case setting_id_t::DEPENDS_MS:
1867 {
1868 string dependency_name = read_value_resolved(setting.c_str(), input_pos, i, end,
1869 service_arg, lookup_var);
1870 settings.depends.emplace_back(load_service(dependency_name.c_str()),
1871 dependency_type::MILESTONE);
1872 break;
1873 }
1874 case setting_id_t::WAITS_FOR:
1875 {
1876 string dependency_name = read_value_resolved(setting.c_str(), input_pos, i, end,
1877 service_arg, lookup_var);
1878 settings.depends.emplace_back(load_service(dependency_name.c_str()),
1879 dependency_type::WAITS_FOR);
1880 break;
1881 }
1882 case setting_id_t::WAITS_FOR_D:
1883 {
1884 string waitsford = read_setting_value(input_pos, i, end);
1885 process_dep_dir(settings.depends, waitsford, dependency_type::WAITS_FOR);
1886 break;
1887 }
1888 case setting_id_t::DEPENDS_ON_D:
1889 {
1890 string depends_on_d = read_setting_value(input_pos, i, end);
1891 process_dep_dir(settings.depends, depends_on_d, dependency_type::REGULAR);
1892 break;
1893 }
1894 case setting_id_t::DEPENDS_MS_D:
1895 {
1896 string depends_ms_d = read_setting_value(input_pos, i, end);
1897 process_dep_dir(settings.depends, depends_ms_d, dependency_type::MILESTONE);
1898 break;
1899 }
1900 case setting_id_t::AFTER:
1901 {
1902 string after_name = read_value_resolved(setting.c_str(), input_pos, i, end,
1903 service_arg, lookup_var);
1904 settings.after_svcs.emplace_back(std::move(after_name));
1905 break;
1906 }
