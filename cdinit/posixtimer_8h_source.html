53 {
54 auto &real_timer_queue = this->queue_for_clock(clock_type::SYSTEM);
55 auto &mono_timer_queue = this->queue_for_clock(clock_type::MONOTONIC);
56
57 if (siginfo.get_signo() == SIGALRM) {
58 time_val curtime;

60 if (! real_timer_queue.empty()) {
61 this->get_time(curtime, clock_type::SYSTEM, true);

63 set_timer_from_queue(real_timer, real_timer_queue);

141 // starts (if not started) a timer to timeout at the given time. Resets the expiry count to 0.
142 // enable: specifies whether to enable reporting of timeouts/intervals
143 void set_timer(timer_handle_t &timer_id, const timespec &timeout, const timespec &interval,
144 bool enable, clock_type clock = clock_type::MONOTONIC) noexcept
145 {
146 std::lock_guard<decltype(Base::lock)> guard(Base::lock);
147
148 timer_queue_t &timer_queue = this->queue_for_clock(clock);
149 timer_t &timer = timer_for_clock(clock);
150
151 auto &ts = timer_queue.node_data(timer_id);
152 ts.interval_time = interval;
153 ts.expiry_count = 0;
154 ts.enabled = enable;
155
156 if (timer_queue.is_queued(timer_id)) {
157 // Already queued; alter timeout
158 if (timer_queue.set_priority(timer_id, timeout)) {
159 if (clock != clock_type::MONOTONIC || provide_mono_timer) {
160 set_timer_from_queue(timer, timer_queue);
161 }
162 }
163 }
164 else {
165 if (timer_queue.insert(timer_id, timeout)) {
166 if (clock != clock_type::MONOTONIC || provide_mono_timer) {
167 set_timer_from_queue(timer, timer_queue);
168 }
169 }
170 }

195 {
196 timer_queue_t &timer_queue = this->queue_for_clock(clock);
197 timer_t &timer = timer_for_clock(clock);
198
199 if (timer_queue.is_queued(timer_id)) {
200 bool was_first = (&timer_queue.get_root()) == &timer_id;
201 timer_queue.remove(timer_id);

203 set_timer_from_queue(timer, timer_queue);
