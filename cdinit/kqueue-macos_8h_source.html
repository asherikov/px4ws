22// "kqueue"-based event loop mechanism.
23//
24// kqueue is available on BSDs and Mac OS X, though there are subtle differences from OS to OS.
25//
26// kqueue supports watching file descriptors (input and output as separate watches only),
27// signals, child processes, and timers. Unfortunately support for the latter two is imperfect;
28// it is not possible to reserve process watches in advance; timers can only be active, count
29// down immediately when created, and cannot be reset to another time. For timers especially
30// the problems are significant: we can't allocate timers in advance, and we can't even feasibly
31// manage our own timer queue via a single kqueue-backed timer. Therefore, an alternate timer
32// mechanism must be used together with kqueue.

94 // The flag to specify poll() semantics for regular file readiness: that is, we want
95 // ready-for-read to be returned even at end of file:

106 void process_events(struct kevent *events, int r)
107 {
108 std::lock_guard<decltype(Base::lock)> guard(Base::lock);
109
110 for (int i = 0; i < r; i++) {
111 if (events[i].filter == EVFILT_READ || events[i].filter == EVFILT_WRITE) {
112 int flags = events[i].filter == EVFILT_READ ? IN_EVENTS : OUT_EVENTS;
113 auto r = Base::receive_fd_event(*this, fd_r(events[i].ident), events[i].udata, flags);
114 if (std::get<0>(r) == 0) {
115 // we use EV_CLEAR to clear the EOF status of fifos/pipes (and wait for
116 // another connection).
117 events[i].flags = EV_DISABLE | EV_CLEAR;
118 }
119 else {
120 events[i].flags = EV_ENABLE;
121 }
122 }
123 else {
124 events[i].flags = EV_DISABLE;
125 }
126 }
127
128 // Now we disable all received events, to simulate EV_DISPATCH. Note that EV_DISPATH is
129 // actually available on MacOS, but we can't use it due to the signal processing bug.
130 kevent(kqfd, events, r, nullptr, 0, nullptr);
131 }

191 // fd: file descriptor to watch
192 // userdata: data to associate with descriptor
193 // flags: IN_EVENTS | OUT_EVENTS | ONE_SHOT
194 // (only one of IN_EVENTS/OUT_EVENTS can be specified)
195 // soft_fail: true if unsupported file descriptors should fail by returning false instead
196 // of throwing an exception
197 // returns: true on success; false if file descriptor type isn't supported and emulate == true
198 // throws: std::system_error or std::bad_alloc on failure
199 bool add_fd_watch(int fd, void *userdata, int flags, bool enabled = true, bool emulate = false)
200 {
201 short filter = (flags & IN_EVENTS) ? EVFILT_READ : EVFILT_WRITE;
202
203 if (filter == EVFILT_READ && POLL_SEMANTICS == 0 && emulate) {
204 // We can't request poll semantics, so check for regular file:
205 struct stat statbuf;
206 if (fstat(fd, &statbuf) == -1) {
207 throw std::system_error(errno, std::system_category());
208 }
209 if ((statbuf.st_mode & S_IFMT) == S_IFREG) {
210 // Regular file: emulation required
211 return false;
212 }
213 }
214
215 int fflags = (filter == EVFILT_READ) ? POLL_SEMANTICS : 0;
216
217 struct kevent kev;
218 EV_SET(&kev, fd, filter, EV_ADD | (enabled ? 0 : EV_DISABLE), fflags, 0, userdata);
219 if (kevent(kqfd, &kev, 1, nullptr, 0, nullptr) == -1) {
220 // Note that kqueue supports EVFILT_READ on regular file fd's, but not EVFILT_WRITE.
221 if (filter == EVFILT_WRITE && errno == EINVAL && emulate) {
222 return false; // emulate
223 }
224 throw std::system_error(errno, std::system_category());
225 }
226 return true;
227 }

229 // returns: 0 on success
230 // IN_EVENTS if in watch requires emulation
231 // OUT_EVENTS if out watch requires emulation
232 int add_bidi_fd_watch(int fd, void *userdata, int flags, bool emulate = false)
233 {
234#ifdef EV_RECEIPT
235 struct kevent kev[2];
236 struct kevent kev_r[2];
237 short rflags = EV_ADD | ((flags & IN_EVENTS) ? 0 : EV_DISABLE) | EV_RECEIPT;
238 short wflags = EV_ADD | ((flags & OUT_EVENTS) ? 0 : EV_DISABLE) | EV_RECEIPT;
239 EV_SET(&kev[0], fd, EVFILT_READ, rflags, POLL_SEMANTICS, 0, userdata);
240 EV_SET(&kev[1], fd, EVFILT_WRITE, wflags, 0, 0, userdata);
241
242 int r = kevent(kqfd, kev, 2, kev_r, 2, nullptr);
243
244 if (r == -1) {
245 throw std::system_error(errno, std::system_category());
246 }
247
248 // Some possibilities:
249 // - both ends failed. We'll throw an error rather than allowing emulation.
250 // - read watch failed, write succeeded : should not happen.
251 // - read watch added, write failed: if emulate == true, succeed;
252 // if emulate == false, remove read and fail.
253
254 if (kev_r[0].data != 0) {
255 // read failed
256 throw std::system_error(kev_r[0].data, std::system_category());
257 }
258
259 if (kev_r[1].data != 0) {
260 if (emulate) {
261 // We can emulate, but, do we have correct semantics?
262 if (POLL_SEMANTICS != 0) {
263 return OUT_EVENTS;
264 }
265
266 // if we can't get poll semantics, emulate for read as well:
267 // first remove read watch:
268 EV_SET(&kev[0], fd, EVFILT_READ, EV_DELETE, 0, 0, userdata);
269 kevent(kqfd, kev, 1, nullptr, 0, nullptr);
270 return IN_EVENTS | OUT_EVENTS;
271 }
272 // remove read watch
273 EV_SET(&kev[0], fd, EVFILT_READ, EV_DELETE, 0, 0, userdata);
274 kevent(kqfd, kev, 1, nullptr, 0, nullptr);
275 // throw exception
276 throw std::system_error(kev_r[1].data, std::system_category());
277 }
278
279 return 0;
280#else
281 // OpenBSD doesn't have EV_RECEIPT: install the watches one at a time
282 struct kevent kev[1];
283
284 short rflags = EV_ADD | ((flags & IN_EVENTS) ? 0 : EV_DISABLE);
285 short wflags = EV_ADD | ((flags & OUT_EVENTS) ? 0 : EV_DISABLE);
286 EV_SET(&kev[0], fd, EVFILT_READ, rflags, 0, 0, userdata);
287
288 int r = kevent(kqfd, kev, 1, nullptr, 0, nullptr);
289
290 if (r == -1) {
291 throw std::system_error(errno, std::system_category());
292 }
293
294 EV_SET(&kev[0], fd, EVFILT_WRITE, wflags, 0, 0, userdata);
295
296 r = kevent(kqfd, kev, 1, nullptr, 0, nullptr);
297
298 if (r == -1) {
299 if (emulate) {
300 return OUT_EVENTS;
301 }
302 // remove read watch
303 EV_SET(&kev[0], fd, EVFILT_READ, EV_DELETE, 0, 0, userdata);
304 kevent(kqfd, kev, 1, nullptr, 0, nullptr);
305 // throw exception
306 throw std::system_error(errno, std::system_category());
307 }
308
309 return 0;
310#endif

354 // If events are pending, process an unspecified number of them.
355 // If no events are pending, wait until one event is received and
356 // process this event (and possibly any other events received
357 // simultaneously).
358 // If processing an event removes a watch, there is a possibility
359 // that the watched event will still be reported (if it has
360 // occurred) before pull_events() returns.
361 //
362 // do_wait - if false, returns immediately if no events are
363 // pending.
364 void pull_events(bool do_wait)
365 {
366 struct kevent events[16];
367 struct timespec ts;
368
369 // wait_ts remains null for an infinite wait; it is later set to either a 0 timeout
370 // if do_wait is false (or if we otherwise won't wait due to events being detected
371 // early) or is set to an appropriate timeout for the next timer's timeout.
372 struct timespec *wait_ts = nullptr;
373
374 Base::lock.lock();
375
376 // Check whether any timers are pending, and what the next timeout is.
377 this->process_monotonic_timers(do_wait, ts, wait_ts);
378
379 const sigset_t &active_sigmask = this->get_active_sigmask();
380 Base::lock.unlock();
381
382 // using sigjmp/longjmp is ugly, but there is no other way. If a signal that we're watching is
383 // received during polling, it will longjmp back to here:
384 if (sigsetjmp(this->get_sigreceive_jmpbuf(), 1) != 0) {
385 this->process_signal();
386 do_wait = false;
387 }
388
389 if (! do_wait) {
390 ts.tv_sec = 0;
391 ts.tv_nsec = 0;
392 wait_ts = &ts;
393 }
394
395 // Run kevent with signals unmasked:
396 this->sigmaskf(SIG_UNBLOCK, &active_sigmask, nullptr);
397 int r = kevent(kqfd, nullptr, 0, events, 16, wait_ts);
398 this->sigmaskf(SIG_BLOCK, &active_sigmask, nullptr);
399
400 if (r == -1 || r == 0) {
401 // signal or no events
402 if (r == 0 && do_wait) {
403 // timeout:
404 Base::lock.lock();
405 this->process_monotonic_timers();
406 Base::lock.unlock();
407 }
408 return;
409 }
410
411 ts = time_val(0, 0);
412
413 do {
414 process_events(events, r);
415 r = kevent(kqfd, nullptr, 0, events, 16, &ts);
416 } while (r > 0);
417 }
