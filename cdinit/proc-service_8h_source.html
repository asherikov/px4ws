690 time_val current_time;
691 event_loop.get_time(current_time, clock_type::MONOTONIC);
692 time_val int_diff = current_time - restart_interval_time;
693 if (int_diff < restart_interval) {
694 // Within the restart limiting interval; check number of restarts
695 if (restart_interval_count >= max_restart_interval_count) {
696 log(loglevel_t::ERROR, "Service ", get_name(), " restarting too quickly; stopping.");
697 set_target_state(service_state_t::STOPPED);
698 return false;
699 }
700 ++restart_interval_count;
701 }
702 else {
703 // Not within the last limiting interval; start a new interval
704 restart_interval_time = current_time;
705 restart_interval_count = 1;
706 }
707 }
708
709 return true;
710 }
