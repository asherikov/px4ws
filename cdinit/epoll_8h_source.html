122 void process_events(epoll_event *events, int r)
123 {
124 std::lock_guard<decltype(Base::lock)> guard(Base::lock);
125
126 for (int i = 0; i < r; i++) {
127 void * ptr = events[i].data.ptr;
128
129 if (ptr == &sigfd) {
130 // Signal
131 sigdata_t siginfo;
132 while (true) {
133 int r = read(sigfd, &siginfo.info, sizeof(siginfo.info));
134 if (r == -1) break;
135 auto iter = sigdataMap.find(siginfo.get_signo());
136 if (iter != sigdataMap.end()) {
137 void *userdata = (*iter).second;
138 if (Base::receive_signal(*this, siginfo, userdata)) {
139 sigdelset(&sigmask, siginfo.get_signo());
140 }
141 }
142 }
143 signalfd(sigfd, &sigmask, SFD_NONBLOCK | SFD_CLOEXEC);
144 }
145 else {
146 int flags = 0;
147 (events[i].events & EPOLLIN) && (flags |= IN_EVENTS);
148 (events[i].events & EPOLLOUT) && (flags |= OUT_EVENTS);
149 // We mustn't introduce IN/OUT events for error conditions as we don't know which are being
150 // watched! Just set ERR_EVENTS.
151 (events[i].events & EPOLLHUP) && (flags |= ERR_EVENTS);
152 (events[i].events & EPOLLERR) && (flags |= ERR_EVENTS);
153 auto r = Base::receive_fd_event(*this, fd_r(), ptr, flags);
154 if (std::get<0>(r) != 0) {
155 enable_fd_watch_nolock(fd_r().get_fd(std::get<1>(r)), ptr, std::get<0>(r));
156 }
157 }
158 }

205 // fd: file descriptor to watch
206 // userdata: data to associate with descriptor
207 // flags: IN_EVENTS | OUT_EVENTS | ONE_SHOT
208 // soft_fail: true if unsupported file descriptors should fail by returning false instead
209 // of throwing an exception
210 // returns: true on success; false if file descriptor type isn't supported and soft_fail == true
211 // throws: std::system_error or std::bad_alloc on failure
212 bool add_fd_watch(int fd, void *userdata, int flags, bool enabled = true, bool soft_fail = false)
213 {
214 struct epoll_event epevent;
215 // epevent.data.fd = fd;
216 epevent.data.ptr = userdata;
217 epevent.events = 0;
218
219 if (flags & ONE_SHOT) {
220 epevent.events = EPOLLONESHOT;
221 }
222 if ((flags & IN_EVENTS) && enabled) {
223 epevent.events |= EPOLLIN;
224 }
225 if ((flags & OUT_EVENTS) && enabled) {
226 epevent.events |= EPOLLOUT;
227 }
228
229 if (epoll_ctl(epfd, EPOLL_CTL_ADD, fd, &epevent) == -1) {
230 if (soft_fail && errno == EPERM) {
231 return false;
232 }
233 throw std::system_error(errno, std::system_category());
234 }
235 return true;
236 }

367 // If events are pending, process an unspecified number of them.
368 // If no events are pending, wait until one event is received and
369 // process this event (and possibly any other events received
370 // simultaneously).
371 // If processing an event removes a watch, there is a possibility
372 // that the watched event will still be reported (if it has
373 // occurred) before pull_events() returns.
374 //
375 // do_wait - if false, returns immediately if no events are
376 // pending.
377 void pull_events(bool do_wait)
378 {
379 epoll_event events[16];
380 int r = epoll_wait(epfd, events, 16, do_wait ? -1 : 0);
381 if (r == -1 || r == 0) {
382 // signal or no events
383 return;
384 }
385
386 do {
387 process_events(events, r);
388 r = epoll_wait(epfd, events, 16, 0);
389 } while (r > 0);
390 }
