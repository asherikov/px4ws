15/**
16 * Priority queue implementation based on a heap with parameterised fan-out. All nodes are stored
17 * in a vector, with the root at position 0, and each node has N child nodes, at positions
18 * (p * N + 1) through (p * N + N) where p is the (parent) node position.
19 *
20 * With N=2, this is a binary heap. Higher values of N may give better performance due to better
21 * cache locality, but also increase fan-out which will (if too high) also reduce performance.
22 *
23 * The destructor will not clean up (destruct) objects that have been added to the queue. If the
24 * destructor of the element type (T) is non-trivial, all handles should be de-allocated before
25 * destroying the queue.
26 *
27 * Implementation details:
28 *
29 * Adding a node returns a "handle", which maintains an index into the heap. When the position of
30 * a node in the heap changes, its handle must be updated (the advantage is that changing priority
31 * of or removing a node does not require a linear search for the node).
32 *
33 * Node data is actually stored as part of the handle, not in the queue.
34 *
35 * To add a node to the queue, it is inserted at the end and then "bubbled down" to its correct
36 * location according to priority. To removing a node, the node is replaced with the last node in
37 * the vector and then that is "bubbled up" to the correct position.
38 *
39 * Parameters:
40 *
41 * T : node data type
42 * P : priority type (eg int)
43 * Compare : functional object type to compare priorities
44 * N : fan out factor (number of child nodes per node)
45 */
46template <typename T, typename P, typename Compare = std::less<P>, int N = 4>

120 bool bubble_down(hindex_t pos, handle_t * ohndl, const P &op) noexcept
121 {
122 Compare lt;
123 while (pos > 0) {
124 hindex_t parent = (pos - 1) / N;
125 if (! lt(op, hvec[parent].prio)) {
126 break;
127 }
128
129 hvec[pos] = std::move(hvec[parent]);
130 hvec[pos].hnd->heap_index = pos;
131 pos = parent;
132 }
133
134 hvec[pos].hnd = ohndl;
135 hvec[pos].prio = std::move(op);
136 ohndl->heap_index = pos;
137
138 return pos == 0;
139 }

148 void bubble_up(hindex_t pos, handle_t &h, const P &p) noexcept
149 {
150 hindex_t rmax = hvec.size() - 1;
151 if (rmax == 0) {
152 return;
153 }
154
155 Compare lt;
156 hindex_t max = (rmax - 1) / N;
157
158 while (pos <= max) {
159 // Find (select) the smallest child node
160 hindex_t lchild = pos * N + 1;
161 hindex_t selchild = lchild;
162 hindex_t rchild = std::min(lchild + N, rmax);
163 for (hindex_t i = lchild + 1; i < rchild; i++) {
164 if (lt(hvec[i].prio, hvec[selchild].prio)) {
165 selchild = i;
166 }
167 }
168
169 if (! lt(hvec[selchild].prio, p)) {
170 break;
171 }
172
173 hvec[pos] = std::move(hvec[selchild]);
174 hvec[pos].hnd->heap_index = pos;
175 pos = selchild;
176 }
177
178 hvec[pos].hnd = &h;
179 hvec[pos].prio = std::move(p);
180 h.heap_index = pos;
181 }

202 // Allocate a slot, but do not incorporate into the heap:
203 // u... : parameters for data constructor T::T(...)
204 template <typename ...U> void allocate(handle_t & hnd, U&&... u)
205 {
206 // Note: this can be constexpr in C++20
207 const hindex_t max_allowed = hvec.max_size();
208
209 if (num_nodes == max_allowed) {
210 throw std::bad_alloc();
211 }
212
213 num_nodes++;
214
215 if (DASYNQ_EXPECT(hvec.capacity() < num_nodes, 0)) {
216 hindex_t half_point = max_allowed / 2;
217 try {
218 if (DASYNQ_EXPECT(num_nodes < half_point, 1)) {
219 hvec.reserve(num_nodes * 2);
220 }
221 else {
222 hvec.reserve(max_allowed);
223 }
224 }
225 catch (...) {
226 // try with just the needed number of nodes:
227 try {
228 hvec.reserve(num_nodes);
229 }
230 catch (...) {
231 num_nodes--;
232 throw;
233 }
234 }
235 }
236
237 new (& hnd.hd_u.hd) T(std::forward<U>(u)...);
238 hnd.heap_index = -1;
239 }

306 // Increase key
307 hvec[heap_index].prio = p;
