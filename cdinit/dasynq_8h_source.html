11// Dasynq uses a "mix-in" pattern to produce an event loop implementation incorporating selectable
12// implementations of various components (main backend, timers, child process watch mechanism etc). In C++
13// this can be achieved by a template for some component which extends its own type parameter:
14//
15// template <typename Base> class X : public B { .... }
16//
17// (Note that in a sense this is actually the opposite of the so-called "Curiously Recurring Template"
18// pattern, which can be used to achieve a similar goal). We can chain several such components together to
19// "mix in" the functionality of each into the final class, eg:
20//
21// template <typename T> using loop_t =
22// epoll_loop<interrupt_channel<timer_fd_events<child_proc_events<T>>>>;
23//
24// (which defines an alias template "loop_t", whose implementation will use the epoll backend, a standard
25// interrupt channel implementation, a timerfd-based timer implementation, and the standard child process
26// watch implementation). We sometimes need the base class to be able to call derived-class members: to do
27// this we pass a reference to the derived instance into a template member function in the base, for example
28// the "init" function:
29//
30// template <typename T> void init(T *derived)
31// {
32// // can call method on derived:
33// derived->add_listener();
34// // chain to next class:
35// Base::init(derived);
36// }
37//
38// The 'loop_t' defined above is a template for a usable backend mechanism for the event_loop template
39// class. At the base all this is the event_dispatch class, defined below, which receives event
40// notifications and inserts them into a queue for processing. The event_loop class, also below, wraps this
41// (via composition) in an interface which can be used to register/de-register/enable/disable event
42// watchers, and which can process the queued events by calling the watcher callbacks. The event_loop class
43// also provides some synchronisation to ensure thread-safety, and abstracts away some differences between
44// backends.
45//
46// The differences are exposed as traits, partly via a separate traits class (loop_traits_t as defined
47// below, which contains the "main" traits, particularly the sigdata_t, fd_r and fd_s types). Note that the
48// event_dispatch class exposes the loop traits as traits_t, and these are then potentially augmented at
49// each stage of the mechanism inheritance chain (i.e. the final traits are exposed as
50// `loop_t<event_dispatch>::traits_t'.
51//
52// The trait members are:
53// sigdata_t - a wrapper for the siginfo_t type or equivalent used to pass signal parameters
54// fd_r - a file descriptor wrapper, if the backend is able to retrieve the file descriptor when
55// it receives an fd event. Not all backends can do this.
56// fd_s - a file descriptor storage wrapper. If the backend can retrieve file descriptors, this
57// will be empty (and ideally zero-size), otherwise it stores a file descriptor.
58// With an fd_r and fd_s instance you can always retrieve the file descriptor:
59// `fdr.get_fd(fds)' will return it.
60// has_bidi_fd_watch
61// - boolean indicating whether a single watch can support watching for both input and output
62// events simultaneously
63// has_separate_rw_fd_watches
64// - boolean indicating whether it is possible to add separate input and output watches for the
65// same fd. Either this or has_bidi_fd_watch must be true.
66// interrupt_after_fd_add
67// - boolean indicating if a loop interrupt must be forced after adding/enabling an fd watch.
68// interrupt_after_signal_add
69// - boolean indicating if a loop interrupt must be forced after adding or enabling a signal
70// watch.
71// supports_non_oneshot_fd
72// - boolean; if true, event_dispatch can arm an fd watch without ONESHOT and returning zero
73// events from receive_fd_event (the event notification function) will leave the descriptor
74// armed. If false, all fd watches are effectively ONESHOT (they can be re-armed immediately
75// after delivery by returning an appropriate event flag mask).
76// full_timer_support
77// - boolean indicating that the monotonic and system clocks are actually different clocks and
78// that timers against the system clock will work correctly if the system clock time is
79// adjusted. If false, the monotonic clock may not be present at all (monotonic clock will map
80// to system clock), and timers against either clock are not guaranteed to work correctly if
81// the system clock is adjusted.

398 // Do standard post-dispatch processing for a watcher. This handles the case of removing or
399 // re-queueing watchers depending on the rearm type. This is called from the individual
400 // watcher dispatch functions to handle REMOVE or REQUEUE re-arm values.
401 template <typename Loop> void post_dispatch(Loop &loop, base_watcher *watcher, rearm rearm_type)
402 {
403 if (rearm_type == rearm::REMOVE) {
404 loop_access::get_base_lock(loop).unlock();
405 loop_access::release_watcher(loop, watcher);
406 watcher->watch_removed();
407 loop_access::get_base_lock(loop).lock();
408 }
409 else if (rearm_type == rearm::REQUEUE) {
410 loop_access::requeue_watcher(loop, watcher);
411 }
412 }

506 // Receive fd event delivered from backend mechansim. Returns the desired watch mask, as per
507 // set_fd_enabled, which can be used to leave the watch disabled, re-enable it or re-enable
508 // one direction of a bi-directional watcher.
509 template <typename T>
510 std::tuple<int, typename Traits::fd_s> receive_fd_event(T &loop_mech, typename Traits::fd_r fd_r,
511 void * userdata, int flags) noexcept
512 {
513 base_fd_watcher * bfdw = static_cast<base_fd_watcher *>(userdata);
514
515 bfdw->event_flags |= flags;
516 typename Traits::fd_s watch_fd_s {bfdw->watch_fd};
517
518 base_watcher * bwatcher = bfdw;
519
520 bool is_multi_watch = bfdw->watch_flags & multi_watch;
521 if (is_multi_watch) {
522 base_bidi_fd_watcher *bbdw = static_cast<base_bidi_fd_watcher *>(bwatcher);
523 bbdw->watch_flags &= ~flags;
524 if ((flags & IN_EVENTS) && (flags & OUT_EVENTS)) {
525 // Queue the secondary watcher first:
526 queue_watcher(&bbdw->out_watcher);
527 }
528 else if (flags & OUT_EVENTS) {
529 // Use the secondary watcher for queueing:
530 bwatcher = &(bbdw->out_watcher);
531 }
532 }
533
534 queue_watcher(bwatcher);
535
536 if (is_multi_watch && ! traits_t::has_separate_rw_fd_watches) {
537 // If this is a bidirectional fd-watch, it has been disabled in *both* directions
538 // as the event was delivered. However, the other direction should not be disabled
539 // yet, so we need to re-enable:
540 int in_out_mask = IN_EVENTS | OUT_EVENTS;
541 if ((bfdw->watch_flags & in_out_mask) != 0) {
542 // We need to re-enable the other channel now:
543 return std::make_tuple((bfdw->watch_flags & in_out_mask) | ONE_SHOT, watch_fd_s);
544 // We are the polling thread: don't need to interrupt polling, even if it would
545 // normally be required.
546 }
547 }
548
549 return std::make_tuple(0, watch_fd_s);
550 }

588 // Call with lock free.

590 {

606 release_watcher(watcher);

705 // There is a complex problem with most asynchronous event notification mechanisms
706 // when used in a multi-threaded environment. Generally, a file descriptor or other
707 // event type that we are watching will be associated with some data used to manage
708 // that event source. For example a web server needs to maintain information about
709 // each client connection, such as the state of the connection (what protocol version
710 // has been negotiated, etc; if a transfer is taking place, what file is being
711 // transferred etc).
712 //
713 // However, sometimes we want to remove an event source (eg webserver wants to drop
714 // a connection) and delete the associated data. The problem here is that it is
715 // difficult to be sure when it is ok to actually remove the data, since when
716 // requesting to unwatch the source in one thread it is still possible that an
717 // event from that source is just being reported to another thread (in which case
718 // the data will be needed).
719 //
720 // To solve that, we:
721 // - allow only one thread to poll for events at a time, using a lock
722 // - use the same lock to prevent polling, if we want to unwatch an event source
723 // - generate an event to interrupt, when necessary, any polling that may already be occurring
724 // in another thread
725 // - mark handlers as active if they are currently executing, and
726 // - when removing an active handler, simply set a flag which causes it to be
727 // removed once the current processing is finished, rather than removing it
728 // immediately.
729 //
730 // In particular the lock mechanism for preventing multiple threads polling and
731 // for allowing polling to be interrupted is tricky. We can't use a simple mutex
732 // since there is significant chance that it will be highly contended and there
733 // are no guarantees that its acquisition will be fair. In particular, we don't
734 // want a thread that is trying to unwatch a source being starved while another
735 // thread polls the event source.
736 //
737 // So, we use two wait queues protected by a single mutex. The "attn_waitqueue"
738 // (attention queue) is the high-priority queue, used for threads wanting to
739 // unwatch event sources. The "wait_waitquueue" is the queue used by threads
740 // that wish to actually poll for events, while they are waiting for the main
741 // queue to become quiet.
742 // - The head of the "attn_waitqueue" is always the holder of the lock
743 // - Therefore, a poll-waiter must be moved from the wait_waitqueue to the
744 // attn_waitqueue to actually gain the lock. This is only done if the
745 // attn_waitqueue is otherwise empty.
746 // - The mutex only protects manipulation of the wait queues, and so should not
747 // be highly contended.
748 //
749 // To claim the lock for a poll-wait, the procedure is:
750 // - check if the attn_waitqueue is empty;
751 // - if it is, insert node at the head, thus claiming the lock, and return
752 // - otherwise, insert node in the wait_waitqueue, and wait
753 // To claim the lock for an unwatch, the procedure is:
754 // - insert node in the attn_waitqueue
755 // - if the node is at the head of the queue, lock is claimed; return
756 // - otherwise, if a poll is in progress, interrupt it
757 // - wait until our node is at the head of the attn_waitqueue
758 //
759 // Some backends also need to interrupted in order to add a new watch (eg select/pselect).
760 // However, the attn_waitqueue lock doesn't generally need to be obtained for this.
761

1109 // Interrupt the current poll-waiter, if necessary - that is, if the loop is multi-thread safe, and if
1110 // there is currently another thread polling the backend event mechanism.
1111 void interrupt_if_necessary()
1112 {
1113 wait_lock.lock();
1114 bool attn_q_empty = attn_waitqueue.is_empty(); // (always true for single-threaded loops)
1115 wait_lock.unlock();
1116
1117 if (! attn_q_empty) {
1118 loop_mech.interrupt_wait();
1119 }
1120 }

1122 // Acquire the attention lock (when held, ensures that no thread is polling the AEN
1123 // mechanism). This can be used to safely remove watches, since it is certain that
1124 // notification callbacks won't be run while the attention lock is held. Any in-progress
1125 // poll will be interrupted so that the lock should be acquired quickly.
1126 void get_attn_lock(waitqueue_node<T_Mutex> &qnode) noexcept
1127 {
1128 std::unique_lock<T_Mutex> ulock(wait_lock);
1129 attn_waitqueue.queue(&qnode);
1130 if (! attn_waitqueue.check_head(qnode)) {
1131 if (long_poll_running) {
1132 // We want to interrupt any in-progress poll so that the attn queue will progress
1133 // but we don't want to do that unnecessarily. If we are 2nd in the queue then the
1134 // head must be doing the poll; interrupt it. Otherwise, we assume the 2nd has
1135 // already interrupted it.
1136 if (attn_waitqueue.get_second() == &qnode) {
1137 loop_mech.interrupt_wait();
1138 }
1139 }
1140 while (! attn_waitqueue.check_head(qnode)) {
1141 qnode.wait(ulock);
1142 }
1143 }
1144 }
1145
1146 // Acquire the attention lock, but without interrupting any poll that's in progress
1147 // (prefer to fail in that case).
1148 bool poll_attn_lock(waitqueue_node<T_Mutex> &qnode) noexcept
1149 {
1150 std::unique_lock<T_Mutex> ulock(wait_lock);
1151 if (long_poll_running) {
1152 // There are poll-waiters, bail out
1153 return false;
1154 }
1155
1156 // Nobody's doing a long poll, wait until we're at the head of the attn queue and return
1157 // success:
1158 attn_waitqueue.queue(&qnode);
1159 while (! attn_waitqueue.check_head(qnode)) {
1160 qnode.wait(ulock);
1161 }
1162
1163 return true;
1164 }

1166 // Acquire the poll-wait lock (to be held when polling the AEN mechanism; lower priority than
1167 // the attention lock). The poll-wait lock is used to prevent more than a single thread from
1168 // polling the event loop mechanism at a time; if this is not done, it is basically
1169 // impossible to safely deregister watches.
1170 void get_pollwait_lock(waitqueue_node<T_Mutex> &qnode) noexcept
1171 {
1172 std::unique_lock<T_Mutex> ulock(wait_lock);
1173 if (attn_waitqueue.is_empty()) {
1174 // Queue is completely empty:
1175 attn_waitqueue.queue(&qnode);
1176 }
1177 else {
1178 wait_waitqueue.queue(&qnode);
1179 }
1180
1181 while (! attn_waitqueue.check_head(qnode)) {
1182 qnode.wait(ulock);
1183 }
1184
1185 long_poll_running = true;
1186 }
1187
1188 // Release the poll-wait/attention lock.
1189 void release_lock(waitqueue_node<T_Mutex> &qnode) noexcept
1190 {
1191 std::unique_lock<T_Mutex> ulock(wait_lock);
1192 long_poll_running = false;
1193 waitqueue_node<T_Mutex> * nhead = attn_waitqueue.unqueue();
1194 if (nhead != nullptr) {
1195 // Someone else now owns the lock, signal them to wake them up
1196 nhead->signal();
1197 }
1198 else {
1199 // Nobody is waiting in attn_waitqueue (the high-priority queue) so check in
1200 // wait_waitqueue (the low-priority queue)
1201 if (! wait_waitqueue.is_empty()) {
1202 auto nhead = wait_waitqueue.get_head();
1203 wait_waitqueue.unqueue();
1204 attn_waitqueue.queue(nhead);
1205 long_poll_running = true;
1206 nhead->signal();
1207 }
1208 }
1209 }

1227 // Depending on the rearm value, we re-arm, remove, or disarm the watcher, etc.
1228 rearm process_fd_rearm(base_fd_watcher * bfw, rearm rearm_type) noexcept
1229 {
1230 bool emulatedfd = static_cast<base_watcher *>(bfw)->emulatefd;
1231
1232 if (emulatedfd) {
1233 if (rearm_type == rearm::REARM) {
1234 bfw->emulate_enabled = true;
1235 rearm_type = rearm::REQUEUE;
1236 }
1237 else if (rearm_type == rearm::DISARM) {
1238 bfw->emulate_enabled = false;
1239 }
1240 else if (rearm_type == rearm::NOOP) {
1241 if (bfw->emulate_enabled) {
1242 rearm_type = rearm::REQUEUE;
1243 }
1244 }
1245 }
1246 else if (rearm_type == rearm::REARM) {
1247 set_fd_enabled_nolock(bfw, bfw->watch_fd,
1248 bfw->watch_flags & (IN_EVENTS | OUT_EVENTS), true);
1249 }
1250 else if (rearm_type == rearm::DISARM) {
1251 loop_mech.disable_fd_watch_nolock(bfw->watch_fd, bfw->watch_flags);
1252 }
1253 else if (rearm_type == rearm::REMOVE) {
1254 loop_mech.remove_fd_watch_nolock(bfw->watch_fd, bfw->watch_flags);

1259 // Process rearm option from the primary watcher in bidi_fd_watcher
1260 rearm process_primary_rearm(base_bidi_fd_watcher * bdfw, rearm rearm_type) noexcept
1261 {
1262 bool emulatedfd = static_cast<base_watcher *>(bdfw)->emulatefd;
1263
1264 // Called with lock held
1265 if (rearm_type == rearm::REMOVE) {
1266 bdfw->read_removed = 1;
1267
1268 if (backend_traits_t::has_separate_rw_fd_watches) {
1269 bdfw->watch_flags &= ~IN_EVENTS;
1270 if (! emulatedfd) {
1271 loop_mech.remove_fd_watch_nolock(bdfw->watch_fd, IN_EVENTS);
1272 }
1273 return bdfw->write_removed ? rearm::REMOVE : rearm::NOOP;
1274 }
1275 else {
1276 if (! bdfw->write_removed) {
1277 if (bdfw->watch_flags & IN_EVENTS) {
1278 bdfw->watch_flags &= ~IN_EVENTS;
1279 if (! emulatedfd) {
1280 set_fd_enabled_nolock(bdfw, bdfw->watch_fd, bdfw->watch_flags,
1281 bdfw->watch_flags != 0);
1282 }
1283 }
1284 return rearm::NOOP;
1285 }
1286 else {
1287 // both removed: actually remove
1288 if (! emulatedfd) {
1289 loop_mech.remove_fd_watch_nolock(bdfw->watch_fd, 0 /* not used */);
1290 }
1291 return rearm::REMOVE;
1292 }
1293 }
1294 }
1295 else if (rearm_type == rearm::DISARM) {
1296 bdfw->watch_flags &= ~IN_EVENTS;
1297
1298 if (! emulatedfd) {
1299 if (! backend_traits_t::has_separate_rw_fd_watches) {
1300 int watch_flags = bdfw->watch_flags & (IN_EVENTS | OUT_EVENTS);
1301 set_fd_enabled_nolock(bdfw, bdfw->watch_fd, watch_flags, watch_flags != 0);
1302 }
1303 else {
1304 loop_mech.disable_fd_watch_nolock(bdfw->watch_fd, IN_EVENTS);
1305 }
1306 }
1307 }
1308 else if (rearm_type == rearm::REARM) {
1309 if (! emulatedfd) {
1310 bdfw->watch_flags |= IN_EVENTS;
1311 if (! backend_traits_t::has_separate_rw_fd_watches) {
1312 int watch_flags = bdfw->watch_flags;
1313 set_fd_enabled_nolock(bdfw, bdfw->watch_fd,
1314 watch_flags & (IN_EVENTS | OUT_EVENTS), true);
1315 }
1316 else {
1317 set_fd_enabled_nolock(bdfw, bdfw->watch_fd, IN_EVENTS, true);
1318 }
1319 }
1320 else {
1321 bdfw->watch_flags &= ~IN_EVENTS;
1322 rearm_type = rearm::REQUEUE;
1323 }
1324 }
1325 else if (rearm_type == rearm::NOOP) {
1326 if (bdfw->emulatefd) {
1327 if (bdfw->watch_flags & IN_EVENTS) {
1328 bdfw->watch_flags &= ~IN_EVENTS;
1329 rearm_type = rearm::REQUEUE;
1330 }
1331 }
1332 }

1336 // Process re-arm for the secondary (output) watcher in a Bi-direction Fd watcher.
1337 rearm process_secondary_rearm(base_bidi_fd_watcher * bdfw, base_watcher * outw, rearm rearm_type) noexcept
1338 {
1339 bool emulatedfd = outw->emulatefd;
1340
1341 // Called with lock held
1342 if (emulatedfd) {
1343 if (rearm_type == rearm::REMOVE) {
1344 bdfw->write_removed = 1;
1345 bdfw->watch_flags &= ~OUT_EVENTS;
1346 rearm_type = bdfw->read_removed ? rearm::REMOVE : rearm::NOOP;
1347 }
1348 else if (rearm_type == rearm::DISARM) {
1349 bdfw->watch_flags &= ~OUT_EVENTS;
1350 }
1351 else if (rearm_type == rearm::REARM) {
1352 bdfw->watch_flags &= ~OUT_EVENTS;
1353 rearm_type = rearm::REQUEUE;
1354 }
1355 else if (rearm_type == rearm::NOOP) {
1356 if (bdfw->watch_flags & OUT_EVENTS) {
1357 bdfw->watch_flags &= ~OUT_EVENTS;
1358 rearm_type = rearm::REQUEUE;
1359 }
1360 }
1361 return rearm_type;
1362 }
1363 else if (rearm_type == rearm::REMOVE) {
1364 bdfw->write_removed = 1;
1365
1366 if (backend_traits_t::has_separate_rw_fd_watches) {
1367 bdfw->watch_flags &= ~OUT_EVENTS;
1368 loop_mech.remove_fd_watch_nolock(bdfw->watch_fd, OUT_EVENTS);
1369 return bdfw->read_removed ? rearm::REMOVE : rearm::NOOP;
1370 }
1371 else {
1372 if (! bdfw->read_removed) {
1373 if (bdfw->watch_flags & OUT_EVENTS) {
1374 bdfw->watch_flags &= ~OUT_EVENTS;
1375 set_fd_enabled_nolock(bdfw, bdfw->watch_fd, bdfw->watch_flags, true);
1376 }
1377 return rearm::NOOP;
1378 }
1379 else {
1380 // both removed: actually remove
1381 loop_mech.remove_fd_watch_nolock(bdfw->watch_fd, 0 /* not used */);
1382 return rearm::REMOVE;
1383 }
1384 }
1385 }
1386 else if (rearm_type == rearm::DISARM) {
1387 bdfw->watch_flags &= ~OUT_EVENTS;
1388
1389 if (! backend_traits_t::has_separate_rw_fd_watches) {
1390 int watch_flags = bdfw->watch_flags;
1391 set_fd_enabled_nolock(bdfw, bdfw->watch_fd, watch_flags & (IN_EVENTS | OUT_EVENTS), true);
1392 }
1393 else {
1394 loop_mech.disable_fd_watch_nolock(bdfw->watch_fd, OUT_EVENTS);
1395 }
1396 }
1397 else if (rearm_type == rearm::REARM) {
1398 bdfw->watch_flags |= OUT_EVENTS;
1399
1400 if (! backend_traits_t::has_separate_rw_fd_watches) {
1401 int watch_flags = bdfw->watch_flags;
1402 set_fd_enabled_nolock(bdfw, bdfw->watch_fd, watch_flags & (IN_EVENTS | OUT_EVENTS), true);
1403 }
1404 else {
1405 set_fd_enabled_nolock(bdfw, bdfw->watch_fd, OUT_EVENTS | ONE_SHOT, true);
1406 }
1407 }
1408 return rearm_type;
1409 }

1412 {
1413 if (rearm_type == rearm::REMOVE || rearm_type == rearm::DISARM) {
1414 loop_mech.unreserve_child_watch_nolock(bcw->watch_handle);
1415 }
1416 }

1418 void process_timer_rearm(base_timer_watcher *btw, rearm rearm_type) noexcept
1419 {
1420 // Called with lock held
1421 if (rearm_type == rearm::REARM) {
1422 loop_mech.enable_timer_nolock(btw->timer_handle, true, btw->clock);
1423 }
1424 else if (rearm_type == rearm::REMOVE) {

1426 }

1432 // Process queued events; returns true if any events were processed.
1433 // limit - maximum number of events to process before returning; -1 for
1434 // no limit.
1435 bool process_events(int limit) noexcept
1436 {
1437 loop_mech.lock.lock();
1438
1439 if (limit == 0) {
1440 return false;
1441 }
1442
1443 // limit processing to the number of events currently queued, to avoid prolonged processing
1444 // of watchers which requeueu themselves immediately (including file watchers which are using
1445 // emulation for watching regular files)
1446 //
1447 // If limit is -1 (no limit) we rely on this being always larger than/equal to the number of
1448 // queued events when cast to size_t (which is unsigned).
1449 limit = std::min(size_t(limit), loop_mech.num_queued_events());
1450
1451 base_watcher * pqueue = loop_mech.pull_queued_event();
1452 bool active = false;
1453
1454 while (pqueue != nullptr) {
1455
1456 pqueue->active = true;
1457 active = true;
1458
1459 base_bidi_fd_watcher *bbfw = nullptr;
1460
1461 // (Above variables are initialised only to silence compiler warnings).
1462
1463 if (pqueue->watchType == watch_type_t::SECONDARYFD) {
1464 // construct a pointer to the main watcher, using integer arithmetic to avoid undefined
1465 // pointer arithmetic:
1466 uintptr_t rp = (uintptr_t)pqueue;
1467
1468 // Here we take the offset of a member from a non-standard-layout class, which is
1469 // specified to have undefined result by the C++ language standard, but which
1470 // in practice works fine:
1471 _Pragma ("GCC diagnostic push")
1472 _Pragma ("GCC diagnostic ignored \"-Winvalid-offsetof\"")
1473 rp -= offsetof(base_bidi_fd_watcher, out_watcher);
1474 _Pragma ("GCC diagnostic pop")
1475 bbfw = (base_bidi_fd_watcher *)rp;
1476
1477 // issue a secondary dispatch:
1478 bbfw->dispatch_second(this);
1479 }
1480 else {
1481 pqueue->dispatch(this);
1482 }
1483
1484 if (limit > 0) {
1485 limit--;
1486 if (limit == 0) break;
1487 }
1488 pqueue = loop_mech.pull_queued_event();
1489 }
1490
1491 loop_mech.lock.unlock();
1492 return active;
1493 }
1494
1495 public:
1496
1497 using fd_watcher = dprivate::fd_watcher<my_event_loop_t>;
1498 using bidi_fd_watcher = dprivate::bidi_fd_watcher<my_event_loop_t>;
1499 using signal_watcher = dprivate::signal_watcher<my_event_loop_t>;
1500 using child_proc_watcher = dprivate::child_proc_watcher<my_event_loop_t>;
1501 using timer = dprivate::timer<my_event_loop_t>;

1504 template <typename D> using bidi_fd_watcher_impl = dprivate::bidi_fd_watcher_impl<my_event_loop_t, D>;
1505 template <typename D> using signal_watcher_impl = dprivate::signal_watcher_impl<my_event_loop_t, D>;
1506 template <typename D> using child_proc_watcher_impl = dprivate::child_proc_watcher_impl<my_event_loop_t, D>;
1507 template <typename D> using timer_impl = dprivate::timer_impl<my_event_loop_t, D>;
1508
1509 // Poll the event loop and process any pending events (up to a limit). If no events are pending, wait
1510 // for and process at least one event.
1511 void run(int limit = -1) noexcept
1512 {
1513 // Poll the mechanism first, in case high-priority events are pending:
1514 waitqueue_node<T_Mutex> qnode;
1515 get_pollwait_lock(qnode);
1516 loop_mech.pull_events(false);
1517 release_lock(qnode);
1518

1520 // Pull events from the AEN mechanism and insert them in our internal queue:
1521 get_pollwait_lock(qnode);
1522 loop_mech.pull_events(true);
1523 release_lock(qnode);
1524 }
1525 }

1532 loop_mech.pull_events(false);
1533 release_lock(qnode);
1534 }
1535
1536 process_events(limit);
1537 }
