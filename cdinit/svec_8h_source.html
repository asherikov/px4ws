21// Helper function - copy/move vector contents from old to new array
22// (choice of copy vs move depends on whether we can move without throwing)
23template <typename T,
24 bool __use_move = (std::is_nothrow_move_constructible<T>::value || !std::is_copy_constructible<T>::value),
25 bool __copy_is_noexcept = std::is_nothrow_copy_constructible<T>::value>
26struct move_helper
27{
28 static void move(T *from, T *to, size_type count) noexcept
29 {
30 // nothrow copy; we can safely copy-and-destruct as we go
31 for (size_type i = 0; i < count; i++) {
32 new (&to[i]) T(from[i]);
33 from[i].T::~T();
34 }
35 }
36};

38template <typename T>
39struct move_helper<T, false, false>
40{
41 static void move(T *from, T *to, size_type count)
42 {
43 // copy may throw. Wait until the whole vector is copied before destructing.
44 size_type i;
45 try {
46 for (i = 0; i < count; i++) {
47 new (&to[i]) T(from[i]);
48 }
49
50 for (i = 0; i < count; i++) {
51 from[i].T::~T();
52 }
53 }
54 catch (...) {
55 // we have to destruct any constructed items
56 while (i > 0) {
57 to[i].T::~T();
58 }
59 throw;
60 }
61 }
62};

64template <typename T, bool __dontcare>
65struct move_helper<T, true, __dontcare>
66{
67 // we always use noexcept, even if move may throw we cannot safely recover, so best
68 // to terminate.
69 static void move(T *from, T *to, size_type count) noexcept
70 {
71 for (size_type i = 0; i < count; i++) {
72 new (&to[i]) T(std::move(from[i]));
73 from[i].T::~T();
74 }

114 bool ensure_capacity(size_type c, bool exact = false)
115 noexcept(std::is_nothrow_move_constructible<T>::value || std::is_nothrow_copy_constructible<T>::value)
116 {
117 if (capacity_v >= c) return true;

122 if (!exact) {
123 if (max_size() / 2 <= capacity_v) {
124 new_capacity = max_size();

130 }

133 }
134 }
135 }
