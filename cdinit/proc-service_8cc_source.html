19// Given a string and a list of pairs of (start,end) indices for each argument in that string,
20// store a null terminator for the argument. Return a `char *` vector containing the beginning
21// of each argument and a trailing nullptr. (The returned array is invalidated if the string is
22// later modified).
23std::vector<const char *> separate_args(ha_string &s,
24 const std::list<std::pair<unsigned,unsigned>> &arg_indices)
25{
26 std::vector<const char *> r;
27 r.reserve(arg_indices.size() + 1);
28
29 // First store nul terminator for each part:
30 for (auto index_pair : arg_indices) {
31 if (index_pair.second < s.length()) {
32 s[index_pair.second] = 0;
33 }
34 }
35
36 // Now we can get the C string (c_str) and store offsets into it:
37 const char * cstr = s.c_str();
38 for (auto index_pair : arg_indices) {
39 r.push_back(cstr + index_pair.first);
40 }
41 r.push_back(nullptr);
42 return r;
43}

47 tracking_child = true;

103

122 // Somehow the process managed to complete before we even saw the exec() status.

321void process_service::handle_exit_status() noexcept
322{
323 bool did_exit = exit_status.did_exit();
324 bool was_signalled = exit_status.was_signalled();
325 auto current_state = get_state();
326
327 if (notification_fd != -1) {
328 readiness_watcher.deregister(event_loop);
329 bp_sys::close(notification_fd);
330 notification_fd = -1;
331 }
332
333 if (!exit_status.did_exit_clean() && current_state != service_state_t::STOPPING) {
334 if (did_exit) {
335 log(loglevel_t::ERROR, "Service ", get_name(), " process terminated with exit code ",
336 exit_status.get_exit_status());
337 }
338 else if (was_signalled) {
339 log(loglevel_t::ERROR, "Service ", get_name(), " terminated due to signal ",
340 exit_status.get_signal());
341 }
342 }
343
344 if (waiting_stopstart_timer) {
345 process_timer.stop_timer(event_loop);
346 waiting_stopstart_timer = false;
347 if (current_state == service_state_t::STARTED) {
348 // Must have been in smooth recovery and waiting for readiness notification.
349 // Treat this the same as if we were STARTING:
350 current_state = service_state_t::STARTING;
351 }
352 }
353
354#if USE_UTMPX
355 if (*inittab_id || *inittab_line) {
356 clear_utmp_entry(inittab_id, inittab_line);
357 }
358#endif
359
360 if (current_state == service_state_t::STARTING) {
361 // If state is STARTING, we must be waiting for readiness notification; the process has
362 // terminated before becoming ready.
363 stop_reason = stopped_reason_t::FAILED;
364 failed_to_start();
365 }
366 else if (current_state == service_state_t::STOPPING) {
367 // We won't log a non-zero exit status or termination due to signal here -
368 // we assume that the process died because we signalled it.
369 if (waiting_stopstart_timer) {
370 process_timer.stop_timer(event_loop);
371 }
372 if (!waiting_for_deps) {
373 if (stop_pid == -1 && !waiting_for_execstat) {
374 stop_issued = false; // reset for next time
375 stopped();
376 }
377 }
378 else if (get_target_state() == service_state_t::STARTED && !pinned_stopped) {
379 initiate_start();
380 }
381 }
382 else if (smooth_recovery && current_state == service_state_t::STARTED && check_restart()) {
383 // unexpected termination, with smooth recovery
384 doing_smooth_recovery = true;
385 do_smooth_recovery();
386 return;
387 }
388 else {
389 handle_unexpected_termination();
390 }
391 services->process_queues();
392}

423void bgproc_service::handle_exit_status() noexcept
424{
425 // For bgproc services, receiving exit status can mean one of two things:
426 // 1. We were launching the process, and it finished (possibly after forking). If it did fork
427 // we want to obtain the process id of the process that we should now monitor, the actual
428 // daemon. Or,
429 // 2. The above has already happened, and we are monitoring the daemon process, which has now
430 // terminated for some reason.
431
432 begin:
433 bool did_exit = exit_status.did_exit();
434 bool was_signalled = exit_status.was_signalled();
435 auto current_state = get_state();
436
437 if (!exit_status.did_exit_clean() && current_state != service_state_t::STOPPING) {
438 if (did_exit) {
439 log(loglevel_t::ERROR, "Service ", get_name(), " process terminated with exit code ",
440 exit_status.get_exit_status());
441 }
442 else if (was_signalled) {
443 log(loglevel_t::ERROR, "Service ", get_name(), " terminated due to signal ",
444 exit_status.get_signal());
445 }
446 }
447
448 if (waiting_stopstart_timer) {
449 process_timer.stop_timer(event_loop);
450 waiting_stopstart_timer = false;
451 }
452
453 if (doing_smooth_recovery) {
454 doing_smooth_recovery = false;
455
456 // We're either started, or stopping (i.e. we were requested to stop during smooth recovery).
457
458 if (current_state == service_state_t::STOPPING) {
459 // Stop was issued during smooth recovery
460 if ((did_exit && exit_status.get_exit_status() != 0) || was_signalled) {
461 if (!waiting_for_deps) {
462 stopped();
463 }
464 else if (get_target_state() == service_state_t::STARTED && !pinned_stopped) {
465 initiate_start();
466 }
467 }
468 else {
469 // We need to re-read the PID, since it has now changed.
470 if (pid_file.length() != 0) {
471 auto pid_result = read_pid_file(&exit_status);
472
473 if (waiting_for_deps) {
474 // don't do anything else until dependents have stopped
475 return;
476 }
477
478 switch (pid_result) {
479 case pid_result_t::FAILED:
480 case pid_result_t::TERMINATED:
481 // Failed startup: no auto-restart.
482 stopped();
483 break;
484 case pid_result_t::OK:
485 // We now need to bring down the daemon process
486 bring_down();
487 break;
488 }
489 }
490 }
491 services->process_queues();
492 return;
493 }
494 else /* if (service_state == service_state_t::STARTED) */ {
495 bool need_stop = false;
496 if ((did_exit && exit_status.get_exit_status() != 0) || was_signalled) {
497 need_stop = true;
498 }
499 else {
500 // We need to re-read the PID, since it has now changed.
501 if (pid_file.length() != 0) {
502 auto pid_result = read_pid_file(&exit_status);
503
504 switch (pid_result) {
505 case pid_result_t::FAILED:
506 // Failed startup: no auto-restart.
507 need_stop = true;
508 break;
509 case pid_result_t::TERMINATED:
510 goto begin;
511 case pid_result_t::OK:
512 break;
513 }
514 }
515 }
516
517 if (need_stop) {
518 // Failed startup: no auto-restart.
519 stop_reason = stopped_reason_t::TERMINATED;
520 unrecoverable_stop();
521 services->process_queues();
522 }
523
524 return;
525 }
526 }
527
528 if (current_state == service_state_t::STARTING) {
529 if (exit_status.did_exit_clean()) {
530 auto pid_result = read_pid_file(&exit_status);
531 switch (pid_result) {
532 case pid_result_t::FAILED:
533 // Failed startup: no auto-restart.
534 stop_reason = stopped_reason_t::FAILED;
535 failed_to_start();
536 break;
537 case pid_result_t::TERMINATED:
538 // started, but immediately terminated
539 started();
540 goto begin;
541 case pid_result_t::OK:
542 started();
543 break;
544 }
545 }
546 else {
547 stop_reason = stopped_reason_t::FAILED;
548 failed_to_start();
549 }
550 }
551 else if (current_state == service_state_t::STOPPING) {
552 // We won't log a non-zero exit status or termination due to signal here -
553 // we assume that the process died because we signalled it.
554 if (stop_pid == -1 && !waiting_for_execstat) {
555 stop_issued = false;
556 stopped();
557 }
558 }
559 else {
560 // we must be STARTED
561 if (smooth_recovery && get_target_state() == service_state_t::STARTED && check_restart()) {
562 doing_smooth_recovery = true;
563 do_smooth_recovery();
564 if (get_state() != service_state_t::STARTED) {
565 doing_smooth_recovery = false;
566 }
567 return;
568 }
569 handle_unexpected_termination();
570 }
571 services->process_queues();
572}

597void scripted_service::handle_exit_status() noexcept
598{
599 bool did_exit = exit_status.did_exit();
600 bool was_signalled = exit_status.was_signalled();
601 auto current_state = get_state();
602
603 // For a scripted service, a termination occurs in one of three main cases:
604 // - the start script completed (or failed), when service was STARTING
605 // - the start script was interrupted to cancel startup; state is STOPPING
606 // - the stop script complete (or failed), state is STOPPING
607
608 if (current_state == service_state_t::STOPPING) {
609 // We might be running the stop script, or we might be running the start script and have issued
610 // a cancel order via SIGINT:
611 if (interrupting_start) {
612 if (waiting_stopstart_timer) {
613 process_timer.stop_timer(event_loop);
614 waiting_stopstart_timer = false;
615 }
616 // We issued a start interrupt, so we expected this failure:
617 if (did_exit && exit_status.get_exit_status() != 0) {
618 log(loglevel_t::NOTICE, "Service ", get_name(), " start cancelled; exit code ",
619 exit_status.get_exit_status());
620 // Assume that a command terminating normally (with failure status) requires no cleanup:
621 stopped();
622 }
623 else {
624 if (was_signalled) {
625 log(loglevel_t::NOTICE, "Service ", get_name(), " start cancelled from signal ",
626 exit_status.get_signal());
627 }
628 // If the start script completed successfully, or was interrupted via our signal,
629 // we want to run the stop script to clean up:
630 bring_down();
631 }
632 interrupting_start = false;
633 }
634 else if (exit_status.did_exit_clean()) {
635 // We were running the stop script and finished successfully
636 stopped();
637 }
638 else {
639 // ??? failed to stop! Let's log it as warning:
640 if (did_exit) {
641 log(loglevel_t::WARN, "Service ", get_name(), " stop command failed with exit code ",
642 exit_status.get_exit_status());
643 }
644 else if (was_signalled) {
645 log(loglevel_t::WARN, "Service ", get_name(), " stop command terminated due to signal ",
646 exit_status.get_signal());
647 }
648 // Even if the stop script failed, assume that service is now stopped, so that any dependencies
649 // can be stopped. There's not really any other useful course of action here.
650 stopped();
651 }
652 services->process_queues();
653 }
654 else { // STARTING
655 if (exit_status.did_exit_clean()) {
656 started();
657 }
658 else if (was_signalled && exit_status.get_signal() == SIGINT && onstart_flags.skippable) {
659 // A skippable service can be skipped by interrupting (eg by ^C if the service
660 // starts on the console).
661 start_skipped = true;
662 started();
663 }
664 else {
665 // failed to start
666 if (did_exit) {
667 log(loglevel_t::ERROR, "Service ", get_name(), " command failed with exit code ",
668 exit_status.get_exit_status());
669 }
670 else if (was_signalled) {
671 log(loglevel_t::ERROR, "Service ", get_name(), " command terminated due to signal ",
672 exit_status.get_signal());
673 }
674 stop_reason = stopped_reason_t::FAILED;
675 failed_to_start();
676 }
677 services->process_queues();
678 }

704bgproc_service::read_pid_file(proc_status_t *exit_status) noexcept
705{
706 const char *pid_file_c = pid_file.c_str();
707 int fd = bp_sys::open(pid_file_c, O_CLOEXEC);
708 if (fd == -1) {
709 log(loglevel_t::ERROR, get_name(), ": read pid file: ", strerror(errno));
710 return pid_result_t::FAILED;
711 }
712
713 char pidbuf[21]; // just enough to hold any 64-bit integer
714 int r = complete_read(fd, pidbuf, 20);
715 if (r < 0) {
716 // Could not read from PID file
717 log(loglevel_t::ERROR, get_name(), ": could not read from pidfile; ", strerror(errno));
718 bp_sys::close(fd);
719 return pid_result_t::FAILED;
720 }
721
722 bp_sys::close(fd);
723 pidbuf[r] = 0; // store nul terminator
724
725 bool valid_pid = false;
726 errno = 0;
727 char *eptr;
728 unsigned long long v = std::strtoull(pidbuf, &eptr, 0);
729 if (errno == 0 && eptr != pidbuf) {
730 if (v <= make_unsigned_val(std::numeric_limits<pid_t>::max())) {
731 pid = (pid_t) v;
732 valid_pid = true;
733 }
734 }
735
736 if (valid_pid) {
737 siginfo_t child_info;
738 child_info.si_pid = 0; // for portability
739 int waitid_r = bp_sys::waitid(P_PID, pid, &child_info, WNOHANG | WEXITED);
740 if (waitid_r == -1 && errno == ECHILD) {
741 // We can't track this child - check process exists:
742 if (bp_sys::kill(pid, 0) == 0 || errno != ESRCH) {
743 log(loglevel_t::WARN, get_name(), ": unable to supervise process with pid ", pid);
744 tracking_child = false;
745 return pid_result_t::OK;
746 }
747
748 log(loglevel_t::ERROR, get_name(), ": pid read from pidfile (", pid, ") is not valid");
749 pid = -1;
750 return pid_result_t::FAILED;
751 }
752 else if (waitid_r == 0) {
753 if (child_info.si_pid != 0) {
754 pid = -1;
755 *exit_status = proc_status_t(child_info.si_code, child_info.si_status);
756 return pid_result_t::TERMINATED;
757 }
758
759 // Child hasn't terminated, but we can track it (no error)
760 child_listener.add_reserved(event_loop, pid, dasynq::DEFAULT_PRIORITY - 10);
761 tracking_child = true;
762 reserved_child_watch = true;
763 return pid_result_t::OK;
764 }
765 }
766
767 log(loglevel_t::ERROR, get_name(), ": pid read from pidfile (", pid, ") is not valid");
768 pid = -1;
769 return pid_result_t::FAILED;
770}

772void process_service::bring_down() noexcept
773{
774 if (stop_pid != -1 || stop_issued) {
775 // waiting for stop command to complete (or for process to die after it has complete);
776 // can't do anything here.
777 return;
778 }
779 if (waiting_for_execstat) {
780 // The process is still starting. This should be uncommon, but can occur during
781 // smooth recovery (or it may mean the stop command process is still starting). We can't
782 // do much now; we have to wait until we get the status, and then act appropriately.
783 return;
784 }
785 else if (pid != -1) {
786 // The process is still kicking on - must actually kill it.
787 if (!stop_command.empty() && !stop_issued) {
788 if (start_stop_process(stop_arg_parts)) {
789 goto arm_timer;
790 }
791
792 // stop-command failed, need to try something else:
793 if (term_signal != 0) {
794 kill_pg(term_signal);
795 }
796 else {
797 kill_pg(SIGKILL);
798 }
799 }
800 else if (term_signal != 0) {
801 // We signal the process group (-pid) rather than just the process as there's less
802 // risk then of creating an orphaned process group:
803 kill_pg(term_signal);
804 }
805
806 if (stop_pid == -1 && !tracking_child) {
807 // If we have no way of tracking when the child terminates, assume stopped now
808 stopped();
809 return;
810 }
811
812 arm_timer:
813
814 stop_issued = true; // (don't try again)
815
816 // If there's a stop timeout, arm the timer now:
817 if (stop_timeout != time_val(0,0)) {
818 process_timer.arm_timer_rel(event_loop, stop_timeout);
819 waiting_stopstart_timer = true;
820 }
821
822 // The rest is done in handle_exit_status.
823 }
824 else {
825 // The process is already dead (possibly, we are in smooth recovery waiting for timer)
826 doing_smooth_recovery = false;
827 if (waiting_restart_timer) {
828 process_timer.stop_timer(event_loop);
829 waiting_restart_timer = false;
830 }
831 stopped();
832 }
833}

883bool process_service::start_stop_process(const std::vector<const char *> &cmd) noexcept
884{
885 // In general, you can't tell whether fork/exec is successful. We use a pipe to communicate
886 // success/failure from the child to the parent. The pipe is set CLOEXEC so a successful
887 // exec closes the pipe, and the parent sees EOF. If the exec is unsuccessful, the errno
888 // is written to the pipe, and the parent can read it.
889
890 int pipefd[2];
891 if (bp_sys::pipe2(pipefd, O_CLOEXEC)) {
892 log(loglevel_t::ERROR, get_name(), ": can't create status check pipe (for stop command): ",
893 strerror(errno));
894 return false;
895 }
896
897 const char * logfile = this->logfile.c_str();
898 if (*logfile == 0) {
899 logfile = "/dev/null";
900 }
901
902 bool child_status_registered = false;
903
904 // Set up complete, now fork and exec:
905
906 pid_t forkpid;
907
908 try {
909 stop_pipe_watcher.add_watch(event_loop, pipefd[0], dasynq::IN_EVENTS);
910 child_status_registered = true;
911
912 // We specify a high priority (i.e. low priority value) so that process termination is
913 // handled early. This means we have always recorded that the process is terminated by the
914 // time that we handle events that might otherwise cause us to signal the process, so we
915 // avoid sending a signal to an invalid (and possibly recycled) process ID.
916 forkpid = stop_watcher.fork(event_loop, reserved_stop_watch, dasynq::DEFAULT_PRIORITY - 10);
917 reserved_stop_watch = true;
918 }
919 catch (std::exception &e) {
920 log(loglevel_t::ERROR, get_name(), ": could not fork (for stop command): ", e.what());
921 goto out_cs_h;
922 }
923
924 if (forkpid == 0) {
925 close(pipefd[0]);
926 const char * working_dir_c = service_dsc_dir;
927 if (! working_dir.empty()) working_dir_c = working_dir.c_str();
928 run_proc_params run_params{cmd.data(), working_dir_c, logfile, pipefd[1], run_as_uid, run_as_gid, rlimits};
929 run_params.on_console = false;
930 run_params.in_foreground = false;
931 run_params.csfd = -1;
932 run_params.socket_fd = socket_fd;
933 run_params.notify_fd = -1;
934 run_params.force_notify_fd = -1;
935 run_params.notify_var = nullptr;
936 run_params.env_file = env_file.c_str();
937 run_params.nice_is_set = nice_is_set;
938 run_params.nice = nice;
939 #if SUPPORT_CGROUPS
940 run_params.run_in_cgroup = run_in_cgroup.c_str();
941 #endif
942 #if SUPPORT_CAPABILITIES
943 run_params.cap_iab = cap_iab.get();
944 run_params.secbits = secbits;
945 run_params.no_new_privs = onstart_flags.no_new_privs;
946 #endif
947 #if SUPPORT_IOPRIO
948 run_params.ioprio = ioprio;
949 #endif
950 #if SUPPORT_OOM_ADJ
951 run_params.oom_adj_is_set = oom_adj_is_set;
952 run_params.oom_adj = oom_adj;
953 #endif
954 run_child_proc(run_params);
955 }
956 else {
957 // Parent process
958 stop_pid = forkpid;
959
960 bp_sys::close(pipefd[1]); // close the 'other end' fd
961 waiting_for_execstat = true;
962 return true;
963 }
964
965 // Failure exit:
966
967 out_cs_h:
968 if (child_status_registered) {
969 stop_pipe_watcher.deregister(event_loop);
970 }
971
972 bp_sys::close(pipefd[0]);
973 bp_sys::close(pipefd[1]);
974
975 return false;
976}
