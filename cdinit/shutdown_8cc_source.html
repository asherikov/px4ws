142 // Write buffer contents out to file descriptor. The descriptor is assumed to be non-blocking.
143 // returns ENDFILE if there is no more content to flush (buffer is now empty) or OK otherwise.
144 fill_status flush(int fd)
145 {
146 int to_write = get_contiguous_length(get_ptr(0));
147
148 if (overflow_marker != -1) {
149 if (overflow_marker == 0) {
150 // output (remainder of) overflow message
151 int l = std::strlen(overflow_msg_ptr);
152 int r = write(fd, overflow_msg_ptr, l);
153 if (r == l) {
154 // entire message has been written; next marker is in buffer
155 int16_t overflow_marker16;
156 extract(reinterpret_cast<char *>(&overflow_marker16), 0, sizeof(overflow_marker16));
157 overflow_marker = overflow_marker16;
158 consume(sizeof(overflow_marker16));
159
160 // no more overflow markers?
161 if (overflow_marker == -1) {
162 last_overflow = -1;
163 }
164 return get_length() == 0 ? fill_status::ENDFILE : fill_status::OK;
165 }
166 if (r > 0) {

168 }
169 return fill_status::OK;
170 }
171
172 to_write = std::min(to_write, overflow_marker);
173 }
174
175 int r = write(fd, get_ptr(0), to_write);
176 if (r > 0) {
177 consume(r);
178 if (overflow_marker != -1) {
179 overflow_marker -= r;
180 last_overflow -= r;
181 if (overflow_marker == 0) {
182 overflow_msg_ptr = output_lost_msg;
183 }
184 }
185 }
186
187 return get_length() == 0 ? fill_status::ENDFILE : fill_status::OK;
188 }

190 // Mark overflow occurred. Call this only when the buffer is full.
191 // The marker is put after the most recent newline in the buffer, if possible, so that whole
192 // lines are retained in the buffer. In some cases marking overflow will not add a new overflow
193 // marker but simply trim the buffer to an existing marker.
194 void mark_overflow()
195 {
196 // Try to find the last newline in the buffer
197 int begin = 0;
198 if (last_overflow != -1) {
199 begin = last_overflow + sizeof(int16_t);
200 }
201 int end = get_length() - 1 - sizeof(int16_t); // -1, then -2 for storage of marker
202
203 int i;
204 for (i = end; i >= begin; i--) {
205 if ((*this)[i] == '\n') break;
206 }
207
208 if (last_overflow != -1 && i < begin) {
209 // No new line after existing marker: trim all beyond that marker, don't
210 // create a new marker:
211 trim_to(last_overflow + sizeof(uint16_t));
212 return;
213 }
214
215 if (i < begin) {
216 // No newline in the whole buffer... we'll put the overflow marker at the end,
217 // on the assumption that it is better to output a partial line than it is to
218 // discard the entire buffer:
219 last_overflow = get_length() - sizeof(int16_t);
220 overflow_marker = last_overflow;
221 int16_t overflow16 = -1;
222 char * overflow16_ptr = reinterpret_cast<char *>(&overflow16);
223 *get_ptr(last_overflow + 0) = overflow16_ptr[0];
224 *get_ptr(last_overflow + 1) = overflow16_ptr[1];
225 return;
226 }
227
228 // We found a newline, put the overflow marker just after it:
229 int new_overflow = i + 1;
230 if (last_overflow != -1) {
231 int16_t new_overflow16 = new_overflow;
232 char * new_overflow16_ptr = reinterpret_cast<char *>(&new_overflow16);
233 *get_ptr(last_overflow + 0) = new_overflow16_ptr[0];
234 *get_ptr(last_overflow + 1) = new_overflow16_ptr[1];
235 }
236 last_overflow = new_overflow;
237 if (overflow_marker == -1) {
238 overflow_marker = last_overflow;
239 }
240
241 int16_t overflow16 = -1;
242 char * overflow16_ptr = reinterpret_cast<char *>(&overflow16);
243 *get_ptr(last_overflow + 0) = overflow16_ptr[0];
244 *get_ptr(last_overflow + 0) = overflow16_ptr[1];
245 trim_to(last_overflow + sizeof(int16_t));
246 }

441// Actually shut down the system.
442void do_system_shutdown(shutdown_type_t shutdown_type)
443{
444 using namespace std;
445
446 // Mask all signals to prevent death of our parent etc from terminating us
447 sigset_t allsigs;
448 sigfillset(&allsigs);
449 sigprocmask(SIG_SETMASK, &allsigs, nullptr);
450
451 int reboot_type = RB_AUTOBOOT; // reboot
452 const char *shutdown_type_arg = "reboot";
453#if defined(RB_POWER_OFF)
454 if (shutdown_type == shutdown_type_t::POWEROFF) {
455 reboot_type = RB_POWER_OFF;
456 shutdown_type_arg = "poweroff";
457 }
458#elif defined(RB_POWEROFF)
459 // FreeBSD spells it slightly differently
460 if (shutdown_type == shutdown_type_t::POWEROFF) {
461 reboot_type = RB_POWEROFF;
462 shutdown_type_arg = "poweroff";
463 }
464#elif defined(RB_POWERDOWN)
465 // NetBSD (at least) uses RB_POWERDOWN rather than RB_POWER_OFF
466 if (shutdown_type == shutdown_type_t::POWEROFF) {
467 reboot_type = RB_POWERDOWN;
468 shutdown_type_arg = "poweroff";
469 }
470#endif
471#if defined(RB_HALT_SYSTEM)
472 // Linux
473 if (shutdown_type == shutdown_type_t::HALT) {
474 reboot_type = RB_HALT_SYSTEM;
475 shutdown_type_arg = "halt";
476 }
477#elif defined(RB_HALT)
478 // NetBSD, FreeBSD
479 if (shutdown_type == shutdown_type_t::HALT) {
480 reboot_type = RB_HALT;
481 shutdown_type_arg = "halt";
482 }
483#endif
484#if defined(RB_KEXEC)
485 if (shutdown_type == shutdown_type_t::KEXEC) reboot_type = RB_KEXEC;
486#endif
487
488 // Write to console rather than any terminal, since we lose the terminal it seems:
489 int consfd = open("/dev/console", O_WRONLY);
490 if (consfd != STDOUT_FILENO && consfd != -1) {
491 dup2(consfd, STDOUT_FILENO);
492 }
493
494 loop_t loop;
495 subproc_buffer sub_buf {loop, STDOUT_FILENO};
496
497 sub_buf.append("Sending TERM/KILL to all processes...\n");
498
499 // Send TERM/KILL to all (remaining) processes
500 kill(-1, SIGTERM);
501
502 // 1 second delay (while outputting from sub_buf):
503 bool timeout_reached = false;
504 dasynq::time_val timeout {1, 0};
505 dasynq::time_val interval {0,0};
506 loop_t::timer::add_timer(loop, clock_type::MONOTONIC, true /* relative */,
507 timeout.get_timespec(), interval.get_timespec(),
508 [&](loop_t &eloop, int expiry_count) -> rearm {
509
510 timeout_reached = true;
511 return rearm::REMOVE;
512 });
513
514 do {
515 loop.run();
516 } while (! timeout_reached);
517
518 kill(-1, SIGKILL);
519
520 // Attempt to execute shutdown hook at three possible locations:
521 const char * const hook_paths[] = {
522 "/etc/dinit/shutdown-hook",
523 "/lib/dinit/shutdown-hook"
524 };
525
526 bool do_unmount_ourself = true;
527 const int execmask = S_IXOTH | S_IXGRP | S_IXUSR;
528 struct stat statbuf;
529
530 for (size_t i = 0; i < sizeof(hook_paths) / sizeof(hook_paths[0]); ++i) {
531 int stat_r = lstat(hook_paths[i], &statbuf);
532 if (stat_r == 0 && (statbuf.st_mode & execmask) != 0) {
533 sub_buf.append("Executing shutdown hook...\n");
534 const char *prog_args[] = { hook_paths[i], shutdown_type_arg, nullptr };
535 try {
536 auto r = run_process(prog_args, loop, sub_buf);
537 if (r.did_exit() && r.get_exit_status() == 0) {
538 do_unmount_ourself = false;
539 }
540 }
541 catch (std::exception &e) {
542 sub_buf.append("Couldn't fork for shutdown-hook: ");
543 sub_buf.append(e.what());
544 sub_buf.append("\n");
545 }
546 break;
547 }
548 }
549
550 // perform shutdown
551 if (do_unmount_ourself) {
552 sub_buf.append("Turning off swap...\n");
553 swap_off(loop, sub_buf);
554 sub_buf.append("Unmounting disks...\n");
555 unmount_disks(loop, sub_buf);
556 }
557
558 sync();
559
560 sub_buf.append("Issuing shutdown via kernel...\n");
561 loop.poll(); // give message a chance to get to console
562#ifdef __NetBSD__
563 reboot(reboot_type, NULL);
564#else
565 if (reboot(reboot_type)) {
566 // we're in trouble now
567 sub_buf.append("reboot: ");
568 sub_buf.append(strerror(errno));
569 if (shutdown_type == shutdown_type_t::KEXEC) {
570 sub_buf.append(
571 "\nIt is possible that no suitable kernel image was loaded before"
572 "\nreboot with kexec was attempted.\n"
573 );
574 }
575 while (true) loop.run();
576 }
577#endif
578}

580// Watcher for subprocess output.
581class subproc_out_watch : public loop_t::fd_watcher_impl<subproc_out_watch>
582{
583 subproc_buffer &sub_buf;
584 bool in_overflow = false;
585
586 rearm read_overflow(int fd)
587 {
588 char buf[128];
589 int r = read(fd, buf, 128);
590 if (r == 0 || (r == -1 && errno != EAGAIN)) {
591 return rearm::NOOP; // leave disarmed
592 }
593 if (r == -1) {
594 return rearm::REARM;
595 }
596
597 // How much space is available?
598 int fr = sub_buf.get_free();
599 for (int b = r - std::min(r, fr); b < r; b++) {
600 if (buf[b] == '\n') {
601 // Copy the (partial) line into sub_buf and leave overflow mode
602 sub_buf.append(buf + b, r - b);
603 in_overflow = false;
604 }
605 }
606 return rearm::REARM;
607 }
608
609 public:
610 subproc_out_watch(subproc_buffer &sub_buf_p) : sub_buf(sub_buf_p) {}
611
612 rearm fd_event(loop_t &, int fd, int flags)
613 {
614 // if current status is reading overflow, read and discard until newline
615 if (in_overflow) {
616 return read_overflow(fd);
617 }
618
619 auto r = sub_buf.fill(fd);
620 if (r == subproc_buffer::fill_status::FULL) {
621 sub_buf.mark_overflow();
622 in_overflow = true;
623 return read_overflow(fd);
624 }
625 else if (r == subproc_buffer::fill_status::ENDFILE) {
626 return rearm::NOOP;
627 }
628
629 return rearm::REARM; // re-enable watcher
630 }
631};

633// Run process, put its output through the subprocess buffer
634// may throw: std::system_error, std::bad_alloc
635static loop_t::child_proc_watcher::proc_status_t run_process(const char * prog_args[],
636 loop_t &loop, subproc_buffer &sub_buf)
637{
638 class sp_watcher_t : public loop_t::child_proc_watcher_impl<sp_watcher_t>
639 {
640 public:
641 bool terminated = false;
642 proc_status_t exit_status;
643
644 rearm status_change(loop_t &, pid_t child, proc_status_t status)
645 {
646 terminated = true;
647 exit_status = status;
648 return rearm::REMOVE;
649 }
650 };
651
652 sp_watcher_t sp_watcher;
653
654 // Create output pipe
655 bool have_pipe = true;
656 int pipefds[2];
657 if (dasynq::pipe2(pipefds, O_NONBLOCK) == -1) {
658 sub_buf.append("Warning: ");
659 sub_buf.append(prog_args[0]);
660 sub_buf.append(": could not create pipe for subprocess output\n");
661 have_pipe = false;
662 // Note, we proceed and let the sub-process run with our stdout/stderr.
663 }
664
665 subproc_out_watch owatch {sub_buf};
666
667 if (have_pipe) {
668 try {
669 owatch.add_watch(loop, pipefds[0], dasynq::IN_EVENTS);
670 }
671 catch (...) {
672 // failed to create the watcher for the subprocess output; again, let it run with
673 // our stdout/stderr
674 sub_buf.append("Warning: could not create output watch for subprocess\n");
675 close(pipefds[0]);
676 close(pipefds[1]);
677 have_pipe = false;
678 }
679 }
680
681 // If we've buffered any messages/output, give them a chance to go out now:
682 loop.poll();
683
684 pid_t ch_pid = sp_watcher.fork(loop);
685 if (ch_pid == 0) {
686 // child
687 // Dup output pipe to stdout, stderr
688 if (have_pipe) {
689 dup2(pipefds[1], STDOUT_FILENO);
690 dup2(pipefds[1], STDERR_FILENO);
691 close(pipefds[0]);
692 close(pipefds[1]);
693 }
694 execv(prog_args[0], const_cast<char **>(prog_args));
695 puts("Failed to execute subprocess: ");
696 perror(prog_args[0]);
697 _exit(1);
698 }
699 if (have_pipe) {
700 close(pipefds[1]);
701 }
702
703 do {
704 loop.run();
705 } while (!sp_watcher.terminated);
706
707 if (have_pipe) {
708 owatch.deregister(loop);
709 close(pipefds[0]);
710 }
711
712 return sp_watcher.exit_status;
713}
