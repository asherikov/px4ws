31bool base_process_service::bring_up() noexcept
32{
33 if (!open_socket()) {
34 return false;
35 }
36
37 bool start_success;
38 if (in_auto_restart) {
39 start_success = restart_ps_process();
40 }
41 else {
42 restart_interval_count = 0;
43 start_success = start_ps_process(exec_arg_parts,

57 }
58 }

64void base_process_service::handle_unexpected_termination() noexcept
65{
66 // unexpected termination, with possible restart
67 stop_reason = stopped_reason_t::TERMINATED;
68
69 // We want to circumvent the normal process of waiting for dependents to stop before we
70 // attempt to restart, for two reasons:
71 // 1) we can restart more quickly
72 // 2) we can use the restart rate-limiting logic from restart_ps_process rather than
73 // the usual start_ps_process (the usual bring-up).
74 // But we need to issue a forced stop and process queues, to discover our eventual target
75 // state (so we know whether we actually want to restart or not).
76 // Note we can't call forced_stop() directly here, because we need to set in_auto_restart in
77 // between do_stop() and processing queues (so that it is set correctly if restart occurs):
78
79 force_stop = true;
80 do_stop();
81 services->process_queues();
82
83 if (get_state() == service_state_t::STOPPING) {
84 // We must be waiting for dependents;
85 // If we're going to restart, we can kick that off now:
86 if (get_target_state() == service_state_t::STARTED && !pinned_stopped) {
87 initiate_start();
88 services->process_queues();
89 }
90 }
91}

93bool base_process_service::start_ps_process(const std::vector<const char *> &cmd, bool on_console) noexcept
94{
95 // In general, you can't tell whether fork/exec is successful. We use a pipe to communicate
96 // success/failure from the child to the parent. The pipe is set CLOEXEC so a successful
97 // exec closes the pipe, and the parent sees EOF. If the exec is unsuccessful, the errno
98 // is written to the pipe, and the parent can read it.
99
100 event_loop.get_time(last_start_time, clock_type::MONOTONIC);
101
102 int pipefd[2];
103 if (bp_sys::pipe2(pipefd, O_CLOEXEC)) {
104 log(loglevel_t::ERROR, get_name(), ": can't create status check pipe: ", strerror(errno));
105 return false;
106 }
107
108 int input_fd;
109 if (!get_input_fd(&input_fd)) {
110 return false;
111 }
112
113 const char * logfile = this->logfile.c_str();
114 if (this->log_type == log_type_id::LOGFILE || this->log_type == log_type_id::NONE) {
115 // Note: if log_type == NONE, logfile should be empty.
116 if (*logfile == 0) {
117 logfile = "/dev/null";
118 }
119 }
120 else /* log_type_id::BUFFER or ::PIPE */ {
121 if (this->log_output_fd == -1) {
122 int logfd[2];
123 // Note: we set CLOEXEC on the file descriptors here; when the output file descriptor is dup'd
124 // to stdout, this will be effectively removed for the output end
125 if (bp_sys::pipe2(logfd, O_CLOEXEC)) {
126 log(loglevel_t::ERROR, get_name(), ": can't create output pipe: ", strerror(errno));
127 goto out_p;
128 }
129 this->log_input_fd = logfd[0];
130 this->log_output_fd = logfd[1];
131 if (this->log_type == log_type_id::BUFFER) {
132 try {
133 this->log_output_listener.add_watch(event_loop, logfd[0], dasynq::IN_EVENTS,
134 false /* not enabled */);
135 }
136 catch (...) {
137 log(loglevel_t::ERROR, get_name(), ": can't add output watch (insufficient resources)");
138 bp_sys::close(this->log_input_fd);
139 bp_sys::close(this->log_output_fd);
140 this->log_input_fd = -1;
141 this->log_output_fd = -1;
142 goto out_p;
143 }
144 }
145 }
146 // (More is done below, after we have performed additional setup)
147 }
148
149 {
150 bool child_status_registered = false;
151 control_conn_t *control_conn = nullptr;
152
153 int control_socket[2] = {-1, -1};
154 int notify_pipe[2] = {-1, -1};
155 bool have_notify = !notification_var.empty() || force_notification_fd != -1;
156 ready_notify_watcher * rwatcher = have_notify ? get_ready_watcher() : nullptr;
157 bool ready_watcher_registered = false;
158
159 if (onstart_flags.pass_cs_fd) {
160 if (dinit_socketpair(AF_UNIX, SOCK_STREAM, /* protocol */ 0, control_socket, SOCK_NONBLOCK)) {
161 log(loglevel_t::ERROR, get_name(), ": can't create control socket: ", strerror(errno));
162 goto out_lfd;
163 }
164
165 // Make the server side socket close-on-exec:
166 int fdflags = bp_sys::fcntl(control_socket[0], F_GETFD);
167 bp_sys::fcntl(control_socket[0], F_SETFD, fdflags | FD_CLOEXEC);
168
169 try {
170 control_conn = new control_conn_t(event_loop, services, control_socket[0]);
171 }
172 catch (std::exception &exc) {
173 log(loglevel_t::ERROR, get_name(), ": can't launch process; out of memory");
174 goto out_cs;
175 }
176 }
177
178 if (have_notify) {
179 // Create a notification pipe:
180 if (bp_sys::pipe2(notify_pipe, 0) != 0) {
181 log(loglevel_t::ERROR, get_name(), ": can't create notification pipe: ", strerror(errno));
182 goto out_cs_h;
183 }
184
185 // Set the read side as close-on-exec:
186 int fdflags = bp_sys::fcntl(notify_pipe[0], F_GETFD);
187 bp_sys::fcntl(notify_pipe[0], F_SETFD, fdflags | FD_CLOEXEC);
188
189 // add, but don't yet enable, readiness watcher:
190 try {
191 rwatcher->add_watch(event_loop, notify_pipe[0], dasynq::IN_EVENTS, false);
192 ready_watcher_registered = true;
193 }
194 catch (std::exception &exc) {
195 log(loglevel_t::ERROR, get_name(), ": can't add notification watch: ", exc.what());
196 goto out_cs_h;
197 }
198 }
199
200 if (log_type == log_type_id::BUFFER) {
201 // Set watcher enabled if space in buffer
202 if (log_buf_size > 0) {
203 // Append a "restarted" message to buffer contents
204 const char *restarting_msg = "\n(dinit: note: service restarted)\n";
205 unsigned restarting_msg_len = strlen(restarting_msg);
206 bool trailing_nl = log_buffer[log_buf_size - 1] == '\n';
207 if (trailing_nl) {
208 ++restarting_msg; // trim leading newline
209 --restarting_msg_len;
210 }
211 if (log_buf_size + restarting_msg_len >= log_buf_max) {
212 goto skip_enable_log_watch;
213 }
214 if (!ensure_log_buffer_backing(log_buf_size + restarting_msg_len)) {
215 goto skip_enable_log_watch;
216 }
217 memcpy(log_buffer.data() + log_buf_size, restarting_msg, restarting_msg_len);
218 log_buf_size += restarting_msg_len;
219 }
220 log_output_listener.set_enabled(event_loop, true);
221 }
222 skip_enable_log_watch: ;
223
224 // Set up complete, now fork and exec:
225
226 pid_t forkpid;
227
228 try {
229 child_status_listener.add_watch(event_loop, pipefd[0], dasynq::IN_EVENTS);
230 child_status_registered = true;
231
232 // We specify a high priority (i.e. low priority value) so that process termination is
233 // handled early. This means we have always recorded that the process is terminated by the
234 // time that we handle events that might otherwise cause us to signal the process, so we
235 // avoid sending a signal to an invalid (and possibly recycled) process ID.
236 forkpid = child_listener.fork(event_loop, reserved_child_watch, dasynq::DEFAULT_PRIORITY - 10);
237 reserved_child_watch = true;
238 }
239 catch (std::exception &e) {
240 log(loglevel_t::ERROR, get_name(), ": could not fork: ", e.what());
241 goto out_cs_h;
242 }
243
244 if (forkpid == 0) {
245 const char * working_dir_c = service_dsc_dir;
246 if (!working_dir.empty()) working_dir_c = working_dir.c_str();
247 after_fork(getpid());
248 run_proc_params run_params{cmd.data(), working_dir_c, logfile, pipefd[1], run_as_uid, run_as_gid, rlimits};
249 run_params.on_console = on_console;
250 run_params.in_foreground = !onstart_flags.shares_console;
251 run_params.unmask_sigint = onstart_flags.unmask_intr;
252 run_params.csfd = control_socket[1];
253 run_params.socket_fd = socket_fd;
254 run_params.notify_fd = notify_pipe[1];
255 run_params.force_notify_fd = force_notification_fd;
256 run_params.notify_var = notification_var.c_str();
257 run_params.env_file = env_file.c_str();
258 run_params.output_fd = log_output_fd;
259 run_params.input_fd = input_fd;
260 run_params.nice_is_set = nice_is_set;
261 run_params.nice = nice;
262 #if SUPPORT_CGROUPS
263 run_params.run_in_cgroup = run_in_cgroup.c_str();
264 #endif
265 #if SUPPORT_CAPABILITIES
266 run_params.cap_iab = cap_iab.get();
267 run_params.secbits = secbits;
268 run_params.no_new_privs = onstart_flags.no_new_privs;
269 #endif
270 #if SUPPORT_IOPRIO
271 run_params.ioprio = ioprio;
272 #endif
273 #if SUPPORT_OOM_ADJ
274 run_params.oom_adj_is_set = oom_adj_is_set;
275 run_params.oom_adj = oom_adj;
276 #endif
277 run_child_proc(run_params);
278 }
279 else {
280 // Parent process
281 pid = forkpid;
282
283 bp_sys::close(pipefd[1]); // close the 'other end' fd
284 if (control_socket[1] != -1) bp_sys::close(control_socket[1]);
285 if (notify_pipe[1] != -1) bp_sys::close(notify_pipe[1]);
286 notification_fd = notify_pipe[0];
287 waiting_for_execstat = true;
288 return true;
289 }
290
291 // Failure exit:
292
293 out_cs_h:
294 if (child_status_registered) {
295 child_status_listener.deregister(event_loop);
296 }
297
298 if (notify_pipe[0] != -1) bp_sys::close(notify_pipe[0]);
299 if (notify_pipe[1] != -1) bp_sys::close(notify_pipe[1]);
300 if (ready_watcher_registered) {
301 rwatcher->deregister(event_loop);
302 }
303
304 if (onstart_flags.pass_cs_fd) {
305 delete control_conn;
306
307 out_cs:
308 bp_sys::close(control_socket[0]);
309 bp_sys::close(control_socket[1]);
310 }
311 }
312
313 out_lfd:
314 if (log_input_fd != -1) {
315 log_output_listener.deregister(event_loop);
316 bp_sys::close(log_input_fd);
317 bp_sys::close(log_output_fd);
318 log_input_fd = -1;
319 log_output_fd = -1;
320 }
321
322 out_p:
323 bp_sys::close(pipefd[0]);
324 bp_sys::close(pipefd[1]);
325
326 return false;
327}

355void base_process_service::do_restart() noexcept
356{
357 // Actually perform process restart. We may be in smooth recovery (state = STARTED) or this may
358 // be a regular restart.
359
360 waiting_restart_timer = false;
361 auto service_state = get_state();
362
363 if (!start_ps_process(exec_arg_parts, have_console || onstart_flags.shares_console)) {
364 if (service_state == service_state_t::STARTING) {
365 failed_to_start();
366 }
367 else {
368 // smooth recovery failure
369 unrecoverable_stop();
370 }
371 services->process_queues();
372 }
373 else {
374 // started process successfully (at least as far as fork)
375 if (start_timeout != time_val(0,0)) {
376 process_timer.arm_timer_rel(event_loop, start_timeout);
377 waiting_stopstart_timer = true;
378 }
379 }
380}

382bool base_process_service::restart_ps_process() noexcept
383{
384 using time_val = dasynq::time_val;
385
386 time_val current_time;
387 event_loop.get_time(current_time, clock_type::MONOTONIC);
388
389 // Check if enough time has lapsed since the previous restart. If not, start a timer:
390 time_val tdiff = current_time - last_start_time;
391 if (restart_delay <= tdiff) {
392 // > restart delay (normally 200ms)
393 do_restart();
394 }
395 else {
396 time_val timeout = restart_delay - tdiff;
397 process_timer.arm_timer_rel(event_loop, timeout);
398 waiting_restart_timer = true;
399 }
400 return true;
401}

404{

438void base_process_service::kill_pg(int signo) noexcept
439{
440 if (onstart_flags.signal_process_only) {
441 bp_sys::kill(pid, signo);
442 }
443 else {
444 pid_t pgid = bp_sys::getpgid(pid);
445 if (pgid == -1) {
446 // On some OSes (eg OpenBSD) we aren't allowed to get the pgid of a process in a different
447 // session. If the process is in a different session, however, it must be a process group
448 // leader and the pgid must equal the process id.
449 pgid = pid;
450 }
451 bp_sys::kill(-pgid, signo);

455void base_process_service::timer_expired() noexcept
456{
457 waiting_stopstart_timer = false;
458
459 // Timer expires if:
460 // We are stopping, including after having startup cancelled (stop timeout, state is STOPPING); We are
461 // starting (start timeout, state is STARTING); We are waiting for restart timer before restarting,
462 // including smooth recovery (restart timeout, state is STARTING or STARTED).
463 if (get_state() == service_state_t::STOPPING) {
464 kill_with_fire();
465 }
466 else if (pid != -1) {
467 // Starting, start timed out.
468 log(loglevel_t::WARN, "Service ", get_name(), " with pid ", pid,
469 " exceeded allowed start time; cancelling.");
470 interrupt_start();
471 stop_reason = stopped_reason_t::TIMEDOUT;
472 failed_to_start(false, false);
473 }
474 else {
475 // STARTING / STARTED, and we have no pid: must be restarting (smooth recovery if STARTED)
476 do_restart();
477 }
478}

488bool base_process_service::open_socket() noexcept
489{
490 if (socket_path.empty() || socket_fd != -1) {
491 // No socket, or already open
492 return true;
493 }
494
495 const char * saddrname = socket_path.c_str();
496
497 // Check the specified socket path
498 struct stat stat_buf;
499 if (stat(saddrname, &stat_buf) == 0) {
500 if ((stat_buf.st_mode & S_IFSOCK) == 0) {
501 // Not a socket
502 log(loglevel_t::ERROR, get_name(), ": activation socket file exists (and is not a socket)");
503 return false;
504 }
505 }
506 else if (errno != ENOENT) {
507 // Other error
508 log(loglevel_t::ERROR, get_name(), ": error checking activation socket: ", strerror(errno));
509 return false;
510 }
511
512 // Remove stale socket file (if it exists).
513 // We won't test the return from unlink - if it fails other than due to ENOENT, we should get an
514 // error when we try to create the socket anyway.
515 unlink(saddrname);
516
517 uint sockaddr_size = offsetof(struct sockaddr_un, sun_path) + socket_path.length() + 1;
518 struct sockaddr_un * name = static_cast<sockaddr_un *>(malloc(sockaddr_size));
519 if (name == nullptr) {
520 log(loglevel_t::ERROR, get_name(), ": opening activation socket: out of memory");
521 return false;
522 }
523
524 name->sun_family = AF_UNIX;
525 strcpy(name->sun_path, saddrname);
526
527 int sockfd = dinit_socket(AF_UNIX, SOCK_STREAM, 0, SOCK_NONBLOCK | SOCK_CLOEXEC);
528 if (sockfd == -1) {
529 log(loglevel_t::ERROR, get_name(), ": error creating activation socket: ", strerror(errno));
530 free(name);
531 return false;
532 }
533
534 if (bind(sockfd, (struct sockaddr *) name, sockaddr_size) == -1) {
535 log(loglevel_t::ERROR, get_name(), ": error binding activation socket: ", strerror(errno));
536 close(sockfd);
537 free(name);
538 return false;
539 }
540
541 free(name);
542
543 // POSIX (1003.1, 2013) says that fchown and fchmod don't necessarily work on sockets. We have to
544 // use chown and chmod instead.
545 if (chown(saddrname, socket_uid, socket_gid)) {
546 log(loglevel_t::ERROR, get_name(), ": error setting activation socket owner/group: ",
547 strerror(errno));
548 close(sockfd);
549 return false;
550 }
551
552 if (chmod(saddrname, socket_perms) == -1) {
553 log(loglevel_t::ERROR, get_name(), ": Error setting activation socket permissions: ",
554 strerror(errno));
555 close(sockfd);
556 return false;
557 }
558
559 if (listen(sockfd, 128) == -1) { // 128 "seems reasonable".
560 log(loglevel_t::ERROR, ": error listening on activation socket: ", strerror(errno));
561 close(sockfd);
562 return false;
563 }
564
565 socket_fd = sockfd;
566 return true;
567}

569bool base_process_service::ensure_log_buffer_backing(unsigned new_size) noexcept
570{
571 // Note: we manage capacity manually to avoid it exceeding maximum
572 if (log_buffer.size() < new_size) {
573 if (log_buffer.capacity() < new_size) {
574 try {
575 unsigned new_capacity = std::max((unsigned)log_buffer.capacity() * 2, new_size);
576 new_capacity = std::min(new_capacity, log_buf_max);
577 log_buffer.reserve(new_capacity);
578 log_buffer.resize(new_capacity);

580 catch (std::bad_alloc &badalloc) {

582 return false;
583 }
584 }
