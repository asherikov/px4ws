95
96 for (int i = 0; i <= max_fd; i++) {
97 if (FD_ISSET(i, read_set_p)) {

276 // If events are pending, process an unspecified number of them.
277 // If no events are pending, wait until one event is received and
278 // process this event (and possibly any other events received
279 // simultaneously).
280 // If processing an event removes a watch, there is a possibility
281 // that the watched event will still be reported (if it has
282 // occurred) before pull_events() returns.
283 //
284 // do_wait - if false, returns immediately if no events are
285 // pending.
286 void pull_events(bool do_wait) noexcept
287 {
288 struct timeval ts;
289 struct timeval *wait_ts = nullptr;
290
291 Base::lock.lock();
292
293 // Check whether any timers are pending, and what the next timeout is.
294 // Check whether any timers are pending, and what the next timeout is.
295 this->process_monotonic_timers(do_wait, ts, wait_ts);
296
297 fd_set read_set_c;
298 fd_set write_set_c;
299 fd_set err_set; // "exceptional conditions" eg urgent data
300
301 read_set_c = read_set;
302 write_set_c = write_set;
303 err_set = read_set;
304
305 const sigset_t &active_sigmask = this->get_active_sigmask();
306
307 int nfds = max_fd + 1;
308 Base::lock.unlock();
309
310 // using sigjmp/longjmp is ugly, but there is no other way. If a signal that we're watching is
311 // received during polling, it will longjmp back to here:
312 if (sigsetjmp(this->get_sigreceive_jmpbuf(), 1) != 0) {
313 this->process_signal();
314 do_wait = false;
315 }
316
317 if (!do_wait) {
318 ts.tv_sec = 0;
319 ts.tv_usec = 0;
320 wait_ts = &ts;
321 }
322
323 std::atomic_signal_fence(std::memory_order::memory_order_release);
324
325 this->sigmaskf(SIG_UNBLOCK, &active_sigmask, nullptr);
326 int r = select(nfds, &read_set_c, &write_set_c, &err_set, wait_ts);
327 // Note, a signal may be received here and the handler may perform siglongjmp to the above
328 // established jmpbuf; that means we will execute the select statement again, but that's fine.
329 this->sigmaskf(SIG_BLOCK, &active_sigmask, nullptr);
330
331 if (r == -1 || r == 0) {
332 // signal or no events
333 if (r == 0 && do_wait) {
334 // timeout:
335 Base::lock.lock();
336 this->process_monotonic_timers();
337 Base::lock.unlock();
338 }
339
340 return;
341 }
342
343 process_events(&read_set_c, &write_set_c, &err_set);
344 }
