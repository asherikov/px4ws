94static void get_remote_env(int csfd, cpbuffer_t &rbuffer)
95{
96 char buf[2] = { (char)cp_cmd::GETALLENV, 0 };
97 write_all_x(csfd, buf, 2);
98
99 wait_for_reply(rbuffer, csfd);
100
101 if (rbuffer[0] != (char)cp_rply::ALLENV) {
102 throw dinit_protocol_error();
103 }
104
105 // 1-byte packet header, then size_t data size
106 constexpr size_t allenv_hdr_size = 1 + sizeof(size_t);
107 rbuffer.fill_to(csfd, allenv_hdr_size);
108
109 size_t data_size;
110 rbuffer.extract(&data_size, 1, sizeof(data_size));
111 rbuffer.consume(allenv_hdr_size);
112
113 if (data_size == 0) return;
114
115 if (rbuffer.get_length() == 0) {
116 fill_some(rbuffer, csfd);
117 }
118
119 std::string env_var;
120
121 while (data_size > 0) {
122 // look for a nul terminator
123 get_var:
124 unsigned contig_len = rbuffer.get_contiguous_length(rbuffer.get_ptr(0));
125 unsigned check_len = std::min((size_t) contig_len, data_size);
126 for (unsigned i = 0; i < check_len; ++i) {
127 if (rbuffer[i] == '\0') {
128 // append the last portion
129 env_var.append(rbuffer.get_ptr(0), rbuffer.get_ptr(0) + i);
130 rbuffer.consume(i + 1);
131 data_size -= (i + 1);
132
133 menv.set_var(std::move(env_var));
134 env_var.clear();
135
136 if (data_size == 0) {
137 // that's the last one
138 return;
139 }
140
141 goto get_var;
142 }
143 }
144
145 // copy what we have so far to the string, and fill some more
146 env_var.append(rbuffer.get_ptr(0), rbuffer.get_ptr(0) + check_len);
147 rbuffer.consume(check_len);
148 data_size -= check_len;
149
150 if (data_size == 0) {
151 // This shouldn't happen, we didn't find the nul terminator at the end
152 throw dinit_protocol_error();
153 }
154
155 if (rbuffer.get_length() == 0) {
156 fill_some(rbuffer, csfd);
157 }
158 }

516 }

665 auto resolve_var = [&](const string &name) {
666 if (offline_operation && !issued_var_subst_warning) {
667 report_general_warning("Variable substitution performed by dinitcheck "
668 "for file paths may not match dinit daemon (environment may differ); "
669 "use --online to avoid this warning");
670 issued_var_subst_warning = true;
671 }
672 return resolve_env_var(name, renvmap);
673 };
674
675 service_settings_wrapper<prelim_dep> settings;
676
677 string line;
678 service_file.exceptions(ios::badbit);
679
680 file_input_stack input_stack;
681 input_stack.push(std::move(service_filename), std::move(service_file));
682
683 try {
684 process_service_file(name, input_stack,
685 [&](string &line, file_pos_ref input_pos, string &setting, setting_op_t op,
686 string_iterator &i, string_iterator &end) -> void {
687
688 auto process_dep_dir_n = [&](std::list<prelim_dep> &deplist,
689 const std::string &waitsford,
690 dependency_type dep_type) -> void {
691 const string &service_filename = input_stack.current_file_name();
692 process_dep_dir(name.c_str(), service_filename, deplist, waitsford, dep_type);
693 };
694
695 auto load_service_n = [&](const string &dep_name) -> const string & {
696 return dep_name;
697 };
698
699 try {
700 process_service_line(settings, name.c_str(), service_arg.c_str(), line,
701 input_pos, setting, op, i, end, load_service_n, process_dep_dir_n,
702 resolve_var);
703 }
704 catch (service_description_exc &exc) {
705 if (exc.service_name.empty()) {
706 exc.service_name = name;
707 }
708 report_service_description_exc(exc);
709 }
710 },
711 service_arg.c_str(), resolve_var);
712 }
713 catch (std::system_error &sys_err)
714 {
715 report_error(sys_err, name);
716 throw service_load_exc(name, input_stack.current_file_name() + ": error while reading service description");
717 }
718
719 auto report_err = [&](const char *msg) {
720 report_service_description_err(name, msg);
721 };
722
723 environment srv_env{};
724
725 // Fill user vars before reading env file
726 if (settings.export_passwd_vars) {
727 try {
728 fill_environment_userinfo(settings.run_as_uid, name, srv_env);
729 }
730 catch (service_load_exc &load_exc) {
731 report_service_description_err(name, load_exc.exc_description);
732 }
733 }
734
735 // Set service name in environment if desired
736 if (settings.export_service_name) {
737 std::string envname = "DINIT_SERVICE=";
738 envname += name;
739 srv_env.set_var(std::move(envname));
740 }
741
742 if (!settings.env_file.empty()) {
743 try {
744 std::string fullpath = combine_paths(service_wdir, settings.env_file.c_str());
745
746 auto log_inv_env_setting = [&](int line_num) {
747 report_service_description_err(name,
748 std::string("Invalid environment variable setting in environment file " + fullpath
749 + " (line ") + std::to_string(line_num) + ")");
750 };
751 auto log_bad_env_command = [&](int line_num) {
752 report_service_description_err(name,
753 std::string("Bad command in environment file ") + fullpath + " (line " + std::to_string(line_num) + ")");
754 };
755
756 read_env_file_inline(fullpath.c_str(), false, srv_env, true, log_inv_env_setting, log_bad_env_command);
757 } catch (const std::system_error &se) {
758 report_service_description_err(name, std::string("could not load environment file: ") + se.what());
759 }
760 }
761
762 renvmap = srv_env.build(menv);
763
764 settings.finalise(report_err, nullptr, report_err, resolve_var);
765
766 if (!settings.working_dir.empty()) {
767 service_wdir = settings.working_dir;
768 }
769 int oflags = O_DIRECTORY;
770#ifdef O_PATH
771 oflags |= O_PATH;
772#else
773 oflags |= O_RDONLY;
774#endif
775 dirfd = open(service_wdir.c_str(), oflags);
776 if (dirfd < 0) {
777 report_service_description_err(name,
778 std::string("could not open service working directory: ") + strerror(errno));
779 dirfd = AT_FDCWD;
780 }
781
782 auto check_command = [&](const char *setting_name, const char *command) {
783 struct stat command_stat;
784 if (command[0] != '/') {
785 report_service_description_err(name,
786 std::string("executable '") + command + "' is not an absolute path");
787 }
788 else if (fstatat(dirfd, command, &command_stat, 0) == -1) {
789 report_service_description_err(name,
790 std::string("could not stat ") + setting_name + " executable '" + command
791 + "': " + strerror(errno));
792 }
793 else {
794 if ((command_stat.st_mode & S_IFMT) != S_IFREG) {
795 report_service_description_err(name, std::string(setting_name) + " executable '"
796 + command + "' is not a regular file.");
797 }
798 else if ((command_stat.st_mode & S_IXUSR) == 0) {
799 report_service_description_err(name, std::string(setting_name) + " executable '" + command
800 + "' is not executable by owner.");
801 }
802 }
803 };
804
805 if (!settings.command.empty()) {
806 int offset_start = settings.command_offsets.front().first;
807 int offset_end = settings.command_offsets.front().second;
808 check_command("command", settings.command.substr(offset_start, offset_end - offset_start).c_str());
809 }
810 if (!settings.stop_command.empty()) {
811 int offset_start = settings.stop_command_offsets.front().first;
812 int offset_end = settings.stop_command_offsets.front().second;
813 check_command("stop command",
814 settings.stop_command.substr(offset_start, offset_end - offset_start).c_str());
815 }
816
817 if (settings.log_type == log_type_id::LOGFILE && !settings.logfile.empty()) {
818 string logfile_dir = parent_path(settings.logfile);
819 if (!logfile_dir.empty()) {
820 struct stat logfile_dir_stat;
821 if (fstatat(dirfd, logfile_dir.c_str(), &logfile_dir_stat, 0) == -1) {
822 report_service_description_err(name,
823 std::string("could not access logfile directory '") + logfile_dir + "': " + strerror(errno));
824 }
825 else {
826 if ((logfile_dir_stat.st_mode & S_IFDIR) == 0) {
827 report_service_description_err(name, std::string("logfile directory '")
828 + logfile_dir + "' exists but is not a directory.");
829 }
830 }
831 }
832 }
833
834 if (dirfd != AT_FDCWD) {
835 close(dirfd);
836 }
837
838 return new service_record(name, settings.service_type, settings.chain_to_name, settings.depends,
839 settings.before_svcs, settings.after_svcs, settings.consumer_of_name, settings.log_type);
840}
