38// Move an fd, if necessary, to another fd. The original destination fd will be closed.
39// if fd is specified as -1, returns -1 immediately. Returns 0 on success.
40static int move_fd(int fd, int dest)
41{
42 if (fd == -1) return -1;
43 if (fd == dest) return 0;
44
45 if (dup2(fd, dest) == -1) {
46 return -1;
47 }
48
49 close(fd);
50 return 0;
51}

53// Move a file descriptor to another, freeing up the original descriptor so that it can be used
54// for some reserved purpose.
55static int move_reserved_fd(int *fd, int min_fd)
56{
57 int new_fd = fcntl(*fd, F_DUPFD_CLOEXEC, min_fd);
58 if (new_fd != -1) {
59 close(*fd);
60 *fd = new_fd;
61 }
62 return new_fd;
63}
