222static inline int divide_timespec(const struct timespec &num, const struct timespec &den, struct timespec &rem) noexcept
223{
224 if (num.tv_sec < den.tv_sec) {
225 rem = num;
226 return 0;
227 }
228
229 if (num.tv_sec == den.tv_sec) {
230 if (num.tv_nsec < den.tv_nsec) {
231 rem = num;
232 return 0;
233 }
234 if (num.tv_sec == 0) {
235 rem.tv_sec = 0;
236 rem.tv_nsec = num.tv_nsec % den.tv_nsec;
237 return num.tv_nsec / den.tv_nsec;
238 }
239 // num.tv_sec == den.tv_sec and both are >= 1.
240 // The result can only be 1:
241 rem.tv_sec = 0;
242 rem.tv_nsec = num.tv_nsec - den.tv_nsec;
243 return 1;
244 }
245
246 // At this point, num.tv_sec >= 1.
247
248 time_val n = { num.tv_sec, num.tv_nsec };
249 time_val d = { den.tv_sec, den.tv_nsec };
250 time_val r = n;
251
252 // starting with numerator, subtract 1*denominator
253 r -= d;
254
255 // Check now for common case: one timer expiry with no overrun
256 if (r < d) {
257 rem = r;
258 return 1;
259 }
260
261 int nval = 1;
262 int rval = 1; // we have subtracted 1*D already
263
264 // shift denominator until it is greater than / roughly equal to numerator:
265 while (d.seconds() < r.seconds()) {
266 d <<= 1;
267 nval *= 2;
268 }
269
270 while (nval > 0) {
271 if (d <= r) {
272 r -= d;
273 rval += nval;
274 }
275
276 d >>= 1;
277 nval /= 2;
278 }
279
280 rem = r;
281 return rval;
282}

322 // For the specified timer queue, issue expirations for all timers set to expire on or before the given
323 // time (curtime).
324 void process_timer_queue(timer_queue_t &queue, const struct timespec &curtime) noexcept
325 {
326 if (queue.empty()) return;
327
328 // Peek timer queue; calculate difference between current time and timeout
329 const time_val * timeout = &queue.get_root_priority();
330 time_val curtime_tv = curtime;
331 while (*timeout <= curtime_tv) {
332 auto & thandle = queue.get_root();
333 timer_data &data = queue.node_data(thandle);
334 time_val &interval = data.interval_time;
335 data.expiry_count++;
336 queue.pull_root();
337 if (interval.seconds() == 0 && interval.nseconds() == 0) {
338 // Non periodic timer
339 if (data.enabled) {
340 data.enabled = false;
341 int expiry_count = data.expiry_count;
342 data.expiry_count = 0;
343 Base::receive_timer_expiry(thandle, data.userdata, expiry_count);
344 }
345 if (queue.empty()) {
346 break;
347 }
348 }
349 else {
350 // First calculate the overrun in time:
351 time_val overrun = time_val(curtime) - time_val(*timeout);
352
353 // Now we have to divide the time overrun by the period to find the
354 // interval overrun. This requires a division of a value not representable
355 // as a long...
356 struct timespec rem;
357 data.expiry_count += divide_timespec(overrun, interval, rem);
358
359 // new time is current time + interval - remainder:
360 time_val newtime = curtime + interval - rem;
361
362 queue.insert(thandle, newtime);
363 if (data.enabled) {
364 data.enabled = false;
365 int expiry_count = data.expiry_count;
366 data.expiry_count = 0;
367 Base::receive_timer_expiry(thandle, data.userdata, expiry_count);
368 }
369 }
370
371 // repeat until all expired timeouts processed
372 timeout = &queue.get_root_priority();
373 }
374 }

377 // set do_wait to false; otherwise, if any timers are pending, set ts to the delay before
378 // the next timer expires and set wait_ts to &ts.
