37 // returns false it has either deleted the connection or marked it for deletion; we

136bool control_conn_t::process_find_load(cp_cmd pktType)
137{
138 using std::string;
139
140 constexpr int pkt_size = 4;
141
142 if (rbuf.get_length() < pkt_size) {
143 chklen = pkt_size;
144 return true;
145 }
146

181 }

186 }
187 catch (service_load_exc &slexc) {

189 slexc.exc_description);
190 fail_code = cp_rply::SERVICE_LOAD_ERR;
191 }
192 }
193 else {
194 // FINDSERVICE
195 record = services->find_service(service_name.c_str());
196 }
197
198 if (record == nullptr) {

200 if (! queue_packet(std::move(rp_buf))) return false;
201 return true;

205 handle_t handle = allocate_service_handle(record);
206 std::vector<char> rp_buf;
207 rp_buf.reserve(7);

211 rp_buf.push_back(*(((char *) &handle) + i));

300bool control_conn_t::process_start_stop(cp_cmd pktType)
301{
302 using std::string;
303
304 constexpr int pkt_size = 2 + sizeof(handle_t);
305
306 if (rbuf.get_length() < pkt_size) {
307 chklen = pkt_size;
308 return true;
309 }
310
311 // 1 byte: packet type
312 // 1 byte: flags
313 // bit 0: 0 = no pin, 1 = pin
314 // bit 1: 0 = force stop, 1 = not forced ("gentle")
315 // bit 2: 0 = don't restart, 1 = restart after stopping
316 // --- (reserved)
317 // bit 7: 0 = no pre-ack, 1 = issue pre-ack
318 // 4 bytes: service handle
319
320 bool do_pin = ((rbuf[1] & 1) == 1);
321 handle_t handle;
322 rbuf.extract((char *) &handle, 2, sizeof(handle));
323
324 service_record *service = find_service_for_key(handle);
325 if (service == nullptr) {
326 // Service handle is bad
327 char badreqRep[] = { (char)cp_rply::BADREQ };
328 if (!queue_packet(badreqRep, 1)) return false;
329 bad_conn_close = true;
330 return true;
331 }
332 else {
333 char ack_buf[1] = { (char)cp_rply::PREACK };
334 if (rbuf[1] & 128) {
335 // Issue PREACK before doing anything that might change service state (and cause a
336 // service event info packet to be issued as a result). This allows the client to
337 // determine whether the info packets were queued before or after the command was
338 // processed.
339 if (!queue_packet(ack_buf, 1)) return false;
340 }
341
342 ack_buf[0] = (char)cp_rply::ACK;
343
344 switch (pktType) {
345 case cp_cmd::STARTSERVICE:
346 // start service, mark as required
347 if (services->is_shutting_down()) {
348 ack_buf[0] = (char)cp_rply::SHUTTINGDOWN;
349 break;
350 }
351 if ((service->get_state() == service_state_t::STOPPED
352 || service->get_state() == service_state_t::STOPPING)
353 && service->is_stop_pinned()) {
354 ack_buf[0] = (char)cp_rply::PINNEDSTOPPED;
355 break;
356 }
357 if (do_pin) service->pin_start();
358 service->start();
359 services->process_queues();
360 if (service->get_state() == service_state_t::STARTED) ack_buf[0] = (char)cp_rply::ALREADYSS;
361 break;
362 case cp_cmd::STOPSERVICE:
363 {
364 // force service to stop
365 bool do_restart = ((rbuf[1] & 4) == 4);
366 bool gentle = ((rbuf[1] & 2) == 2);
367 if (do_restart && services->is_shutting_down()) {
368 ack_buf[0] = (char)cp_rply::SHUTTINGDOWN;
369 break;
370 }
371 if ((service->get_state() == service_state_t::STARTED
372 || service->get_state() == service_state_t::STARTING)
373 && service->is_start_pinned()) {
374 ack_buf[0] = (char)cp_rply::PINNEDSTARTED;
375 break;
376 }
377 if (gentle) {
378 // Check dependents; return appropriate response if any will be affected
379 bool has_dependents;
380 if (!check_dependents(service, has_dependents)) {
381 return false;
382 }
383 if (has_dependents) {
384 // Reply packet has already been sent
385 goto clear_out;
386 }
387 }
388 service_state_t wanted_state;
389 if (do_restart) {
390 if (! service->restart()) {
391 ack_buf[0] = (char)cp_rply::NAK;
392 break;
393 }
394 wanted_state = service_state_t::STARTED;
395 }
396 else {
397 if (do_pin) service->pin_stop();
398 service->stop(true);
399 service->forced_stop();
400 wanted_state = service_state_t::STOPPED;
401 }
402 services->process_queues();
403 if (service->get_state() == wanted_state) ack_buf[0] = (char)cp_rply::ALREADYSS;
404 break;
405 }
406 case cp_cmd::WAKESERVICE:
407 {
408 // re-attach a service to its (started) dependents, causing it to start.
409 if (services->is_shutting_down()) {
410 ack_buf[0] = (char)cp_rply::SHUTTINGDOWN;
411 break;
412 }
413 if ((service->get_state() == service_state_t::STOPPED
414 || service->get_state() == service_state_t::STOPPING)
415 && service->is_stop_pinned()) {
416 ack_buf[0] = (char)cp_rply::PINNEDSTOPPED;
417 break;
418 }
419 bool found_dpt = false;
420 for (auto dpt : service->get_dependents()) {
421 if (dpt->is_only_ordering()) continue;
422 auto from = dpt->get_from();
423 auto from_state = from->get_state();
424 if (from_state == service_state_t::STARTED || from_state == service_state_t::STARTING) {
425 found_dpt = true;
426 if (!dpt->holding_acq) {
427 dpt->get_from()->start_dep(*dpt);
428 }
429 }
430 }
431 if (!found_dpt) {
432 ack_buf[0] = (char)cp_rply::NAK;
433 }
434
435 if (do_pin) service->pin_start();
436 services->process_queues();
437 if (service->get_state() == service_state_t::STARTED) ack_buf[0] = (char)cp_rply::ALREADYSS;
438 break;
439 }
440 case cp_cmd::RELEASESERVICE:
441 // remove required mark, stop if not required by dependents
442 if (do_pin) service->pin_stop();
443 service->stop(false);
444 services->process_queues();
445 if (service->get_state() == service_state_t::STOPPED) ack_buf[0] = (char)cp_rply::ALREADYSS;
446 break;
447 default:
448 // avoid warning for unhandled switch/case values
449 return false;
450 }
451
452 if (!queue_packet(ack_buf, 1)) return false;
453 }
454
455 clear_out:
456 // Clear the packet from the buffer
457 rbuf.consume(pkt_size);
458 chklen = 0;
459 return true;
460}

612static void fill_status_buffer(char *buffer, service_record *service)
613{
614 buffer[0] = static_cast<char>(service->get_state());
615 buffer[1] = static_cast<char>(service->get_target_state());
616
617 pid_t proc_pid = service->get_pid();
618
619 char b0 = service->is_waiting_for_console() ? 1 : 0;
620 b0 |= service->has_console() ? 2 : 0;
621 b0 |= service->was_start_skipped() ? 4 : 0;
622 b0 |= service->is_marked_active() ? 8 : 0;
623 b0 |= (proc_pid != -1) ? 16 : 0;
624 buffer[2] = b0;
625 buffer[3] = static_cast<char>(service->get_stop_reason());
626
627 buffer[4] = 0; // (if exec failed, these are replaced with stage)
628 buffer[5] = 0;
629
630 if (proc_pid != -1) {
631 memcpy(buffer + 6, &proc_pid, sizeof(proc_pid));
632 }
633 else {
634 // These values only make sense in STOPPING/STOPPED, but we'll fill them in regardless:
635 if (buffer[3] == (char)stopped_reason_t::EXECFAILED) {
636 base_process_service *bsp = (base_process_service *)service;
637 run_proc_err exec_err = bsp->get_exec_err_info();
638 uint16_t stage = (uint16_t)exec_err.stage;
639 memcpy(buffer + 4, &stage, 2);
640 memcpy(buffer + 6, &exec_err.st_errno, sizeof(int));
641 }
642 else {
643 auto exit_status_ps = service->get_exit_status();
644 // There is no portable way to derive the correct exit status value corresponding
645 // to any condition except for a clean exit (0) which is the most important anyway.
646 // We'll use -1 for any other status although this may not be valid.
647 int exit_status = exit_status_ps.did_exit_clean() ? 0 : -1;
648 memcpy(buffer + 6, &exit_status, sizeof(exit_status));
649 }
650 }
651}

845bool control_conn_t::add_service_dep(bool do_enable)
846{
847 // 1 byte packet type
848 // 1 byte dependency type
849 // handle: "from"
850 // handle: "to"
851
852 constexpr int pkt_size = 2 + sizeof(handle_t) * 2;
853
854 if (rbuf.get_length() < pkt_size) {
855 chklen = pkt_size;
856 return true;
857 }
858
859 handle_t from_handle;
860 handle_t to_handle;
861 rbuf.extract((char *) &from_handle, 2, sizeof(from_handle));
862 rbuf.extract((char *) &to_handle, 2 + sizeof(from_handle), sizeof(to_handle));
863
864 service_record *from_service = find_service_for_key(from_handle);
865 service_record *to_service = find_service_for_key(to_handle);
866 if (from_service == nullptr || to_service == nullptr || from_service == to_service) {
867 // Service handle is bad
868 char badreq_rep[] = { (char)cp_rply::BADREQ };
869 if (!queue_packet(badreq_rep, 1)) return false;
870 bad_conn_close = true;
871 return true;
872 }
873
874 // Check dependency type is valid:
875 int dep_type_int = rbuf[1];
876 if (!contains({dependency_type::MILESTONE, dependency_type::REGULAR,
877 dependency_type::WAITS_FOR}, dep_type_int)) {
878 char badreqRep[] = { (char)cp_rply::BADREQ };
879 if (!queue_packet(badreqRep, 1)) return false;
880 bad_conn_close = true;
881 }
882 dependency_type dep_type = static_cast<dependency_type>(dep_type_int);
883
884 // Check current service states are valid for given dep type
885 if (dep_type == dependency_type::REGULAR) {
886 if (from_service->get_state() != service_state_t::STOPPED &&
887 to_service->get_state() != service_state_t::STARTED) {
888 // Cannot create dependency now since it would be contradicted:
889 char nak_rep[] = { (char)cp_rply::NAK };
890 if (! queue_packet(nak_rep, 1)) return false;
891 rbuf.consume(pkt_size);
892 chklen = 0;
893 return true;
894 }
895 }
896
897 // Check for creation of circular dependency chain
898 std::unordered_set<service_record *> dep_marks;
899 std::vector<service_record *> dep_queue;
900 dep_queue.push_back(to_service);
901 while (! dep_queue.empty()) {
902 service_record * sr = dep_queue.back();
903 dep_queue.pop_back();
904 // iterate deps; if dep == from, abort; otherwise add to set/queue
905 // (only add to queue if not already in set)
906 for (auto &dep : sr->get_dependencies()) {
907 service_record * dep_to = dep.get_to();
908 if (dep_to == from_service) {
909 // fail, circular dependency!
910 char nak_rep[] = { (char)cp_rply::NAK };
911 if (! queue_packet(nak_rep, 1)) return false;
912 rbuf.consume(pkt_size);
913 chklen = 0;
914 return true;
915 }
916 if (dep_marks.insert(dep_to).second) {
917 dep_queue.push_back(dep_to);
918 }
919 }
920 }
921 dep_marks.clear();
922 dep_queue.clear();
923
924 bool dep_exists = false;
925 service_dep * dep_record = nullptr;
926
927 // Prevent creation of duplicate dependency:
928 for (auto &dep : from_service->get_dependencies()) {
929 service_record * dep_to = dep.get_to();
930 if (dep_to == to_service && dep.dep_type == dep_type) {
931 // Dependency already exists
932 dep_exists = true;
933 dep_record = &dep;
934 break;
935 }
936 }
937
938 if (! dep_exists) {
939 // Create dependency:
940 dep_record = &(from_service->add_dep(to_service, dep_type));
941 services->process_queues();
942 }
943
944 if (do_enable && contains({service_state_t::STARTED, service_state_t::STARTING},
945 from_service->get_state())) {
946 // The dependency record is activated: mark it as holding acquisition of the dependency, and start
947 // the dependency.
948 if (!services->is_shutting_down()) {
949 dep_record->get_from()->start_dep(*dep_record);
950 services->process_queues();
951 }
952 }
953
954 char ack_rep[] = { (char)cp_rply::ACK };
955 if (! queue_packet(ack_rep, 1)) return false;
956 rbuf.consume(pkt_size);
957 chklen = 0;
958 return true;
959}

1051bool control_conn_t::process_setenv()
1052{
1053 using std::string;
1054
1055 string envVar;
1056 typename string::size_type eq;
1057
1058 constexpr int pkt_size = 4;
1059 char badreqRep[] = { (char)cp_rply::BADREQ };
1060 char okRep[] = { (char)cp_rply::ACK };
1061
1062 if (rbuf.get_length() < pkt_size) {
1063 chklen = pkt_size;
1064 return true;
1065 }
1066
1067 envvar_len_t envvar_len;
1068 rbuf.extract(&envvar_len, 1, sizeof(envvar_len));
1069 if (envvar_len <= 0 || envvar_len > (1024 - 3)) {
1070 goto badreq;
1071 }
1072 chklen = envvar_len + 1 + sizeof(envvar_len); // packet type + (2 byte) length + envvar
1073
1074 if (rbuf.get_length() < chklen) {
1075 // packet not complete yet; read more
1076 return true;
1077 }
1078
1079 envVar = rbuf.extract_string(3, envvar_len);
1080
1081 eq = envVar.find('=');
1082 if (eq == envVar.npos) {
1083 // Unset the env var
1084 main_env.undefine_var(std::move(envVar), true);
1085 }
1086 else if (eq) {
1087 // Regular set
1088 main_env.set_var(std::move(envVar), true);
1089 }
1090 else {
1091 // At the beginning of the string
1092 goto badreq;
1093 }
1094
1095 // Success response
1096 if (!queue_packet(okRep, 1)) return false;
1097
1098 // Clear the packet from the buffer
1099 rbuf.consume(chklen);
1100 chklen = 0;
1101 return true;
1102
1103badreq:
1104 // Queue error response / mark connection bad
1105 if (!queue_packet(badreqRep, 1)) return false;
1106 bad_conn_close = true;
1107 return true;
1108}

1343 std::memcpy(reppkt.data() + 2 + sizeof(uint32_t), &sdirs, sizeof(sdirs));
1344
1345 // Our current working directory, which above are relative to:
1346 // leave sizeof(uint32_t) for size, which we'll fill in afterwards:
1347 std::size_t curpos = reppkt.size() + sizeof(uint32_t);
1348#ifdef PATH_MAX
1349 uint32_t try_path_size = PATH_MAX;
1350#else
1351 uint32_t try_path_size = 2048;
1352#endif
1353 char *wd;
1354 while (true) {
1355 std::size_t total_size = curpos + std::size_t(try_path_size);
1356 if (total_size < curpos) {
1357 // Overflow. In theory we could now limit to size_t max, but the size must already
1358 // be crazy long; let's abort.
1359 char ack_rep[] = { (char)cp_rply::NAK };
1360 if (! queue_packet(ack_rep, 1)) return false;
1361 return true;
1362 }
1363 reppkt.resize(total_size);
1364 wd = getcwd(reppkt.data() + curpos, try_path_size);
1365 if (wd != nullptr) break;
1366
1367 // Keep doubling the path size we try until it's big enough, or we get numeric overflow
1368 uint32_t new_try_path_size = try_path_size * uint32_t(2u);
1369 if (new_try_path_size < try_path_size) {
1370 // Overflow.
1371 char ack_rep[] = { (char)cp_rply::NAK };
1372 return queue_packet(ack_rep, 1);
1373 }
1374 try_path_size = new_try_path_size;
1375 }
1376
1377 uint32_t wd_len = std::strlen(reppkt.data() + curpos);
1378 reppkt.resize(curpos + std::size_t(wd_len));
1379 std::memcpy(reppkt.data() + curpos - sizeof(uint32_t), &wd_len, sizeof(wd_len));
1380
1381 // Each directory in the load path:
1382 for (int i = 0; uint32_t(i) < sdirs; i++) {
1383 const char *sdir = dss->get_service_dir(i);
1384 uint32_t dlen = std::strlen(sdir);
1385 auto cursize = reppkt.size();
1386 reppkt.resize(cursize + sizeof(dlen) + dlen);
1387 std::memcpy(reppkt.data() + cursize, &dlen, sizeof(dlen));
1388 std::memcpy(reppkt.data() + cursize + sizeof(dlen), sdir, dlen);
1389 }
1390
1391 // Total packet size:
1392 uint32_t fsize = reppkt.size();
1393 std::memcpy(reppkt.data() + 2, &fsize, sizeof(fsize));
1394
1395 if (! queue_packet(std::move(reppkt))) return false;
1396 return true;
1397 }
1398 else {
1399 // If we don't know how to deal with the service set type, send a NAK reply:
1400 char ack_rep[] = { (char)cp_rply::NAK };
1401 return queue_packet(ack_rep, 1);
1402 }

1406{
1407 // Try to find a unique handle (integer) in a single pass. Since the map is ordered, we can search until
1408 // we find a gap in the handle values.
1409 handle_t candidate = 0;
1410 for (auto p : key_service_map) {

1412 else break;

1414

1417 // The following operations perform allocation (can throw std::bad_alloc). If an exception occurs we
1418 // must undo any previous actions:
1419 if (is_unique) {
1420 record->add_listener(this);
1421 }
1422
1423 try {
1424 key_service_map[candidate] = record;
1425 service_key_map.insert(std::make_pair(record, candidate));
1426 }
1427 catch (...) {
1428 if (is_unique) {
1429 record->remove_listener(this);
1430 }
1431
1432 key_service_map.erase(candidate);
1433 }
1434

1439{
1440 // For each service handle corresponding to the event, send an information packet.
1441 auto range = service_key_map.equal_range(service);
1442 auto &i = range.first;
1443 auto &end = range.second;
1444 try {
1445 while (i != end) {
1446 uint32_t key = i->second;
1447 std::vector<char> pkt;
1448
1449 // There are two types of service event packet: v5+, and the original. For backwards
1450 // compatibility, we send both. The new packet is sent first since this simplifies
1451 // things for the client (eg if waiting for an event, the client will receive the
1452 // packet with the most information first).
1453
1454 // packet type (byte) + packet length (byte) + event type (byte) + key + status buffer
1455 constexpr int pktsize5 = 3 + sizeof(key) + STATUS_BUFFER5_SIZE;
1456 pkt.reserve(pktsize5);
1457 pkt.push_back((char)cp_info::SERVICEEVENT5);
1458 pkt.push_back(pktsize5);
1459 char *p = (char *)&key;
1460 for (unsigned j = 0; j < sizeof(key); j++) {
1461 pkt.push_back(*p++);
1462 }
1463 pkt.push_back(static_cast<char>(event));
1464 pkt.resize(pktsize5);
1465 fill_status_buffer5(pkt.data() + 3 + sizeof(key), service);
1466 queue_packet(std::move(pkt));
1467
1468 pkt.clear();
1469
1470 constexpr int pktsize = 3 + sizeof(key) + STATUS_BUFFER_SIZE;
1471 pkt.reserve(pktsize);
1472 pkt.push_back((char)cp_info::SERVICEEVENT);
1473 pkt.push_back(pktsize);
1474 p = (char *)&key;
1475 for (unsigned j = 0; j < sizeof(key); j++) {
1476 pkt.push_back(*p++);
1477 }
1478 pkt.push_back(static_cast<char>(event));
1479 pkt.resize(pktsize);
1480 fill_status_buffer(pkt.data() + 3 + sizeof(key), service);
1481 queue_packet(std::move(pkt));

1484 }
1485 }

1513bool control_conn_t::queue_packet(const char *pkt, unsigned size) noexcept
1514{
1515 bool was_empty = outbuf.empty();
1516
1517 // If the queue is empty, we can try to write the packet out now rather than queueing it.
1518 // If the write is unsuccessful or partial, we queue the remainder.
1519 if (was_empty) {
1520 int wr = bp_sys::write(iob.get_watched_fd(), pkt, size);
1521 if (wr == -1) {
1522 if (errno == EPIPE) {
1523 return false;
1524 }
1525 if (errno != EAGAIN && errno != EWOULDBLOCK && errno != EINTR) {
1526 log(loglevel_t::WARN, "Error writing to control connection: ", strerror(errno));
1527 return false;
1528 }
1529 // EAGAIN etc: fall through to below
1530 }
1531 else {
1532 if ((unsigned)wr == size) {
1533 // Ok, all written.
1534 return true;
1535 }
1536 pkt += wr;
1537 size -= wr;
1538 }
1539 }
1540
1541 // Create a vector out of the (remaining part of the) packet:
1542 try {
1543 outbuf.emplace_back(pkt, pkt + size);
1544 outbuf_size += size;
1545 return true;
1546 }
1547 catch (std::bad_alloc &baexc) {
1548 // Mark the connection bad, and stop reading further requests
1549 bad_conn_close = true;
1550 oom_close = true;
1551 if (was_empty) {
1552 // We can't send out-of-memory response as we already wrote as much as we
1553 // could above. Neither can we later send the response since we have currently
1554 // sent an incomplete packet. All we can do is close the connection.
1555 return false;
1556 }
1557 else {
1558 return true;
1559 }
1560 }
1561}

1658bool control_conn_t::send_data() noexcept
1659{
1660 if (outbuf.empty() && bad_conn_close) {
1661 if (oom_close) {
1662 // Send oom response
1663 char oomBuf[] = { (char)cp_rply::OOM };
1664 bp_sys::write(iob.get_watched_fd(), oomBuf, 1);

1666 return true;
1667 }

1669 vector<char> & pkt = outbuf.front();

1672 if (written == -1) {
1673 if (errno == EPIPE) {
1674 // read end closed
1675 return true;
1676 }
1677 else if (errno == EAGAIN || errno == EWOULDBLOCK || errno == EINTR) {
1678 // spurious readiness notification?
1679 return false;
1680 }
1681 else {
1682 log(loglevel_t::ERROR, "Error writing to control connection: ", strerror(errno));
1683 return true;
1684 }
1685 }
1686
1687 outpkt_index += written;
1688 if (outpkt_index == pkt.size()) {
1689 // We've finished this packet, move on to the next:
1690 outbuf_size -= pkt.size();
1691 outbuf.pop_front();

1693 if (oom_close) {
1694 // remain active, try to send cp_rply::OOM shortly
1695 return false;
1696 }
1697 if (outbuf.empty() && bad_conn_close) {
1698 return true;
1699 }
1700 }
1701
1702 // more to send
1703 return false;
