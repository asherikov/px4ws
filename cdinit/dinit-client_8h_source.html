146// Fill a circular buffer from a file descriptor, until it contains at least _rlength_ bytes.
147// Throws cp_read_exception if the requested number of bytes cannot be read, with:
148// errcode = 0 if end of stream (remote end closed)
149// errcode = errno if another error occurred
150// Note that EINTR is ignored (i.e. the read will be re-tried).
151inline void fill_buffer_to(cpbuffer_t &buf, int fd, int rlength)
152{
153 do {
154 int r = buf.fill_to(fd, rlength);
155 if (r == -1) {
156 if (errno != EINTR) {
157 throw cp_read_exception(errno);
158 }
159 }
160 else if (r == 0) {
161 throw cp_read_exception(0);
162 }
163 else {
164 return;
165 }
166 }
167 while (true);
168}

170// Fill a circular buffer from a file descriptor, until it contains at least some more data
171// then it did. Throws cp_read_exception if no more bytes can be read, with:
172// errcode = 0 if end of stream (remote end closed)
173// errcode = errno if another error occurred
174// Note that EINTR is ignored (i.e. the read will be re-tried).
175inline void fill_some(cpbuffer_t &buf, int fd)
176{
177 while(true) {
178 int r = buf.fill(fd);
179 if (r == 0) {
180 throw cp_read_exception(0);
181 }
182 else if (r > 0) {
183 return;
184 }
185
186 if (errno != EINTR) {
187 throw cp_read_exception(errno);
188 }
189
190 // if EINTR, just try again
191 }
192}

194// Wait for a reply packet, skipping over any information packets that are received in the meantime.
195// Note the reply packet may be partially read. Caller is responsible for reading full packet.
196inline void wait_for_reply(cpbuffer_t &rbuffer, int fd)
197{
198 fill_buffer_to(rbuffer, fd, 1);
199
200 while (rbuffer[0] >= 100) {
201 // Information packet; discard.
202 fill_buffer_to(rbuffer, fd, 2);
203 int pktlen = (unsigned char) rbuffer[1];
204
205 rbuffer.consume(1); // Consume one byte so we'll read one byte of the next packet
206 fill_buffer_to(rbuffer, fd, pktlen);
207 rbuffer.consume(pktlen - 1);
208 }
209}

211inline void wait_for_reply(cpbuffer_t &rbuffer, int fd, dinit_cptypes::handle_t handle, observed_states_t *seen_states)
212{
213 fill_buffer_to(rbuffer, fd, 1);
214
215 while (rbuffer[0] >= 100) {
216 // Information packet; discard.
217 cp_info pkt_type = (cp_info) rbuffer[0];
218 fill_buffer_to(rbuffer, fd, 2);
219 unsigned pktlen = (unsigned char) rbuffer[1];
220
221 rbuffer.consume(1); // Consume one byte so we'll read one byte of the next packet
222 fill_buffer_to(rbuffer, fd, pktlen);
223
224 if (value(pkt_type).is_in(cp_info::SERVICEEVENT, cp_info::SERVICEEVENT5)
225 && seen_states != nullptr) {
226
227 // earlier versions do not include status info, the size in that case is
228 // base_pkt_size:
229 constexpr unsigned base_pkt_size = 2 + sizeof(dinit_cptypes::handle_t) + 1;
230
231 if (pktlen < base_pkt_size) {
232 throw dinit_protocol_error();
233 }
234
235 // Extract handle, check for match
236 dinit_cptypes::handle_t ev_handle;
237 rbuffer.extract((char *)&ev_handle, 1, sizeof(ev_handle));
238 service_event_t event = static_cast<service_event_t>(rbuffer[1 + sizeof(ev_handle)]);
239
240 if (ev_handle == handle) {
241 if (event == service_event_t::STOPPED) {
242 seen_states->stopped = true;
243 }
244 if (event == service_event_t::STARTED) {
245 seen_states->started = true;
246 }
247 if (event == service_event_t::FAILEDSTART) {
248 stopped_reason_t stop_reason =
249 static_cast<stopped_reason_t>(rbuffer[base_pkt_size + 2]);
250
251 seen_states->failed_start = true;
252 seen_states->stop_reason = stop_reason;
253
254 int exit_status;
255 int exit_si_code;
256 int exit_si_status;
257 rbuffer.extract((char *)&exit_status, base_pkt_size + 5, sizeof(exit_status));
258 if (pkt_type == cp_info::SERVICEEVENT5) {
259 if (pktlen < base_pkt_size + STATUS_BUFFER5_SIZE) {
260 throw dinit_protocol_error();
261 }
262 exit_si_code = exit_status;
263 rbuffer.extract((char *)&exit_si_status,
264 base_pkt_size + 5 + sizeof(exit_si_code), sizeof(exit_si_status));
265
266 seen_states->exit_si_code = exit_si_code;
267 seen_states->exit_si_status = exit_si_status;
268 }
269
270 seen_states->exit_status = exit_status;
271 }
272 }
273 }

275 rbuffer.consume(pktlen - 1);
276 }

292// Write *all* the requested buffer and re-try if necessary until
293// the buffer is written or an unrecoverable error occurs.
294// Note: count is int
295inline int write_all(int fd, const void *buf, int count)
296{
297 const char *cbuf = static_cast<const char *>(buf);
298 int w = 0;
299 while (count > 0) {
300 int r = write(fd, cbuf, count);
301 if (r == -1) {
302 if (errno == EINTR) continue;
303 return (w > 0) ? w : r;
304 }
305 w += r;
306 cbuf += r;
307 count -= r;
308 }
309 return w;
310}

327// Check the protocol version is compatible with the client.
328// minversion - minimum protocol version that client can speak
329// version - maximum protocol version that client can speak
330// rbuffer, fd - communication buffer and socket
331// returns: the actual protocol version (this may exceed the version specified in 'version',
332// if the daemon is backwards-compatible with that version).
333// throws an exception on protocol mismatch or error.
334inline uint16_t check_protocol_version(int minversion, int version, cpbuffer_t &rbuffer, int fd)
335{
336 using namespace dinit_cptypes;
337 constexpr int bufsize = 1;
338 char buf[bufsize] = { (char)cp_cmd::QUERYVERSION };
339 write_all_x(fd, buf, bufsize);
340
341 wait_for_reply(rbuffer, fd);
342 if (rbuffer[0] != (char)cp_rply::CPVERSION) {
343 throw cp_read_exception{0};
344 }
345
346 // cp_rply::CVERSION, (2 byte) minimum compatible version, (2 byte) actual version
347 constexpr int rbufsize = 1 + 2 * sizeof(uint16_t);
348 fill_buffer_to(rbuffer, fd, rbufsize);
349 uint16_t rminversion;
350 uint16_t cpversion;
351
352 rbuffer.extract(reinterpret_cast<char *>(&rminversion), 1, sizeof(uint16_t));
353 rbuffer.extract(reinterpret_cast<char *>(&cpversion), 1 + sizeof(uint16_t), sizeof(uint16_t));
354 rbuffer.consume(rbufsize);
355
356 if (rminversion > version) {
357 // We are too old
358 throw cp_old_client_exception();
359 }
360 if (cpversion < minversion) {
361 // Server is too old
362 throw cp_old_server_exception();
363 }
364
365 return cpversion;
366}

409// Connect to the dinit daemon, return the connected socket fd.
410// Throws general_error on error.
411inline int connect_to_daemon(const char *control_socket_path)
412{
413 int socknum = socket(AF_UNIX, SOCK_STREAM, 0);
414 if (socknum == -1) {
415 throw general_error(errno, "opening socket");
416 }
417
418 struct sockaddr_un * name;
419 uint sockaddr_size = offsetof(struct sockaddr_un, sun_path) + strlen(control_socket_path) + 1;
420 name = (struct sockaddr_un *) malloc(sockaddr_size);
421 if (name == nullptr) {
422 throw general_error(ENOMEM);
423 }
424
425 name->sun_family = AF_UNIX;
426 strcpy(name->sun_path, control_socket_path);
427
428 int connr = connect(socknum, (struct sockaddr *) name, sockaddr_size);
429 free(name);
430
431 if (connr == -1) {
432 throw control_sock_conn_err(errno, control_socket_path);
433 }
434
435 return socknum;
436}

463// Extract/read a string of specified length from the buffer/socket. The string is consumed
464// from the buffer.
465inline std::string read_string(int socknum, cpbuffer_t &rbuffer, uint32_t length)
466{
467 int rb_len = rbuffer.get_length();
468 if (uint32_t(rb_len) >= length) {
469 std::string r = rbuffer.extract_string(0, length);
470 rbuffer.consume(length);
471 return r;
472 }
473
474 std::string r = rbuffer.extract_string(0, rb_len);
475 uint32_t rlen = length - rb_len;
476 uint32_t clen;
477 do {
478 rbuffer.reset();
479 fill_some(rbuffer, socknum);
480 char *bptr = rbuffer.get_ptr(0);
481 clen = rbuffer.get_length();
482 clen = std::min(clen, rlen);
483 r.append(bptr, clen);
484 rlen -= clen;
485 } while (rlen > 0);
486
487 rbuffer.consume(clen);
488
489 return r;
490}
