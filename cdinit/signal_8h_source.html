10// Support for the standard POSIX signal mechanisms. This can be used by backends that don't
11// otherwise support receiving signals (currently: select/pselect, and kqueue for MacOS (due
12// to a MacOS bug). It is not particularly nice (it involves using siglongjmp out of a signal
13// handler, which POSIX mildly frowns upon) but it's really the only viable way to process signals
14// together with file descriptor / other events and obtain the full siginfo_t data passed to the
15// signal handler.
16
17// Use:
18// - extend the signal_events class below
19// - when wanting to poll for events:
20// - establish a landing pad via sigsetjmp, to the jmpbuf returned by get_sigreceive_jmpbuf()
21// - sigsetjmp will return again with non-zero if a signal is received; in this case,
22// call process_signal(...) and set timeout to zero, since an event has been received
23// - unmask the signal
24// - poll the backend (eg call select())
25// - mask the signal
26
27// sigsetjmp/siglongjmp are used rather than plain setjmp/longjmp to ensure that the signal mask is
28// restored, and avoid the possibility of the signal handler being executed multiple times (there
29// is a single-entry buffer for siginfo data, so if the handler executed multiple times, data would
30// be lost).

162 // Process a received signal
163 // Called when we have received a signal and its handler has been dispatched, at which point the
164 // handler has (a) stored the siginfo received and (b) performed a siglongjmp to a landing pad
165 // established just before the signal was unmasked. This function retrieves the stored siginfo
166 // and passes the signal to the base class (via Base::receive_signal).
167 void process_signal()
168 {
169 using namespace dprivate::signal_mech;
170
171 // We want to prevent the following read of siginfo from moving forward - it must not move
172 // forward prior to the point where the signal was unmasked - so use a fence with acquire semantics:
173 std::atomic_signal_fence(std::memory_order_acquire);
174 // (this is probably unnecessary, due to the setjmp/longjmp combination that should precede the call
175 // to this function, but we play it safe).
176
177 auto * sinfo = get_siginfo();
178 sigdata_t sigdata;
179 sigdata.info = *sinfo;
180
181 Base::lock.lock();
182 void *udata = sig_userdata[sinfo->si_signo];
183 if (udata != nullptr && Base::receive_signal(*this, sigdata, udata)) {
184 if (mask_enables) {
185 sigdelset(&active_sigmask, sinfo->si_signo);
186 }
187 else {
188 sigaddset(&active_sigmask, sinfo->si_signo);
189 }
190 }
191 Base::lock.unlock();
192 }

194 // Process a received signal, and update sigmask - which should reflect the inverse of the
195 // active signal mask. See comments for process_signal() above.
196 void process_signal(sigset_t &sigmask)
197 {
198 using namespace dprivate::signal_mech;
199
200 // We want to prevent the following read of siginfo from moving forward - it must not move
201 // forward prior to the point where the signal was unmasked - so use a fence with acquire semantics:
202 std::atomic_signal_fence(std::memory_order_acquire);
203 // (this is probably unnecessary, due to the setjmp/longjmp combination that should precede the call
204 // to this function, but we play it safe).
205
206 auto * sinfo = get_siginfo();
207 sigdata_t sigdata;
208 sigdata.info = *sinfo;
209
210 Base::lock.lock();
211 void *udata = sig_userdata[sinfo->si_signo];
212 if (udata != nullptr && Base::receive_signal(*this, sigdata, udata)) {
213 if (mask_enables) {
214 sigaddset(&sigmask, sinfo->si_signo);
215 sigdelset(&active_sigmask, sinfo->si_signo);
216 }
217 else {
218 sigdelset(&sigmask, sinfo->si_signo);
219 sigaddset(&active_sigmask, sinfo->si_signo);
220 }
221 }
222 Base::lock.unlock();
223 }
