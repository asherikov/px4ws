167// Complete read - read the specified size until end-of-file or error; continue read if
168// interrupted by signal.
169inline ssize_t complete_read(int fd, void * buf, size_t n)
170{
171 char * cbuf = static_cast<char *>(buf);
172 ssize_t r = 0;
173 while ((size_t)r < n) {
174 ssize_t res = bp_sys::read(fd, cbuf + r, n - r);
175 if (res == 0) {
176 return r;
177 }
178 if (res < 0) {
179 if (errno == EINTR) {
180 continue;
181 }
182
183 // If any other error, and we have successfully read some, return it:
184 if (r == 0) {
185 return -1;
186 }
187 else {
188 return r;
189 }
190 }
191 r += res;
192 }
193 return n;
194}

282inline size_t hash(const string_view &str)
283{
284 size_t end_pos = str.length();
285 size_t hash_val = 0;
286 for (size_t i = 0; i < end_pos; i += sizeof(size_t)) {
287 // collect as many characters as will fit into size_t
288 size_t hash_unit = 0;
289 std::memcpy(&hash_unit, str.data() + i, std::min(sizeof(size_t), end_pos - i));
290 // then incorporate the collected characters into the hash value
291 hash_val *= 31;
292 hash_val += hash_unit;
293 }
294 return hash_val;
295}

345 size_t buckets_to_max(size_t buckets) noexcept
346 {
347 // calculate "buckets * 3 / 4" but without risk of overflow at the multiply stage
348 size_t base = buckets / 4 * 3;
349 size_t extra = (buckets % 4) * 3 / 4;
350 return base + extra;
351 }

353 void do_rehash(size_t new_buckets) noexcept
354 {
355 // note, buckets vector is already at least new_buckets in size
356 size_t old_buckets = buckets.size();
357
358 // first splice all nodes from all buckets into a single list
359 std::list<K> all_nodes;
360 for (size_t i = 0; i < old_buckets; ++i) {
361 all_nodes.splice(all_nodes.end(), buckets[i]);
362 }
363
364 // now put all nodes into the correct bucket
365 auto node_i = all_nodes.begin();
366 while (node_i != all_nodes.end()) {
367 auto next_node_i = std::next(node_i);
368 size_t hashval = hash_f(*node_i);
369 size_t bucket_num = hashval % new_buckets;
370 buckets[bucket_num].splice(buckets[bucket_num].end(), all_nodes, node_i);
371 node_i = next_node_i;
372 }
373 }

460 // implement insert as a template so we can use it for rvalues and lvalues alike
461 template <typename V>
462 std::pair<iterator,bool> do_insert(V &&value)
463 {
464 size_t hashval = hash_f(value);
465 size_t bucket_num;
466
467 if (buckets.empty()) {
468 buckets.resize(4); // good enough starting point
469 current_limit = 3;
470 bucket_num = hashval % buckets.size();
471 }
472 else {
473 // First, check if the value is already present
474 bucket_num = hashval % buckets.size();
475 auto list_it = std::find_if(buckets[bucket_num].begin(), buckets[bucket_num].end(),
476 [&](const key_type &k) { return key_equal_f(k,std::forward<V>(value)); });
477 if (list_it != buckets[bucket_num].end()) {
478 return { { &buckets, list_it, bucket_num }, false };
479 }
480
481 // Not present. Check if we need to expand.
482 if (current_size >= current_limit) {
483 if (buckets.size() <= (buckets.max_size() / 2)) {
484 rehash(buckets.size() * 2);
485 current_limit *= 2;
486 bucket_num = hashval % buckets.size();
487 }
488 else {

490 current_limit = -1;
491 }
492 }
493 }
494
495 auto list_it = buckets[bucket_num].insert(buckets[bucket_num].end(), std::forward<V>(value));
496 ++current_size;
497 return { { &buckets, list_it, bucket_num }, true };
498 }

549 void rehash(size_type new_bucket_count)
550 {
551 // calculate minimum bucket count, limited by maximum possible count.
552 size_t max_buckets = buckets.max_size();
553 size_t max_count = buckets_to_max(max_buckets);
554 size_t min_buckets;
555 if (current_size > max_count) {
556 min_buckets = max_buckets;
557 }
558 else {
559 size_t base = current_size / 3 * 4;
560 size_t extra = (current_size % 3) * 4 / 3;
561 min_buckets = base + extra;
562 }
563
564 new_bucket_count = std::max(new_bucket_count, min_buckets);
565
566 if (new_bucket_count < buckets.size()) {
567 // buckets vector will shrink: we need to reallocate entries into their new buckets first
568 do_rehash(new_bucket_count);
569 buckets.resize(new_bucket_count);
570 }
571 else {
572 // grow the vector, then rehash
573 buckets.resize(new_bucket_count);
574 do_rehash(new_bucket_count);
575 }
576 }
