42// Comparison predicate for environment variables, checking name only
43struct env_equal_name
44{
45 bool operator()(const std::string &a, const std::string &b) const noexcept
46 {
47 size_t a_eq_pos = a.find('=');
48 size_t b_eq_pos = b.find('=');
49 return string_view(a.data(), a_eq_pos) == string_view(b.data(), b_eq_pos);
50 }
51
52 // For comparison between a string and string_view, we can assume the view is just the name
53
54 bool operator()(const std::string &a, string_view b) const noexcept
55 {
56 size_t a_eq_pos = a.find('=');
57 return string_view(a.data(), a_eq_pos) == b;
58 }
59
60 bool operator()(string_view a, const std::string &b) const noexcept
61 {
62 return operator()(b, a);
63 }

138 // return environment variable in form NAME=VALUE. Assumes that the real environment is the parent.
139 string_view get(const std::string &name) const
140 {
141 auto it = set_vars.find(string_view(name));
142 if (it != set_vars.end()) {
143 return *it;
144 }
145
146 if (!keep_parent_env && !import_from_parent.contains(name)) {
147 return {};
148 }
149
150 const char *val = bp_sys::getenv(name.c_str());
151 if (val == nullptr) {
152 return {};
153 }
154 const char *name_and_val = val - (name.length() + 1);
155 size_t val_len = strlen(val);
156 return {name_and_val, name.length() + 1 + val_len};
157 }

159 // Build a mapping excluding named variables (only called if the parent is the real environment).
160 // Note that the return is non-owning, i.e. the variable values are backed by the environment object
161 // and their lifetime is bounded to it.
162 env_map build(const env_names &exclude) const
163 {
164 env_map mapping;
165
166 if (keep_parent_env) {
167 // import all from parent, excluding our own undefines + the exclude set
168 if (bp_sys::environ != nullptr) {
169 unsigned pos = 0;
170 for (char **env = bp_sys::environ; *env != nullptr; ++env) {
171 // find '='
172 char *var_ch;
173 for (var_ch = *env; *var_ch != '='; ++var_ch) {
174 if (*var_ch == '\0') break;
175 }
176 // if this variable doesn't contain '=', ignore it
177 if (*var_ch == '\0') continue;
178 string_view name_view {*env, (size_t)(var_ch - *env)};
179
180 if (undefine.contains(name_view) || exclude.contains(name_view)) {
181 goto next_env_var;
182 }
183
184 mapping.env_list.push_back(*env);
185 mapping.var_map.insert({name_view, pos});
186 ++pos;
187
188 next_env_var: ;
189 }
190 }
191 }
192 else {
193 // import specific items from parent
194 for (const std::string &import_name : import_from_parent) {
195 // POSIX allows that getenv return its result in a static, per-thread buffer. Since this is
196 // ridiculous, we'll assume that all implementations do the sane thing of simply returning
197 // a pointer to the value part of the NAME=VALUE string in the actual environment array:
198 const char *value = getenv(import_name.c_str());
199 if (value == nullptr) continue;
200 // go back through the name and the '=' to the beginning of NAME=VALUE:
201 const char *name_and_val = value - (import_name.length() + 1);
202 mapping.var_map.insert({import_name, mapping.env_list.size()});
203 mapping.env_list.push_back(name_and_val);
204 }
205 }
206
207 // add our own (excluding exclude set)
208 for (const std::string &set_var : set_vars) {
209 size_t eq_pos = set_var.find('=');
210 string_view set_var_name = string_view(set_var.data(), eq_pos);
211
212 if (!exclude.contains(set_var_name)) {
213 auto iter = mapping.var_map.find(set_var_name);
214 if (iter != mapping.var_map.end()) {
215 unsigned pos = iter->second;
216 mapping.env_list[pos] = set_var.c_str();
217 }
218 else {
219 mapping.var_map[set_var_name] = mapping.env_list.size();
220 mapping.env_list.push_back(set_var.c_str());
221 }
222 }
223 }
224 mapping.env_list.push_back(nullptr);
225
226 return mapping;
227 }

229 env_map build(const environment &parent_env) const
230 {
231 env_map mapping;
232
233 // first build base: variables from the parent(s) excluding those specifically excluded
234 if (keep_parent_env) {
235 mapping = parent_env.build(undefine);
236 // remove final null entry:
237 mapping.env_list.resize(mapping.env_list.size() - 1);
238 }
239 else {
240 // import only those specifically chosen
241 for (const std::string &import_name : import_from_parent) {
242 string_view name_and_val = parent_env.get(import_name);
243 if (name_and_val.empty()) continue;
244 mapping.var_map.insert({import_name, mapping.env_list.size()});
245 mapping.env_list.push_back(name_and_val.data());
246 }
247 }
248
249 // add our own (excluding exclude set)
250 for (const std::string &set_var : set_vars) {
251 size_t eq_pos = set_var.find('=');
252 string_view set_var_name = string_view(set_var.data(), eq_pos);
253 auto iter = mapping.var_map.find(set_var_name);
254 if (iter != mapping.var_map.end()) {
255 unsigned pos = iter->second;

260 mapping.env_list.push_back(set_var.c_str());

274 void set_var(std::string &&var_and_val, bool notify = false)
275 {
276 string_view var_name = find_var_name(var_and_val);
277
278 import_from_parent.erase(var_name);
279 auto n_removed = undefine.erase(var_name);
280
281 bool in_sysenv = false;
282 if (notify && n_removed == 0) {
283 // workaround to avoid an allocation; when notify is true,
284 // we know for sure that the value is sanitized from before
285 //
286 // we don't check this when the variable was in undefine,
287 // as that means we were undefining it explicitly
288 auto name_size = var_name.size();
289 var_and_val[name_size] = '\0';
290 in_sysenv = !!bp_sys::getenv(var_and_val.c_str());
291 var_and_val[name_size] = '=';
292 }
293
294 auto insert_result = set_vars.insert(std::move(var_and_val));
295 // if the variable was in sys environment, it is always overridden
296 bool overridden = in_sysenv;
297 if (!insert_result.second) {
298 *insert_result.first = var_and_val;
299 overridden = true;
300 }
301
302 if (notify) {
303 notify_listeners(*insert_result.first, overridden);
304 }
305 }

307 void import_parent_var(std::string &&var_name)
308 {
309 undefine.erase(var_name);
310 set_vars.erase(string_view(var_name));
311 if (!keep_parent_env) {
312 import_from_parent.insert(std::move(var_name));
313 }
314 }
315
316 void undefine_var(std::string &&var_name, bool notify = false)
317 {
318 import_from_parent.erase(var_name);
319
320 auto n_removed = set_vars.erase(string_view(var_name));
321 bool was_set = n_removed > 0;
322 if (notify && !was_set) {
323 // also track if we're undefining it from system environment
324 was_set = !!bp_sys::getenv(var_name.c_str());
325 }
326 if (keep_parent_env) {
327 auto insert_result = undefine.insert(std::move(var_name));
328 if (notify) {
329 notify_listeners(*insert_result.first, was_set);
330 }
331 }
332 else if (notify) {
333 notify_listeners(var_name, was_set);
334 }
335 }

356// Read and set environment variables from a file. May throw std::bad_alloc, std::system_error.
357template <typename LOG_INV_SETTING, typename LOG_BAD_COMMAND>
358inline void read_env_file_inline(const char *env_file_path, bool log_warnings, environment &env,
359 bool throw_on_open_failure, LOG_INV_SETTING &log_inv_setting, LOG_BAD_COMMAND &log_bad_cmd)
360{
361 std::ifstream env_file(env_file_path);
362 if (!env_file) {
363 if (throw_on_open_failure) {
364 throw std::system_error(errno, std::generic_category());
365 }
366 return;
367 }
368
369 env_file.exceptions(std::ios::badbit);
370
371 auto &clocale = std::locale::classic();
372 std::string line;
373 int linenum = 0;
374
375 while (std::getline(env_file, line)) {
376 linenum++;
377 auto lpos = line.begin();
378 auto lend = line.end();
379 while (lpos != lend && std::isspace(*lpos, clocale)) {
380 ++lpos;
381 }
382
383 if (lpos == lend) continue; // empty line
384
385 if (*lpos == '#') {
386 continue;
387 }
388 if (*lpos == '=') {
389 if (log_warnings) {
390 log_inv_setting(linenum);
391 }
392 continue;
393 }
394
395 // "!COMMAND" form.
396 if (*lpos == '!') {
397 ++lpos; // lpos = first char of command
398 auto epos = lpos;
399 do {
400 ++epos;
401 } while(epos != lend && !std::isspace(*epos, clocale));
402
403 const char *lpos_p = line.data() + (lpos - line.begin());
404 string_view cmd {lpos_p, (size_t)(epos - lpos)};
405
406 std::vector<string_view> cmd_args;
407 while (epos != lend) {
408 // skip whitespace
409 while (std::isspace(*epos, clocale)) {
410 ++epos;
411 if (epos == lend) goto process_cmd; // no more args
412 }
413 // all non-ws is argument until next ws
414 const char *arg_begin = line.c_str() + (epos - line.begin());
415 auto arg_begin_i = epos;
416 while (epos != lend && !std::isspace(*epos)) {
417 ++epos;
418 }
419 cmd_args.push_back(string_view {arg_begin, (size_t)(epos - arg_begin_i)});
420 }
421
422 process_cmd:
423
424 if (cmd == "clear") {
425 env.clear_no_inherit();
426 }
427 else if (cmd == "unset") {
428 for (string_view arg : cmd_args) {
429 env.undefine_var(std::string(arg.data(), arg.length()));
430 }
431 }
432 else if (cmd == "import") {
433 for (string_view arg : cmd_args) {
434 env.import_parent_var(std::string(arg.data(), arg.length()));
435 }
436 }
437 else if (log_warnings) {
438 log_bad_cmd(linenum);
439 }
440
441 continue;
442 }
443
444 // ENV=VALUE form.
445 auto name_begin = lpos++;
446 // skip until '=' or whitespace:
447 while (lpos != lend && *lpos != '=' && !std::isspace(*lpos, clocale)) ++lpos;
448 auto name_end = lpos;
449 // skip whitespace:
450 while (lpos != lend && std::isspace(*lpos, clocale)) ++lpos;
451 if (lpos == lend || *lpos != '=') {
452 if (log_warnings) {
453 log_inv_setting(linenum);
454 }
455 continue;
456 }
457
458 ++lpos;
459 auto val_begin = lpos;
460 auto val_end = lend;
461
462 if (val_begin != (name_end + 1) || name_begin != line.begin()) {
463 // there are spaces that we need to eliminate
464 std::string name_and_val;
465 name_and_val.reserve((name_end - name_begin) + 1 + (val_end - val_begin));
466 name_and_val = line.substr(name_begin - line.begin(), name_end - name_begin);
467 name_and_val.append(1, '=');
468 name_and_val.append(val_begin, val_end);
469 env.set_var(std::move(name_and_val));
470 }
471 else {
472 line.shrink_to_fit();
473 env.set_var(std::move(line));
474 }
475 }
476}
