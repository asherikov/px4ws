14// Map of pid_t to void *, with possibility of reserving entries so that mappings can
15// be later added with no danger of allocator exhaustion (bad_alloc).
16class pid_map
17{
18 using bmap_t = btree_set<void *, pid_t>;
19 bmap_t b_map;
20
21 public:
22 using pid_handle_t = bmap_t::handle_t;
23
24 // Map entry: present (bool), data (void *)
25 using entry = std::pair<bool, void *>;
26
27 entry get(pid_t key) noexcept
28 {
29 auto it = b_map.find(key);
30 if (it == nullptr) {
31 return entry(false, nullptr);
32 }
33 return entry(true, b_map.node_data(*it));
34 }
35
36 entry remove(pid_t key) noexcept
37 {
38 auto it = b_map.find(key);
39 if (it == nullptr) {
40 return entry(false, nullptr);
41 }
42 b_map.remove(*it);
43 return entry(true, b_map.node_data(*it));
44 }
45
46 void remove(pid_handle_t &hndl)
47 {
48 if (b_map.is_queued(hndl)) {
49 b_map.remove(hndl);
50 }
51 }
52
53 // Throws bad_alloc on reservation failure
54 void reserve(pid_handle_t &hndl)
55 {
56 b_map.allocate(hndl);
57 }
58
59 void unreserve(pid_handle_t &hndl) noexcept
60 {
61 b_map.deallocate(hndl);
62 }
63
64 void add(pid_handle_t &hndl, pid_t key, void *val) // throws std::bad_alloc
65 {
66 reserve(hndl);
67 b_map.node_data(hndl) = val;
68 b_map.insert(hndl, key);
69 }
70
71 void add_from_reserve(pid_handle_t &hndl, pid_t key, void *val) noexcept
72 {
73 b_map.node_data(hndl) = val;
74 b_map.insert(hndl, key);
75 }
76};

133 template <typename T>
134 bool receive_signal(T & loop_mech, sigdata_t &siginfo, void *userdata)
135 {
136 if (siginfo.get_signo() == SIGCHLD) {
137 reaper_lock.lock();
138
139 siginfo_t child_info;
140 child_info.si_pid = 0; // for portability inc. MacOS
141 while (waitid(P_ALL, 0 /* ignored */, &child_info, WNOHANG | WEXITED) == 0) {
142 pid_t child = child_info.si_pid;
143 if (child == 0) break;
144 auto ent = child_waiters.remove(child);
145 if (ent.first) {
146 Base::receive_child_stat(child, { child_info.si_code, child_info.si_status },
147 ent.second);
148 }
149 child_info.si_pid = 0;
150 }
151
152 reaper_lock.unlock();
153 return false; // leave signal watch enabled
154 }
155 else {
156 return Base::receive_signal(loop_mech, siginfo, userdata);
157 }
