641 cerr << "dinitctl: " << ce.get_action() << ": " << ce.get_arg() << ": " << strerror(ce.get_err()) << "\n";
642 if (user_dinit && ce.get_err() == ENOENT && !user_specified_cs_path) {
643 // It is common enough that users don't realise they need to have a user instance
644 // running in order to control it, so elaborate a little:
645 cerr << "dinitctl: perhaps no user instance is running?\n";
646 }
647 }
648 catch (general_error &ge) {
649 std::cerr << "dinitctl";
650 if (ge.get_action() != nullptr) {
651 std::cerr << ": " << ge.get_action();
652 std::string &arg = ge.get_arg();
653 if (!arg.empty()) {
654 std::cerr << " " << arg;
655 }
656 }
657 if (ge.get_err() != 0) {
658 std::cerr << ": " << strerror(ge.get_err());
659 }
660 std::cerr << '\n';
661 }
662 return 1;
663}

686// Load a service: issue load command, wait for reply. Return true on success, display error message
687// and return false on failure.
688// socknum - the socket fd to communicate via
689// rbuffer - the buffer for communication
690// name - the name of the service to load
691// handle - where to store the handle of the loaded service
692// state - where to store the state of the loaded service (may be null).
693// write_error - whether to write an error message if the service can't be loaded
694static bool load_service(int socknum, cpbuffer_t &rbuffer, const char *name, handle_t *handle,
695 service_state_t *state, bool write_error=true)
696{
697 // Load 'to' service:
698 if (issue_load_service(socknum, name)) {
699 return false;
700 }
701
702 wait_for_reply(rbuffer, socknum);
703
704 if (check_load_reply(socknum, rbuffer, handle, state, write_error) != 0) {
705 return false;
706 }
707
708 return true;
709}

711// Get the service name for a given handle, by querying the daemon.
712static std::string get_service_name(int socknum, cpbuffer_t &rbuffer, handle_t handle)
713{
714 auto m = membuf()
715 .append((char) cp_cmd::QUERYSERVICENAME)
716 .append((char) 0)
717 .append(handle);
718 write_all_x(socknum, m);
719
720 wait_for_reply(rbuffer, socknum);
721
722 if (rbuffer[0] != (char)cp_rply::SERVICENAME) {
723 throw cp_read_exception{0};
724 }
725
726 // 1 byte reserved
727 // uint16_t size
728 fill_buffer_to(rbuffer, socknum, 2 + sizeof(uint16_t));
729 uint16_t namesize;
730 rbuffer.extract(&namesize, 2, sizeof(uint16_t));
731 rbuffer.consume(2 + sizeof(uint16_t));
732
733 std::string name;
734
735 do {
736 if (rbuffer.get_length() == 0) {
737 fill_some(rbuffer, socknum);
738 }
739
740 size_t to_extract = std::min(size_t(rbuffer.get_length()), namesize - name.length());
741 size_t contiguous_len = rbuffer.get_contiguous_length(rbuffer.get_ptr(0));
742 if (contiguous_len <= to_extract) {
743 name.append(rbuffer.get_ptr(0), contiguous_len);
744 rbuffer.consume(contiguous_len);
745 name.append(rbuffer.get_ptr(0), to_extract - contiguous_len);
746 rbuffer.consume(to_extract - contiguous_len);
747 }
748 else {
749 name.append(rbuffer.get_ptr(0), to_extract);
750 rbuffer.consume(to_extract);
751 break;
752 }
753
754 } while (name.length() < namesize);
755
756 return name;
757}

827// Process a SERVICEEVENT[5] packet if it is related to the specified service handle, and
828// optionally report the service status to the user (verbose == true). The caller must ensure that
829// a complete packet of type SERVICEEVENT[5] is present in the buffer before calling. The size of
830// the packet should be provided as pktlen.
831//
832// Returns 0 if the service started (do_stop == false) or stopped (do_stop == true), 1 if
833// start/stop was cancelled or failed, -1 when the service event is not related to given service
834// handle or does not correspond to a start (or stop) or failure.
835static int process_service_event(cpbuffer_t &rbuffer, unsigned pktlen, handle_t handle,
836 const std::string &service_name, bool do_stop, bool verbose)
837{
838 using std::cout;
839
840 // earlier versions do not include status info, the size in that case is
841 // base_pkt_size:
842 constexpr unsigned base_pkt_size = 2 + sizeof(handle_t) + 1;
843 if (pktlen < base_pkt_size) {
844 throw dinit_protocol_error();
845 }
846
847 // version 5 packets include extended staus info:
848 if (rbuffer[0] == (char)cp_info::SERVICEEVENT5 && pktlen < (base_pkt_size + STATUS_BUFFER5_SIZE)) {
849 throw dinit_protocol_error();
850 }
851
852 service_event_t completion_event;
853 service_event_t cancelled_event;
854 if (do_stop) {
855 completion_event = service_event_t::STOPPED;
856 cancelled_event = service_event_t::STOPCANCELLED;
857 }
858 else {
859 completion_event = service_event_t::STARTED;
860 cancelled_event = service_event_t::STARTCANCELLED;
861 }
862
863 handle_t ev_handle;
864 rbuffer.extract((char *)&ev_handle, 2, sizeof(ev_handle));
865 service_event_t event = static_cast<service_event_t>(rbuffer[2 + sizeof(ev_handle)]);
866 if (ev_handle == handle) {
867 if (event == completion_event) {
868 if (verbose) {
869 cout << "Service '" << service_name << "' " << describe_state(do_stop)
870 << ".\n";
871 }
872 rbuffer.consume(pktlen);
873 return 0;
874 }
875 else if (event == cancelled_event) {
876 if (verbose) {
877 cout << "Service '" << service_name << "' " << describe_verb(do_stop)
878 << " cancelled.\n";
879 }
880 rbuffer.consume(pktlen);
881 return 1;
882 }
883 else if (!do_stop && event == service_event_t::FAILEDSTART) {
884 if (verbose) {
885 cout << "Service '" << service_name << "' failed to start.\n";
886 if (pktlen >= base_pkt_size + STATUS_BUFFER_SIZE) {
887 uint16_t launch_stage;
888 rbuffer.extract((char *)&launch_stage, base_pkt_size + 4,
889 sizeof(uint16_t));
890
891 stopped_reason_t stop_reason = static_cast<stopped_reason_t>(rbuffer[base_pkt_size + 3]);
892 int exit_status;
893 int exit_si_code = 0;
894 int exit_si_status = 0;
895 rbuffer.extract((char *)&exit_status, base_pkt_size + 6, sizeof(exit_status));
896 if (rbuffer[0] == (char)cp_info::SERVICEEVENT5) {
897 exit_si_code = exit_status;
898 rbuffer.extract((char *)&exit_si_status,
899 base_pkt_size + 6 + sizeof(exit_si_code), sizeof(exit_si_status));
900 }
901
902 print_failure_details(stop_reason, launch_stage, exit_status, exit_si_code, exit_si_status);
903 }
904 }
905 rbuffer.consume(pktlen);
906 return 1;
907 }
908 }
909 rbuffer.consume(pktlen);
910 return -1;
911}

913// Wait for a service to reached stopped (do_stop == true) or started (do_stop == false) state.
914// Returns 0 if the service started/stopped, 1 if start/stop was cancelled or failed.
915static int wait_service_state(int socknum, cpbuffer_t &rbuffer, handle_t handle,
916 const std::string &service_name, bool do_stop, bool verbose)
917{
918 // Wait until service started:
919 int r = rbuffer.fill_to(socknum, 2);
920 while (r > 0) {
921 if (rbuffer[0] >= 100) {
922 unsigned pktlen = (unsigned char) rbuffer[1];
923 fill_buffer_to(rbuffer, socknum, pktlen);
924
925 if (value((cp_info)rbuffer[0]).is_in(cp_info::SERVICEEVENT, cp_info::SERVICEEVENT5)) {
926 int ret = process_service_event(rbuffer, pktlen, handle, service_name, do_stop, verbose);
927 if (ret >= 0) {
928 return ret;
929 }
930 }
931 else {
932 rbuffer.consume(pktlen);
933 }
934
935 r = rbuffer.fill_to(socknum, 2);
936 }
937 else {
938 // Not an information packet?
939 throw dinit_protocol_error();
940 }
941 }
942
943 if (r == -1) {
944 perror("dinitctl: read");
945 }
946 else {
947 throw dinit_protocol_error();
948 }
949
950 return 1;
951}

953// Start/stop/restart a service
954static int start_stop_service(int socknum, cpbuffer_t &rbuffer, const char *service_name,
955 ctl_cmd command, bool do_pin, bool do_force, bool wait_for_service, bool ignore_unstarted,
956 bool verbose)
957{
958 using namespace std;
959
960 bool do_stop = (command == ctl_cmd::STOP_SERVICE || command == ctl_cmd::RELEASE_SERVICE);
961
962 service_state_t state;
963 handle_t handle;
964
965 if (command != ctl_cmd::RESTART_SERVICE && command != ctl_cmd::STOP_SERVICE
966 && command != ctl_cmd::RELEASE_SERVICE) {
967 ignore_unstarted = false;
968 }
969
970 if (!load_service(socknum, rbuffer, service_name, &handle, &state, !ignore_unstarted)) {
971 return ignore_unstarted ? 0 : 1;
972 }
973
974 service_state_t wanted_state = do_stop ? service_state_t::STOPPED : service_state_t::STARTED;
975 cp_cmd pcommand;
976 switch (command) {
977 case ctl_cmd::STOP_SERVICE:
978 case ctl_cmd::RESTART_SERVICE: // stop, and then start
979 pcommand = cp_cmd::STOPSERVICE;
980 break;
981 case ctl_cmd::RELEASE_SERVICE:
982 pcommand = cp_cmd::RELEASESERVICE;
983 break;
984 case ctl_cmd::START_SERVICE:
985 pcommand = cp_cmd::STARTSERVICE;
986 break;
987 case ctl_cmd::WAKE_SERVICE:
988 pcommand = cp_cmd::WAKESERVICE;
989 break;
990 default:
991 // can't get here (hopefully)
992 pcommand = cp_cmd::STOPSERVICE;
993 }
994
995 observed_states_t seen_states;
996
997 // Need to issue command (eg STOPSERVICE/STARTSERVICE)
998 // We'll do this regardless of the current service state / target state, since issuing
999 // start/stop also sets or clears the "explicitly started" flag on the service.
1000 // (For other commands we could potentially avoid sending the command if we had more
1001 // information about the service state, but it doesn't seem worth the effort to implement
1002 // that optimisation).
1003 {
1004 char flags = (do_pin ? 1 : 0) | ((pcommand == cp_cmd::STOPSERVICE && !do_force) ? 2 : 0);
1005 if (command == ctl_cmd::RESTART_SERVICE) {
1006 flags |= (4 | 128); // restart, pre-ack
1007 }
1008
1009 auto m = membuf()
1010 .append((char) pcommand)
1011 .append(flags)
1012 .append(handle);
1013 write_all_x(socknum, m);
1014
1015 wait_for_reply(rbuffer, socknum, handle, &seen_states);
1016 cp_rply reply_pkt_h = (cp_rply)rbuffer[0];
1017 rbuffer.consume(1); // consume header
1018
1019 if (reply_pkt_h == cp_rply::PREACK) {
1020 // We should consider state changes seen only after the PREACK (i.e. between the
1021 // PREACK and the main reply):
1022 seen_states.started = false;
1023 seen_states.stopped = false;
1024 seen_states.failed_start = false;
1025
1026 // PREACK will be followed by a 2nd reply, get that now:
1027 wait_for_reply(rbuffer, socknum, handle, &seen_states);
1028 reply_pkt_h = (cp_rply)rbuffer[0];
1029 rbuffer.consume(1);
1030 }
1031
1032 if (reply_pkt_h == cp_rply::ALREADYSS) {
1033 bool already = (state == wanted_state);
1034 if (verbose) {
1035 if (command == ctl_cmd::RESTART_SERVICE) {
1036 cout << "Service restarted.\n";
1037 }
1038 else {
1039 cout << "Service " << (already ? "(already) " : "")
1040 << describe_state(do_stop) << ".\n";
1041 }
1042 }
1043 return 0; // success!
1044 }
1045 if (reply_pkt_h == cp_rply::PINNEDSTARTED) {
1046 cerr << "dinitctl: cannot stop service '" << service_name << "' as it is pinned started\n";
1047 return 1;
1048 }
1049 if (reply_pkt_h == cp_rply::PINNEDSTOPPED) {
1050 cerr << "dinitctl: cannot start service '" << service_name << "' as it is pinned stopped\n";
1051 return 1;
1052 }
1053 if (reply_pkt_h == cp_rply::DEPENDENTS && pcommand == cp_cmd::STOPSERVICE) {
1054 cerr << "dinitctl: cannot stop service '" << service_name << "' due to the following dependents:\n";
1055 if (command != ctl_cmd::RESTART_SERVICE) {
1056 cerr << "(only direct dependents are listed. Exercise caution before using '--force' !!)\n";
1057 }
1058 // size_t number, N * handle_t handles
1059 size_t number;
1060 rbuffer.fill_to(socknum, sizeof(number));
1061 rbuffer.extract(&number, 0, sizeof(number));
1062 rbuffer.consume(sizeof(number));
1063 std::vector<handle_t> handles;
1064 handles.reserve(number);
1065 for (size_t i = 0; i < number; i++) {
1066 handle_t handle;
1067 rbuffer.fill_to(socknum, sizeof(handle_t));
1068 rbuffer.extract(&handle, 0, sizeof(handle));
1069 handles.push_back(handle);
1070 rbuffer.consume(sizeof(handle));
1071 }
1072 // Print the directly affected dependents:
1073 cerr << " ";
1074 for (handle_t handle : handles) {
1075 cerr << " " << get_service_name(socknum, rbuffer, handle);
1076 }
1077 cerr << "\n";
1078 return 1;
1079 }
1080 if (reply_pkt_h == cp_rply::NAK && command == ctl_cmd::RESTART_SERVICE) {
1081 if (ignore_unstarted) {
1082 if (verbose) {
1083 cout << "Service '" << service_name << "' is not currently started.\n";
1084 }
1085 return 0;
1086 }
1087 cerr << "dinitctl: cannot restart service; service not started.\n";
1088 return 1;
1089 }
1090 if (reply_pkt_h == cp_rply::NAK && command == ctl_cmd::WAKE_SERVICE) {
1091 cerr << "dinitctl: service has no active dependents, cannot wake.\n";
1092 return 1;
1093 }
1094 if (reply_pkt_h == cp_rply::SHUTTINGDOWN) {
1095 cerr << "dinitctl: cannot start/restart/wake service, shutdown is in progress.\n";
1096 return 1;
1097 }
1098 if (reply_pkt_h != cp_rply::ACK) {
1099 cerr << "dinitctl: protocol error." << endl;
1100 return 1;
1101 }
1102 }
1103
1104 if (!wait_for_service) {
1105 if (verbose) {
1106 cout << "Issued " << describe_verb(do_stop) << " command successfully for service '"
1107 << service_name << "'." << endl;
1108 }
1109 return 0;
1110 }
1111
1112 if (command == ctl_cmd::RESTART_SERVICE) {
1113 // for restart we want to display both "stopped" and "started" statuses
1114 if (seen_states.stopped) {
1115 if (verbose) {
1116 cout << "Service '" << service_name << "' stopped.\n";
1117 }
1118 }
1119 else if (wait_service_state(socknum, rbuffer, handle, service_name, true, verbose) != 0) {
1120 return EXIT_FAILURE;
1121 }
1122 }
1123
1124 if (seen_states.started) {
1125 if (verbose) {
1126 cout << "Service '" << service_name << "' started.\n";
1127 }
1128 }
1129 else if (seen_states.failed_start) {
1130 if (verbose) {
1131 cout << "Service '" << service_name << "' failed to start.\n";
1132 print_failure_details(seen_states.stop_reason, 0 /* not applicable */,
1133 seen_states.exit_status, seen_states.exit_si_code, seen_states.exit_si_status);
1134 }
1135 }
1136 return wait_service_state(socknum, rbuffer, handle, service_name, do_stop, verbose);
1137}

1139// Issue a "load service" command (LOADSERVICE), without waiting for
1140// a response. Returns 1 on failure (with error logged), 0 on success.
1141static int issue_load_service(int socknum, const char *service_name, bool find_only)
1142{
1143 // Build buffer;
1144 srvname_len_t srvname_len = strlen(service_name);
1145 int bufsize = 3 + srvname_len;
1146
1147 std::unique_ptr<char[]> ubuf(new char[bufsize]);
1148 auto buf = ubuf.get();
1149
1150 buf[0] = (char)(find_only ? cp_cmd::FINDSERVICE : cp_cmd::LOADSERVICE);
1151 memcpy(buf + 1, &srvname_len, sizeof(srvname_len));
1152 memcpy(buf + 3, service_name, srvname_len);
1153
1154 write_all_x(socknum, buf, bufsize);
1155
1156 return 0;
1157}

1279static int reload_service(int socknum, cpbuffer_t &rbuffer, const char *service_name, bool verbose)
1280{
1281 using namespace std;
1282
1283 if (issue_load_service(socknum, service_name, true) == 1) {
1284 return 1;
1285 }
1286
1287 wait_for_reply(rbuffer, socknum);
1288
1289 handle_t handle;
1290
1291 if (rbuffer[0] == (char)cp_rply::NOSERVICE) {
1292 rbuffer.consume(1);
1293 // If the service isn't loaded yet at all, just do a basic load:
1294 if (issue_load_service(socknum, service_name, false) == 1) {
1295 return 1;
1296 }
1297
1298 wait_for_reply(rbuffer, socknum);
1299
1300 if (check_load_reply(socknum, rbuffer, &handle, nullptr) != 0) {
1301 return 1;
1302 }
1303
1304 if (verbose) {
1305 cout << "Service '" << service_name << "' reloaded." << endl;
1306 }
1307 return 0;
1308 }
1309
1310 if (check_load_reply(socknum, rbuffer, &handle, nullptr) != 0) {
1311 return 1;
1312 }

1315 {

1318 .append(handle);

1332 }
1333
1334 if (verbose) {
1335 cout << "Service '" << service_name << "' reloaded." << endl;
1336 }
1337 return 0;

1345 if (proto_version >= 5) {
1346 cmdbuf[0] = (char)cp_cmd::LISTSERVICES5;

1443 cout << ' ';

1493static int service_status(int socknum, cpbuffer_t &rbuffer, const char *service_name,
1494 ctl_cmd command, uint16_t proto_version, bool verbose)
1495{
1496 using namespace std;
1497
1498 bool is_status = command == ctl_cmd::SERVICE_STATUS;
1499
1500 if (issue_load_service(socknum, service_name, true) == 1) {
1501 return 1;
1502 }
1503
1504 wait_for_reply(rbuffer, socknum);
1505
1506 handle_t handle;
1507
1508 if (rbuffer[0] == (char)cp_rply::NOSERVICE) {
1509 if (is_status) {
1510 cerr << "dinitctl: service not loaded." << endl;
1511 }
1512 return 1;
1513 }
1514
1515 if (check_load_reply(socknum, rbuffer, &handle, nullptr, is_status) != 0) {
1516 return 1;
1517 }
1518
1519 // Issue STATUS request
1520 {
1521 char status_req_id = proto_version < 5 ? (char)cp_cmd::SERVICESTATUS : (char)cp_cmd::SERVICESTATUS5;
1522 unsigned status_buf_size = proto_version < 5 ? STATUS_BUFFER_SIZE : STATUS_BUFFER5_SIZE;
1523
1524 auto m = membuf()
1525 .append(status_req_id)
1526 .append(handle);
1527 write_all_x(socknum, m);
1528
1529 wait_for_reply(rbuffer, socknum);
1530 if (rbuffer[0] != (char)cp_rply::SERVICESTATUS) {
1531 cerr << "dinitctl: protocol error." << endl;
1532 return 1;
1533 }
1534 rbuffer.consume(1);
1535
1536 fill_buffer_to(rbuffer, socknum, status_buf_size + 1 /* reserved */);
1537 rbuffer.consume(1);
1538
1539 service_state_t current = static_cast<service_state_t>(rbuffer[0]);
1540 service_state_t target = static_cast<service_state_t>(rbuffer[1]);
1541
1542 int console_flags = rbuffer[2];
1543 bool has_console = (console_flags & 2) != 0;
1544 bool waiting_console = (console_flags & 1) != 0;
1545 bool was_skipped = (console_flags & 4) != 0;
1546 bool marked_active = (console_flags & 8) != 0;
1547 bool has_pid = (console_flags & 16) != 0;
1548
1549 stopped_reason_t stop_reason = static_cast<stopped_reason_t>(rbuffer[3]);
1550
1551 pid_t service_pid = -1;
1552 int exit_status = 0;
1553 int exit_si_code = 0;
1554 int exit_si_status = 0;
1555 if (has_pid) {
1556 // 6 = current state (1) + target state (1) + flags (1)
1557 // + stop reason (1) + exec failure stage (2)
1558 rbuffer.extract((char *)&service_pid, 6, sizeof(service_pid));
1559 }
1560 else {
1561 if (proto_version < 5) {
1562 rbuffer.extract((char *)&exit_status, 6, sizeof(exit_status));
1563 }
1564 else {
1565 rbuffer.extract((char *)&exit_si_code, 6, sizeof(exit_si_code));
1566 rbuffer.extract((char *)&exit_si_status, 6 + sizeof(int), sizeof(exit_si_status));
1567 }
1568 }
1569
1570 switch (command) {
1571 case ctl_cmd::IS_STARTED:
1572 case ctl_cmd::IS_FAILED:
1573 if (verbose) {
1574 switch (current) {
1575 case service_state_t::STOPPED:
1576 cout << "STOPPED" << endl;
1577 break;
1578 case service_state_t::STARTING:
1579 cout << "STARTING" << endl;
1580 break;
1581 case service_state_t::STARTED:
1582 cout << "STARTED" << endl;
1583 break;
1584 case service_state_t::STOPPING:
1585 cout << "STOPPING" << endl;
1586 }
1587 }
1588 if (command == ctl_cmd::IS_STARTED) {
1589 // return 0 (success) for started
1590 return current != service_state_t::STARTED;
1591 }
1592 // IS_FAILED:
1593 // return 0 (success) for specific stopped reasons
1594 if (current == service_state_t::STOPPED) {
1595 switch (stop_reason) {
1596 case stopped_reason_t::DEPFAILED:
1597 case stopped_reason_t::FAILED:
1598 case stopped_reason_t::EXECFAILED:
1599 case stopped_reason_t::TIMEDOUT:
1600 return 0;
1601 default:
1602 break;
1603 }
1604 }
1605 return 1;
1606 default:
1607 /* status */
1608 break;
1609 }
1610
1611 cout << "Service: " << service_name << "\n"
1612 " State: ";
1613
1614 switch (current) {
1615 case service_state_t::STOPPED:
1616 cout << "STOPPED";
1617 switch (stop_reason) {
1618 case stopped_reason_t::DEPRESTART:
1619 cout << " (dependency restarted)";
1620 break;
1621 case stopped_reason_t::DEPFAILED:
1622 cout << " (dependency failed/terminated)";
1623 break;
1624 case stopped_reason_t::FAILED:
1625 cout << " (failed to start";
1626 if (proto_version < 5) {
1627 if (exit_status != 0) {
1628 cout << "; ";
1629 print_termination_details(exit_status);
1630 }
1631 }
1632 else {
1633 if (exit_si_status != 0) {
1634 cout << "; ";
1635 print_termination_details(exit_si_code, exit_si_status);
1636 }
1637 }
1638 cout << ")";
1639 break;
1640 case stopped_reason_t::EXECFAILED:
1641 uint16_t launch_stage;
1642 rbuffer.extract((char *)&launch_stage, 4, 2);
1643 if (exit_status == 0) {
1644 // (Protocol version 5+)
1645 exit_status = exit_si_code;
1646 }
1647 cout << " (could not be launched)\n";
1648 cout << " Stage: " << exec_stage_descriptions[launch_stage] << "\n";
1649 cout << " Error: " << strerror(exit_status);
1650 break;
1651 case stopped_reason_t::TERMINATED:
1652 cout << " (terminated";
1653 if (proto_version < 5) {
1654 if (exit_status != 0) {
1655 cout << "; ";
1656 print_termination_details(exit_status);
1657 }
1658 }
1659 else {
1660 if (exit_si_status != 0) {
1661 cout << "; ";
1662 print_termination_details(exit_si_code, exit_si_status);
1663 }
1664 }
1665 cout << ")";
1666 break;
1667 case stopped_reason_t::TIMEDOUT:
1668 cout << " (start timed out)";
1669 break;
1670 case stopped_reason_t::NORMAL:
1671 break;
1672 }
1673 break;
1674 case service_state_t::STARTING:
1675 cout << "STARTING";
1676 if (target == service_state_t::STOPPED) {
1677 cout << " (target state: STOPPED)";
1678 }
1679 break;
1680 case service_state_t::STARTED:
1681 cout << "STARTED";
1682 if (was_skipped) {
1683 cout << " (startup skipped)";
1684 }
1685 break;
1686 case service_state_t::STOPPING:
1687 cout << "STOPPING";
1688 if (target == service_state_t::STARTED) {
1689 cout << " (target state: STARTED)";
1690 }
1691 if (proto_version < 5) {
1692 if (exit_status != 0) {
1693 cout << "(terminated ;";
1694 print_termination_details(exit_status);
1695 cout << ")";
1696 }
1697 }
1698 else {
1699 if (exit_si_status != 0) {
1700 cout << "(terminated ;";
1701 print_termination_details(exit_si_code, exit_si_status);
1702 cout << ")";
1703 }
1704 }
1705 }
1706 if (has_console) {
1707 cout << " (holding console)";
1708 }
1709 if (waiting_console) {
1710 cout << " (waiting for console)";
1711 }
1712 cout << "\n";
1713
1714 if (target == service_state_t::STARTED) {
1715 cout << " Activation: ";
1716 if (marked_active) {
1717 cout << "explicitly started\n";
1718 }
1719 else {
1720 cout << "start due to dependent(s)\n";
1721 }
1722 }
1723
1724 if (service_pid != -1) {
1725 cout << " Process ID: " << service_pid << "\n";
1726 }
1727 }
1728
1729 return 0;
1730}

1820// Get the service description directory for a loaded service
1821static std::string get_service_description_dir(int socknum, cpbuffer_t &rbuffer, handle_t service_handle)
1822{
1823 auto m = membuf()
1824 .append((char)cp_cmd::QUERYSERVICEDSCDIR)
1825 .append<char>(0)
1826 .append(service_handle);
1827
1828 write_all_x(socknum, m);
1829 wait_for_reply(rbuffer, socknum);
1830
1831 if (rbuffer[0] != (char)cp_rply::SVCDSCDIR) {
1832 throw dinit_protocol_error();
1833 }
1834 rbuffer.consume(1);
1835
1836 fill_buffer_to(rbuffer, socknum, 4);
1837
1838 uint32_t sdir_len;
1839 rbuffer.extract(&sdir_len, 0, sizeof(uint32_t));
1840 rbuffer.consume(4);
1841
1842 std::string result_str;
1843 static_assert(sizeof(unsigned) >= sizeof(uint32_t), "");
1844 unsigned needed = sdir_len;
1845
1846 while (needed > 0) {
1847 unsigned available = rbuffer.get_length();
1848
1849 if (available == 0) {
1850 fill_some(rbuffer, socknum);
1851 available = rbuffer.get_length();
1852 }
1853
1854 unsigned to_use = std::min(available, needed);
1855 size_t orig_len = result_str.length();
1856 result_str.resize(orig_len + to_use);
1857 rbuffer.extract(&result_str[orig_len], 0, to_use);
1858 rbuffer.consume(to_use);
1859 needed -= to_use;
1860 }
1861
1862 return result_str;
1863}

1906static int enable_disable_service(int socknum, cpbuffer_t &rbuffer, service_dir_opt &service_dir_opts,
1907 const char *from, const char *to, bool enable, bool verbose, uint16_t proto_version)
1908{

1911 service_state_t from_state = service_state_t::STARTED;
1912 handle_t from_handle;
1913
1914 handle_t to_handle;
1915
1916 vector<string> service_dir_paths;
1917
1918 string service_file_path;
1919 string to_service_file_path;
1920 ifstream service_file;
1921
1922 if (strchr(from, '@') != nullptr) {
1923 cerr << "dinitctl: cannot enable/disable from a service with argument (service@arg).\n";
1924 return 1;
1925 }
1926
1927 if (socknum >= 0) {
1928 if (!load_service(socknum, rbuffer, from, &from_handle, &from_state)
1929 || !load_service(socknum, rbuffer, to, &to_handle, nullptr)) {
1930 return 1;
1931 }
1932
1933 try {
1934 service_dir_paths = get_service_description_dirs(socknum, rbuffer);
1935 }
1936 catch (dinit_protocol_error &) {
1937 cerr << "dinitctl: unknown configuration or protocol error, unable to load service descriptions\n";
1938 }
1939
1940 service_file_path = get_service_descr_filename(socknum, rbuffer, from_handle, from);
1941 to_service_file_path = get_service_descr_filename(socknum, rbuffer, to_handle, to);
1942
1943 // open from file
1944 service_file.open(service_file_path.c_str());
1945 if (!service_file) {
1946 cerr << "dinitctl: could not open service description file '"
1947 << service_file_path << "': " << strerror(errno) << "\n";
1948 return EXIT_FAILURE;
1949 }
1950 }
1951 else {
1952 // offline case
1953 const auto &path_list = service_dir_opts.get_paths();
1954 for (auto &path : path_list) {
1955 service_dir_paths.emplace_back(path.get_dir());
1956 }
1957
1958 find_service_desc(from, service_dir_paths, service_file, service_file_path);
1959 if (!service_file) {
1960 if (errno == ENOENT) {
1961 cerr << "dinitctl: could not locate service file for service '" << from << "'\n";
1962 }
1963 else {
1964 cerr << "dinitctl: could not open service description file '"
1965 << service_file_path << "': " << strerror(errno) << "\n";
1966 }
1967 return EXIT_FAILURE;
1968 }
1969
1970 ifstream to_service_file;
1971 find_service_desc(to, service_dir_paths, to_service_file, to_service_file_path);
1972 if (!to_service_file) {
1973 cerr << "dinitctl: could not locate service file for target service '" << to << "'" << endl;
1974 return EXIT_FAILURE;
1975 }
1976 }
1977
1978 // We now need to read the service file, identify the waits-for.d directory (bail out if more than one),
1979 // make sure the service is not listed as a dependency individually.
1980
1981 string waits_for_d;
1982
1983 service_file.exceptions(ios::badbit);
