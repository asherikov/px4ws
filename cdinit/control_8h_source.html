251inline dasynq::rearm control_conn_cb(eventloop_t * loop, control_conn_watcher * watcher, int revents)
252{
253 // Get the address of the containing control_connt_t object:
254 _Pragma ("GCC diagnostic push")
255 _Pragma ("GCC diagnostic ignored \"-Winvalid-offsetof\"")
256 uintptr_t cc_addr = reinterpret_cast<uintptr_t>(watcher) - offsetof(control_conn_t, iob);
257 control_conn_t *conn = reinterpret_cast<control_conn_t *>(cc_addr);
258 _Pragma ("GCC diagnostic pop")
259
260 if (revents & dasynq::IN_EVENTS) {
261 if (conn->data_ready()) {
262 delete conn;
263 return dasynq::rearm::REMOVED;
264 }
265 }
266 if (revents & dasynq::OUT_EVENTS) {
267 if (conn->send_data()) {
268 delete conn;
269 return dasynq::rearm::REMOVED;
270 }
271 }
272
273 // Accept more commands unless the output buffer high water mark is exceeded
274 int watch_flags = 0;
275 if (!conn->bad_conn_close && conn->outbuf_size < OUTBUF_LIMIT) {
276 watch_flags |= dasynq::IN_EVENTS;
277 }
278 if (!conn->outbuf.empty() || conn->bad_conn_close) {
279 watch_flags |= dasynq::OUT_EVENTS;
280 }
281 watcher->set_watches(watch_flags);
282
283 return dasynq::rearm::NOOP;
284}
