21// "kqueue"-based event loop mechanism.
22//
23// kqueue is available on BSDs and Mac OS X, though there are subtle differences from OS to OS.
24//
25// kqueue supports watching file descriptors (input and output as separate watches only),
26// signals, child processes, and timers. Unfortunately support for the latter two is imperfect;
27// it is not possible to reserve process watches in advance; timers can only be active, count
28// down immediately when created, and cannot be reset to another time. For timers especially
29// the problems are significant: we can't allocate timers in advance, and we can't even feasibly
30// manage our own timer queue via a single kqueue-backed timer. Therefore, an alternate timer
31// mechanism must be used together with kqueue.

183 // The kqueue signal reporting mechanism *coexists* with the regular signal
184 // delivery mechanism without having any connection to it. Whereas regular signals can be
185 // queued (especially "realtime" signals, via sigqueue()), kqueue just maintains a counter
186 // of delivery attempts and clears this when we read the event. What this means is that
187 // kqueue won't necessarily tell us if signals are pending, in the case that:
188 // 1) it already reported the attempted signal delivery and
189 // 2) more than instance one of the same signal was pending at that time and
190 // 3) no more deliveries of the same signal have been attempted in the meantime.
191 // Also, kqueue won't tell us about signals that were pending at the time the signal filter
192 // was added. Finally, because pending signals can be merged, the count of delivery attempts
193 // provided by kqueue does not necessarily match the number of signals actually pending.
194 //
195 // Note that POSIX allows for multiple instances of a signal to be pending even on systems
196 // that don't support queueing of signals.
197 //
198 // Ultimately, this means we need to check for pending signals independently of what kqueue
199 // tells us.
200
201 // Base contains:
202 // lock - a lock that can be used to protect internal structure.
203 // receive*() methods will be called with lock held.
204 // receive_signal(sigdata_t &, user *) noexcept
205 // receive_fd_event(fd_r, user *, int flags) noexcept
206
207 using sigdata_t = kqueue_traits::sigdata_t;
208 using fd_r = typename kqueue_traits::fd_r;

210 // The flag to specify poll() semantics for regular file readiness: that is, we want
211 // ready-for-read to be returned even at end of file:
212#if defined(NOTE_FILE_POLL)
213 // FreeBSD:
214 constexpr static int POLL_SEMANTICS = NOTE_FILE_POLL;
215#else
216 // Note that macOS has an "EV_POLL" defined that looks like it should give poll semantics
217 // when passed as a flag. However, it is filtered at the syscall entry so we cannot use it in
218 // kqueue. (The kernel uses it internally to implement poll()).
219 constexpr static int POLL_SEMANTICS = 0;

222 void process_events(struct kevent *events, int r)
223 {
224 std::lock_guard<decltype(Base::lock)> guard(Base::lock);
225
226 for (int i = 0; i < r; i++) {
227 if (events[i].filter == EVFILT_SIGNAL) {
228 bool reenable = pull_signal(events[i].ident, events[i].udata);
229 events[i].flags = reenable ? EV_ENABLE : EV_DISABLE;
230 }
231 else if (events[i].filter == EVFILT_READ || events[i].filter == EVFILT_WRITE) {
232 int flags = events[i].filter == EVFILT_READ ? IN_EVENTS : OUT_EVENTS;
233 auto r = Base::receive_fd_event(*this, fd_r(events[i].ident), events[i].udata, flags);
234 if (std::get<0>(r) == 0) {
235 // we use EV_CLEAR to clear the EOF status of fifos/pipes (and wait for
236 // another connection).
237 events[i].flags = EV_DISABLE | EV_CLEAR;
238 }
239 else {
240 events[i].flags = EV_ENABLE;
241 }
242 }
243 else {
244 events[i].flags = EV_DISABLE;
245 }
246 }
247
248 // Now we disable all received events, to simulate EV_DISPATCH:
249 kevent(kqfd, events, r, nullptr, 0, nullptr);
250 }

252 // Pull a signal from pending, and report it, until it is no longer pending or the watch
253 // should be disabled. Call with lock held.
254 // Returns: true if watcher should be enabled, false if disabled.
255 bool pull_signal(int signo, void *userdata)
256 {
257 bool enable_filt = true;
258 sigdata_t siginfo;
259
260#if _POSIX_REALTIME_SIGNALS > 0
261 struct timespec timeout = {0, 0};
262 sigset_t sigw_mask;
263 sigemptyset(&sigw_mask);
264 sigaddset(&sigw_mask, signo);
265 int rsigno = sigtimedwait(&sigw_mask, &siginfo.info, &timeout);
266 while (rsigno > 0) {
267 if (Base::receive_signal(*this, siginfo, userdata)) {
268 enable_filt = false;
269 break;
270 }
271 rsigno = sigtimedwait(&sigw_mask, &siginfo.info, &timeout);
272 }
273#else
274 // we have no sigtimedwait.
275 sigset_t pending_sigs;
276 sigpending(&pending_sigs);
277 while (sigismember(&pending_sigs, signo)) {
278 dprivate::dkqueue::get_siginfo(signo, &siginfo.info);
279 if (Base::receive_signal(*this, siginfo, userdata)) {
280 enable_filt = false;
281 break;
282 }
283 sigpending(&pending_sigs);
284 }
285#endif
286 return enable_filt;
287 }

347 // fd: file descriptor to watch
348 // userdata: data to associate with descriptor
349 // flags: IN_EVENTS | OUT_EVENTS | ONE_SHOT
350 // (only one of IN_EVENTS/OUT_EVENTS can be specified)
351 // soft_fail: true if unsupported file descriptors should fail by returning false instead
352 // of throwing an exception
353 // returns: true on success; false if file descriptor type isn't supported and emulate == true
354 // throws: std::system_error or std::bad_alloc on failure
355 bool add_fd_watch(int fd, void *userdata, int flags, bool enabled = true, bool emulate = false)
356 {
357 short filter = (flags & IN_EVENTS) ? EVFILT_READ : EVFILT_WRITE;
358
359 if (filter == EVFILT_READ && POLL_SEMANTICS == 0 && emulate) {
360 // We can't request poll semantics, so check for regular file:
361 struct stat statbuf;
362 if (fstat(fd, &statbuf) == -1) {
363 throw std::system_error(errno, std::system_category());
364 }
365 if ((statbuf.st_mode & S_IFMT) == S_IFREG) {
366 // Regular file: emulation required
367 return false;
368 }
369 }
370
371 int fflags = (filter == EVFILT_READ) ? POLL_SEMANTICS : 0;
372
373 struct kevent kev;
374 EV_SET(&kev, fd, filter, EV_ADD | (enabled ? 0 : EV_DISABLE), fflags, 0, userdata);
375 if (kevent(kqfd, &kev, 1, nullptr, 0, nullptr) == -1) {
376 // Note that kqueue supports EVFILT_READ on regular file fd's, but not EVFILT_WRITE.
377 // Regular files return EINVAL, but /dev/null apparently gives ENODEV (which
378 // documentation doesn't mention).
379 if (filter == EVFILT_WRITE && (errno == EINVAL || errno == ENODEV) && emulate) {
380 return false; // emulate
381 }
382 throw std::system_error(errno, std::system_category());
383 }
384 return true;
385 }

387 // returns: 0 on success
388 // IN_EVENTS if in watch requires emulation
389 // OUT_EVENTS if out watch requires emulation
390 int add_bidi_fd_watch(int fd, void *userdata, int flags, bool emulate = false)
391 {
392#ifdef EV_RECEIPT
393 struct kevent kev[2];
394 struct kevent kev_r[2];
395 short rflags = EV_ADD | ((flags & IN_EVENTS) ? 0 : EV_DISABLE) | EV_RECEIPT;
396 short wflags = EV_ADD | ((flags & OUT_EVENTS) ? 0 : EV_DISABLE) | EV_RECEIPT;
397 EV_SET(&kev[0], fd, EVFILT_READ, rflags, POLL_SEMANTICS, 0, userdata);
398 EV_SET(&kev[1], fd, EVFILT_WRITE, wflags, 0, 0, userdata);
399
400 int r = kevent(kqfd, kev, 2, kev_r, 2, nullptr);
401
402 if (r == -1) {
403 throw std::system_error(errno, std::system_category());
404 }
405
406 // Some possibilities:
407 // - both ends failed. We'll throw an error rather than allowing emulation.
408 // - read watch failed, write succeeded : should not happen.
409 // - read watch added, write failed: if emulate == true, succeed;
410 // if emulate == false, remove read and fail.
411
412 if (kev_r[0].data != 0) {
413 // read failed
414 throw std::system_error(kev_r[0].data, std::system_category());
415 }
416
417 if (kev_r[1].data != 0) {
418 if (emulate) {
419 // We can emulate, but, do we have correct semantics?
420 if (POLL_SEMANTICS != 0) {
421 return OUT_EVENTS;
422 }
423
424 // if we can't get poll semantics, emulate for read as well:
425 // first remove read watch:
426 EV_SET(&kev[0], fd, EVFILT_READ, EV_DELETE, 0, 0, userdata);
427 kevent(kqfd, kev, 1, nullptr, 0, nullptr);
428 return IN_EVENTS | OUT_EVENTS;
429 }
430 // remove read watch
431 EV_SET(&kev[0], fd, EVFILT_READ, EV_DELETE, 0, 0, userdata);
432 kevent(kqfd, kev, 1, nullptr, 0, nullptr);
433 // throw exception
434 throw std::system_error(kev_r[1].data, std::system_category());
435 }
436
437 return 0;
438#else
439 // OpenBSD doesn't have EV_RECEIPT: install the watches one at a time
440 struct kevent kev[1];
441
442 short rflags = EV_ADD | ((flags & IN_EVENTS) ? 0 : EV_DISABLE);
443 short wflags = EV_ADD | ((flags & OUT_EVENTS) ? 0 : EV_DISABLE);
444 EV_SET(&kev[0], fd, EVFILT_READ, rflags, 0, 0, userdata);
445
446 int r = kevent(kqfd, kev, 1, nullptr, 0, nullptr);
447
448 if (r == -1) {
449 throw std::system_error(errno, std::system_category());
450 }
451
452 EV_SET(&kev[0], fd, EVFILT_WRITE, wflags, 0, 0, userdata);
453
454 r = kevent(kqfd, kev, 1, nullptr, 0, nullptr);
455
456 if (r == -1) {
457 if (emulate) {
458 return OUT_EVENTS;
459 }
460 // remove read watch
461 EV_SET(&kev[0], fd, EVFILT_READ, EV_DELETE, 0, 0, userdata);
462 kevent(kqfd, kev, 1, nullptr, 0, nullptr);
463 // throw exception
464 throw std::system_error(errno, std::system_category());
465 }
466
467 return 0;
468#endif
469 }

577 // If events are pending, process an unspecified number of them.
578 // If no events are pending, wait until one event is received and
579 // process this event (and possibly any other events received
580 // simultaneously).
581 // If processing an event removes a watch, there is a possibility
582 // that the watched event will still be reported (if it has
583 // occurred) before pull_events() returns.
584 //
585 // do_wait - if false, returns immediately if no events are
586 // pending.
587 void pull_events(bool do_wait)
588 {
589 struct kevent events[16];
590 struct timespec ts;
591
592 // wait_ts remains null for an infinite wait; it is later set to either a 0 timeout
593 // if do_wait is false (or if we otherwise won't wait due to events being detected
594 // early) or is set to an appropriate timeout for the next timer's timeout.
595 struct timespec *wait_ts = nullptr;
596
597 // Check whether any timers are pending, and what the next timeout is.
598 Base::lock.lock();
599 this->process_monotonic_timers(do_wait, ts, wait_ts);
600 Base::lock.unlock();
601
602 if (! do_wait) {
603 ts.tv_sec = 0;
604 ts.tv_nsec = 0;
605 wait_ts = &ts;
606 }
607
608 int r = kevent(kqfd, nullptr, 0, events, 16, wait_ts);
609 if (r == -1 || r == 0) {
610 // signal or no events
611 if (r == 0 && do_wait) {
612 // timeout:
613 Base::lock.lock();
614 this->process_monotonic_timers();
615 Base::lock.unlock();
616 }
617 return;
618 }
619
620 ts.tv_sec = 0;
621 ts.tv_nsec = 0;
622
623 do {
624 process_events(events, r);
625 r = kevent(kqfd, nullptr, 0, events, 16, &ts);
626 } while (r > 0);
627 }
