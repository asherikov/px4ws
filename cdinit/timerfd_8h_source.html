14// Timer implementation based on Linux's "timerfd".
15
16// We could use one timerfd per timer, but then we need to differentiate timer
17// descriptors from regular file descriptors when events are reported by the loop
18// mechanism so that we can correctly report a timer event or fd event.
19
20// With a file descriptor or signal, we can use the item itself as the identifier for
21// adding/removing watches. For timers, it's more complicated. When we add a timer,
22// we are given a handle; we need to use this to modify the watch. We delegate the
23// process of allocating a handle to a priority heap implementation (BinaryHeap).
24
25template <class Base> class timer_fd_events : public timer_base<Base>

31 // Set the timerfd timeout to match the first timer in the queue (disable the timerfd
32 // if there are no active timers).
33 static void set_timer_from_queue(int fd, timer_queue_t &queue) noexcept
34 {
35 struct itimerspec newtime;
36 if (queue.empty()) {
37 newtime.it_value = {0, 0};
38 newtime.it_interval = {0, 0};
39 }
40 else {
41 newtime.it_value = queue.get_root_priority();
42 newtime.it_interval = {0, 0};
43 }
44 timerfd_settime(fd, TFD_TIMER_ABSTIME, &newtime, nullptr);
45 }

47 void process_timer(clock_type clock, int fd) noexcept
48 {
49 timer_queue_t &queue = this->queue_for_clock(clock);
50 struct timespec curtime;
51 switch (clock) {
52 case clock_type::SYSTEM:
53 clock_gettime(CLOCK_REALTIME, &curtime);
54 break;
55 case clock_type::MONOTONIC:
56 clock_gettime(CLOCK_MONOTONIC, &curtime);
57 break;
58 default:
59 DASYNQ_UNREACHABLE;
60 }
61
62 timer_base<Base>::process_timer_queue(queue, curtime);
63
64 // arm timerfd with timeout from head of queue
65 set_timer_from_queue(fd, queue);
66 }

68 void set_timer(timer_handle_t & timer_id, const time_val &timeouttv, const time_val &intervaltv,
69 timer_queue_t &queue, int fd, bool enable) noexcept
70 {
71 timespec timeout = timeouttv;
72 timespec interval = intervaltv;
73
74 std::lock_guard<decltype(Base::lock)> guard(Base::lock);
75
76 auto &ts = queue.node_data(timer_id);
77 ts.interval_time = interval;
78 ts.expiry_count = 0;
79 ts.enabled = enable;
80
81 if (queue.is_queued(timer_id)) {
82 // Already queued; alter timeout
83 if (queue.set_priority(timer_id, timeout)) {
84 set_timer_from_queue(fd, queue);
85 }
86 }
87 else {
88 if (queue.insert(timer_id, timeout)) {
89 set_timer_from_queue(fd, queue);
90 }
91 }

120 template <typename T> void init(T *loop_mech)

122 timerfd_fd = timerfd_create(CLOCK_MONOTONIC, TFD_CLOEXEC | TFD_NONBLOCK);

157 void stop_timer_nolock(timer_handle_t &timer_id, clock_type clock = clock_type::MONOTONIC) noexcept
158 {
159 timer_queue_t &queue = this->queue_for_clock(clock);
160 int fd = (clock == clock_type::MONOTONIC) ? timerfd_fd : systemtime_fd;
161 if (queue.is_queued(timer_id)) {
162 bool was_first = (&queue.get_root()) == &timer_id;
163 queue.remove(timer_id);
164 if (was_first) {
165 set_timer_from_queue(fd, queue);
