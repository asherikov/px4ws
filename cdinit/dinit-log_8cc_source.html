140void buffered_log_stream::flush_for_release()
141{
142 if (release) return;
143
144 release = true;
145
146 // Try to flush any messages that are currently buffered. (Console is non-blocking
147 // so it will fail gracefully).
148 rearm rearm_val = fd_event(event_loop, fd, dasynq::OUT_EVENTS);
149 if (rearm_val == rearm::DISARM) {
150 // Console has already been released at this point.
151 set_enabled(event_loop, false);
152 }
153 if (rearm_val == rearm::REMOVE) {
154 deregister(event_loop);
155 }
156 // fd_event didn't want to disarm, so must be partway through a message; will
157 // release when it's finished.
158}

160rearm buffered_log_stream::fd_event(eventloop_t &loop, int fd, int flags) noexcept
161{
162 if ((! partway) && (! special) && discarded) {
163 special_buf = "dinit: *** log message discarded due to full buffer ***\n";
164 special = true;
165 discarded = false;
166 msg_index = 0;
167 }
168
169 if ((! partway) && special) {
170 const char * start = special_buf + msg_index;
171 const char * end = start;
172 while (*end != '\n') end++;
173 int r = bp_sys::write(fd, start, end - start + 1);
174 if (r >= 0) {
175 if (start + r > end) {
176 // All written: go on to next message in queue
177 special = false;
178 discarded = false;
179 msg_index = 0;
180
181 if (release) {
182 release_console();
183 return rearm::DISARM;
184 }
185 }
186 else {
187 msg_index += r;
188 return rearm::REARM;
189 }
190 }
191 else if (errno != EAGAIN && errno != EINTR && errno != EWOULDBLOCK) {
192 return rearm::REMOVE;
193 }
194 return rearm::REARM;
195 }
196 else {
197 // Writing from the regular circular buffer
198
199 if (current_index == 0) {
200 release_console();
201 return rearm::DISARM;
202 }
203
204 // We try to find a complete line (terminated by '\n') in the buffer, and write it
205 // out. Since it may span the circular buffer end, it may consist of two distinct spans,
206 // and so we use writev to write them atomically.
207
208 struct iovec logiov[2];
209
210 char *ptr = log_buffer.get_ptr(0);
211 unsigned len = log_buffer.get_contiguous_length(ptr);
212 char *creptr = ptr + len; // contiguous region end
213 char *eptr = std::find(ptr, creptr, '\n');
214
215 bool will_complete = false; // will complete this message?
216 if (eptr != creptr) {
217 eptr++; // include '\n'
218 will_complete = true;
219 }
220
221 len = eptr - ptr;
222
223 logiov[0].iov_base = ptr;
224 logiov[0].iov_len = len;
225 int iovs_to_write = 1;
226
227 // Do we need the second span?
228 if (!will_complete && len != log_buffer.get_length()) {
229 ptr = log_buffer.get_buf_base();
230 creptr = ptr + log_buffer.get_length() - len;
231 eptr = std::find(ptr, creptr, '\n');
232 if (eptr != creptr) {
233 eptr++; // include '\n'
234 // It should not ever be the case that we do not now have a complete message
235 will_complete = true;
236 }
237 logiov[1].iov_base = ptr;
238 logiov[1].iov_len = eptr - ptr;
239 len += logiov[1].iov_len;
240 iovs_to_write = 2;
241 }
242
243 ssize_t r = bp_sys::writev(fd, logiov, iovs_to_write);
244
245 if (r >= 0) {
246 bool complete = (r == len) && will_complete;
247 log_buffer.consume(len);
248 partway = ! complete;
249 if (complete) {
250 current_index -= len;
251 if (current_index == 0 || release) {
252 // No more messages buffered / stop logging to console:
253 release_console();
254 return rearm::DISARM;
255 }
256 }
257 }
258 else if (errno != EAGAIN && errno != EINTR && errno != EWOULDBLOCK) {
259 return rearm::REMOVE;
260 }
261 }
262
263 // We've written something by the time we get here. We could fall through to below, but
264 // let's give other events a chance to be processed by returning now.
265 return rearm::REARM;
266}

392// Variadic method to log a sequence of strings as a single message to a particular facility:
393template <typename ... T> static void push_to_log(int idx, T ... args) noexcept
394{
395 if (! log_current_line[idx]) return;

406// Variadic method to potentially log a sequence of strings as a single message with the given log level:
407template <typename ... T> static void do_log(loglevel_t lvl, bool to_cons, T ... args) noexcept
408{

477// Log part of a message. A series of calls to do_log_part must be followed by a call to do_log_commit.
478static void do_log_part(int idx, const char *arg) noexcept
479{
480 if (log_current_line[idx]) {
481 int amount = sum_length(arg);
482 if (log_stream[idx].get_free() >= amount) {
483 append(log_stream[idx], arg);
484 }
485 else {
486 // we have to discard the message
487 log_stream[idx].rollback_msg();
488 log_current_line[idx] = false;
489 log_stream[idx].mark_discarded();
490 }
491 }
492}

494// Commit a message that was issued as a series of parts (via do_log_part).
495static void do_log_commit(int idx) noexcept
496{
497 if (log_current_line[idx]) {
