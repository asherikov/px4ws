44// Process a dependency directory - filenames contained within correspond to service names which
45// are loaded and added as a dependency of the given type. Expected use is with a directory
46// containing symbolic links to other service descriptions, but this isn't required.
47// Failure to read the directory contents, or to find a service listed within, is not considered
48// a fatal error.
49static void process_dep_dir(dirload_service_set &sset,
50 const char *servicename,
51 const string &service_filename,
52 std::list<prelim_dep> &deplist, const std::string &depdirpath,
53 dependency_type dep_type,
54 const service_record *avoid_circular)
55{
56 std::string depdir_fname = combine_paths(parent_path(service_filename), depdirpath.c_str());
57
58 DIR *depdir = opendir(depdir_fname.c_str());
59 if (depdir == nullptr) {
60 log(loglevel_t::WARN, "Could not open dependency directory '", depdir_fname,
61 "' for ", servicename, " service.");
62 return;
63 }
64
65 errno = 0;
66 dirent * dent = readdir(depdir);
67 while (dent != nullptr) {
68 char * name = dent->d_name;
69 if (name[0] != '.') {
70 try {
71 service_record * sr = sset.load_service(name);
72 deplist.emplace_back(sr, dep_type);
73 }
74 catch (service_not_found &) {
75 log(loglevel_t::WARN, "Ignoring unresolved dependency '", name,
76 "' in dependency directory '", depdirpath,
77 "' for ", servicename, " service.");
78 }
79 }
80 errno = 0; // errno may have changed in the meantime
81 dent = readdir(depdir);
82 }
83
84 if (errno != 0) {
85 log(loglevel_t::WARN, "Error reading dependency directory '", depdirpath,
86 "' for ", servicename, " service.");
87 }
88
89 closedir(depdir);
90}

104// Check for dependency cycles for the specified service (orig) with the given set of dependencies. Report
105// any cycles as occurring in _report_svc_name_.
106static void check_cycle(service_dep_list &deps, service_record *orig, const std::string &report_svc_name)
107{
108 linked_uo_set<service_record *> pending;
109 for (auto &new_dep : deps) {
110 if (new_dep.to == orig) {
111 throw service_cyclic_dependency(report_svc_name);
112 }
113 pending.add_back(new_dep.to);
114 }
115
116 for (auto i = pending.begin(); i != pending.end(); ++i) {
117 auto &dep_list = (*i)->get_dependencies();
118 for (auto &dep : dep_list) {
119 if (dep.get_to() == orig) {
120 throw service_cyclic_dependency(report_svc_name);
121 }
122 pending.add_back(dep.get_to());
123 }
124 }
125}

133// Update the dependencies of the specified service atomically.
134// May fail with bad_alloc, service_cyclic_dependency.
135static void update_depenencies(service_record *service,
136 dinit_load::service_settings_wrapper<prelim_dep> &settings,
137 std::list<service_dep> &before_deps)
138{
139 check_cycle(settings.depends, service);
140
141 std::list<service_dep> &deps = service->get_dependencies();
142 auto first_preexisting = deps.begin();
143
144 auto &depts = service->get_dependents();
145 auto first_pre_dept = depts.begin();
146
147 try {
148 // Insert all new dependents (from "before" relationships) before the first pre-existing dependent
149 for (auto new_dept_i = before_deps.begin(); new_dept_i != before_deps.end(); ) {
150 auto &new_dept = *new_dept_i;
151 depts.insert(depts.begin(), &new_dept);
152 // splice the dependency into the dependent:
153 auto next_dept_i = std::next(new_dept_i);
154 auto &from_deps = new_dept.get_from()->get_dependencies();
155 from_deps.splice(from_deps.begin(), before_deps, new_dept_i);
156 new_dept_i = next_dept_i;
157 }
158
159 // Insert all the new dependencies before the first pre-existing dependency
160 for (auto i = settings.depends.begin(); i != settings.depends.end(); ) {
161 auto &new_dep = *i;
162 service->add_dep(new_dep.to, new_dep.dep_type, first_preexisting);
163 i = settings.depends.erase(i);
164 }
165 }
166 catch (...) {
167 // remove "before" dependencies from dependents
168 for (auto i = depts.begin(); i != first_pre_dept; ) {
169 auto next_i = std::next(i);
170 (*i)->get_from()->rm_dep(**i);
171 i = next_i;
172 }
173
174 // remove the inserted dependencies
175 for (auto i = deps.begin(); i != first_preexisting; ) {
176 i = service->rm_dep(i);
177 }
178
179 // re-throw the exception
180 throw;
181 }
182
183 // Now remove all pre-existing dependencies, except for "before" dependencies (which come from the
184 // linked service and so must be retained)
185 for( ; first_preexisting != deps.end(); ) {
186 if (first_preexisting->dep_type != dependency_type::BEFORE) {
187 first_preexisting = service->rm_dep(first_preexisting);
188 }
189 else {
190 ++first_preexisting;
191 }
192 }
193
194 // Also remove pre-existing "before" dependents (because they come from this service)
195 for( ; first_pre_dept != depts.end(); ) {
196 auto next_pre_dept = std::next(first_pre_dept);
197 if ((*first_pre_dept)->dep_type == dependency_type::BEFORE) {
198 (*first_pre_dept)->get_from()->rm_dep(**first_pre_dept);
199 }
200 first_pre_dept = next_pre_dept;
201 }
202}

292 }
293 }

296 // Cannot change log type
297 if (value(service->get_type()).is_in(service_type_t::PROCESS, service_type_t::BGPROCESS,
298 service_type_t::SCRIPTED)) {
299 base_process_service *bps = static_cast<base_process_service *>(service);
300 if (bps->get_log_mode() != settings.log_type) {
301 throw service_load_exc(name, "cannot change log-type for running service.");
302 }
303 }
304
305 // Already started; we must replace settings on existing service record
306 create_new_record = false;
307 }
308 else if (settings.service_type == service->get_type()) {
309 // No need to create a new record if the type hasn't changed
310 create_new_record = false;
311 }
312
313 return create_new_record;

317 const service_record *avoid_circular)

319 // Load a new service, or reload an already-loaded service.
320
321 // For reload, we have the following problems:
322 // - ideally want to allow changing service type, at least for stopped services. That implies creating
323 // a new (replacement) service_record object, at least in cases where the type does change.
324 // - dependencies may change (including addition of new dependencies which aren't yet loaded). We need
325 // to prevent cyclic dependencies forming.
326 // - We want atomicity. If any new settings are not valid/alterable, or if a cyclic dependency is
327 // created, nothing should change.
328 // - We need to either transfer handles referring to the old service (so that they refer to the new
329 // service), or make them invalid. Or, we alter the original service without creating a new one
330 // (which we can only do if the type doesn't change).
331
332 // Approach:
333 // - determine whether we need a new service record or can alter the existing one
334 // (loading a new service always creates a new record; reload only creates a new record if the service
335 // type changes, and otherwise just changes the existing record in-place).
336 // - if loading a new service, a dummy record is created to enable easy cyclic dependency detection.
337 // (In other cases cycles must be checked by walking the service graph).
338 // The dummy is replaced with the real service once loading is complete (or is removed if it fails).
339 // - process settings from the service file (into a service_settings_wrapper).
340 // - check that the new settings are valid (for reload, if the service is running, check if the settings
341 // can be altered).
342 // - create the new record and install the new settings in it (or the existing record if not creating a
343 // new record). If doing a reload, check for cycles at this point (there is no dummy record in this
344 // case, so the quick cycle detection is not active).
345 // - (if doing a reload, with a new record) move the dependents on the original record to the new record.
346 //
347 // "Before" dependencies require special handling, as a "before = " specified in a service actually creates
348 // a dependency in the specified service on this service. Hence they always require explicit cycle checks
349 // (the quick cycle detection method using a dummy service cannot be used). For reloads this is done early,
350 // for new services it is done late (after the dummy has been removed).
351 //
352 // This is all an intricate dance. If failure occurs at any stage, we must restore the previous state.
353
354 // Limitations:
355 // - caller must check there are no handles (or only a single requesting handle) to the service before
356 // calling
357 // - cannot change the type of a non-stopped service

660 // then load the same service as a real dependency shortly afterwards, which would replace the
661 // placeholder but leave a dangling pointer to it in the list).

713 delete before_svc;
714 throw;

718 before_deps.emplace_back(before_svc, reload_svc, dependency_type::BEFORE);
719 // (note, we may need to adjust the to-service if we create a new service record object
720 // - this will be done later)

901
902 // first link in all the (new) "before" dependents (one way at this stage):
903 auto &dept_list = rval->get_dependents();
904 unsigned added_dependents = 0;
905 try {
906 for (auto &dept : before_deps) {
907 dept_list.push_back(&dept);
908 ++added_dependents;
909 }
910 }
911 catch (...) {
912 // Undo, since the invalid state will cause issues when the new service is disposed of
913 while (added_dependents > 0) {
914 dept_list.pop_back();
915 --added_dependents;
916 }
917 throw;
918 }

920 // --- Point of no return: mustn't fail from here ---
921
922 // Splice in the new "before" dependencies
923 auto i = before_deps.begin();
924 decltype(i) j;
925 while (i != before_deps.end()) {
926 j = std::next(i);
927 i->set_to(rval);
928 auto &from_deps = i->get_from()->get_dependencies();
929 from_deps.splice(from_deps.end(), before_deps, i);
930 i = j;
931 }

933 // Which service are we replacing? (It's either the one that's being reloaded, or a dummy)
934 service_record *orig_svc = (reload_svc != nullptr) ? reload_svc : dummy;
935
936 // Complete dependency/dependent transfers.
937
938 // Remove all "before" dependents from the original service (these were created by the
939 // original service itself)
940 auto &reload_depts = orig_svc->get_dependents();
941 for (auto i = reload_depts.begin(); i != reload_depts.end(); ) {
942 auto next_i = std::next(i);
943 if ((*i)->dep_type == dependency_type::BEFORE) {
944 service_record *before_svc = (*i)->get_from();
945 before_svc->rm_dep(**i);
946 if (before_svc->get_type() == service_type_t::PLACEHOLDER && before_svc->is_unrefd()) {
947 remove_service(before_svc);
948 delete before_svc;
949 }
950 }
951 i = next_i;
952 }
953
954 // Transfer dependents from the original service record to the new record;
955 // set links in all dependents on the original to point to the new service:
956 auto first_new_before = dept_list.begin();
957 dept_list.splice(first_new_before, reload_depts);
958 for (auto &dept : dept_list) {
959 dept->set_to(rval);
960 }
961
962 // Transfer all "before" dependencies (which are actually created by other services) to the
963 // new service
964 auto &dep_list_prev = orig_svc->get_dependencies();
965 auto &dep_list = rval->get_dependencies();
966 for (auto i = dep_list_prev.begin(); i != dep_list_prev.end(); ++i) {
967 if (i->dep_type == dependency_type::BEFORE) {
968 i->set_from(rval);
969 dep_list.splice(dep_list.end(), dep_list_prev, i++);
970 continue;
971 }
972 }
973
974 // Transfer any open file descriptors for (log) output to new service record
975 if (value(service_type).is_in(service_type_t::PROCESS, service_type_t::BGPROCESS, service_type_t::SCRIPTED)) {
976 ((base_process_service *)rval)->set_output_pipe_fds(orig_svc->transfer_output_pipe());
977 auto *orig_consumer = orig_svc->get_log_consumer();
978 if (orig_consumer != nullptr) {
979 orig_consumer->set_consumer_for(rval);
980 }
981 }
982
983 // Remove dependent-link for all dependencies from the original:
984 orig_svc->prepare_for_unload();
985
986 // Remove consumer-for link from consumed service, if it's changing:
987 if (value(orig_svc->get_type()).is_in(service_type_t::PROCESS, service_type_t::BGPROCESS)) {
988 process_service *ps_orig = static_cast<process_service *>(orig_svc);
989 auto *orig_consumed = ps_orig->get_consumed();
990 if (orig_consumed != nullptr && orig_consumed != consumer_of_svc) {
991 orig_consumed->set_log_consumer(nullptr);
992 }
993 }
994
995 // Finally, replace the old service with the new one:
996 auto iter = std::find(records.begin(), records.end(), orig_svc);
997 *iter = rval;
998 delete orig_svc;
999 }
1000
1001 // Mark as consumer for output of target service (if any)
1002 if (value(service_type).is_in(service_type_t::PROCESS, service_type_t::BGPROCESS)) {
1003 process_service *psvc = (process_service *)rval;
1004 psvc->set_consumer_for(consumer_of_svc);
1005 if (consumer_of_svc != nullptr) {
1006 consumer_of_svc->set_log_consumer(psvc);
1007 }
1008 }
1009
1010 return rval;
1011 }
1012 catch (service_description_exc &setting_exc)
1013 {
1014 exception_cleanup();
1015 if (setting_exc.service_name.empty()) {
1016 setting_exc.service_name = name;
1017 }
1018 throw;
1019 }
1020 catch (std::system_error &sys_err)
1021 {
1022 exception_cleanup();
1023 // don't use sys_err.what() since libstdc++ sometimes includes class names (basic_filebuf):
1024 std::string msg = input_stack.current_file_name() + ": " + sys_err.code().message();
1025 throw service_load_exc(name, std::move(msg));
1026 }
1027 catch (std::length_error &len_err) {
1028 // This is pretty much only theoretically possible; we'd normally expect bad_alloc instead.
1029 exception_cleanup();
1030 throw service_load_exc(name, "supported length for string/container exceeded");
1031 }
1032 catch (...) // (should only be std::bad_alloc or service_load_exc)
1033 {
1034 exception_cleanup();
1035 throw;
1036 }
