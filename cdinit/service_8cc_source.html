46void service_record::prepare_for_unload() noexcept
47{
48 // Remove all dependencies:
49 for (auto &dep : depends_on) {
50 service_record *dependency = dep.get_to();
51 auto &dep_dpts = dependency->dependents;
52 dep_dpts.erase(std::find(dep_dpts.begin(), dep_dpts.end(), &dep));
53 if (dep.dep_type == dependency_type::AFTER) {
54 if (dependency->get_type() == service_type_t::PLACEHOLDER) {
55 if (dependency->is_unrefd()) {
56 services->remove_service(dependency);
57 delete dependency;
58 }
59 }
60 }
61 }
62 depends_on.clear();
63
64 // Also remove all dependents. This should not be necessary except for "before" links.
65 for (auto i = dependents.begin(); i != dependents.end();) {
66 service_record *before_svc = (*i)->get_from();
67 before_svc->rm_dep(**i++);
68 if (before_svc->get_type() == service_type_t::PLACEHOLDER && before_svc->is_unrefd()) {
69 services->remove_service(before_svc);

75// Called when a service has actually stopped; dependents have stopped already, unless this stop
76// is due to an unexpected process termination.
77void service_record::stopped() noexcept
78{
79 if (have_console) {
80 bp_sys::tcsetpgrp(0, bp_sys::getpgrp());
81 release_console();
82 }
83
84 force_stop = false;
85
86 // If we are to re-start, restarting should have been set true and desired_state should be STARTED.
87 // (A restart could be cancelled via a separately issued stop, including via a shutdown).
88 bool will_restart = desired_state == service_state_t::STARTED && !pinned_stopped;
89
90 // If we won't restart, break soft dependencies now
91 if (!will_restart) {
92 for (auto dept : dependents) {
93 if (!dept->is_hard()) {
94 // waits-for or soft dependency:
95 if (dept->waiting_on) {
96 dept->waiting_on = false;
97 dept->get_from()->dependency_started();
98 }
99 if (dept->holding_acq) {
100 dept->holding_acq = false;
101 // release without issuing stop, since we're called only when this
102 // service is already stopped/stopping:
103 release(false);
104 }
105 }
106 }
107 }
108
109 for (auto & dependency : depends_on) {
110 // we signal dependencies in case they are waiting for us to stop:
111 dependency.get_to()->dependent_stopped();
112 }
113
114 service_state = service_state_t::STOPPED;
115
116 if (will_restart) {
117 // Desired state is "started".
118 initiate_start();
119 }
120 else {
121 becoming_inactive();
122
123 if (start_explicit) {
124 // If we were explicitly started, our required_by count must be at least 1. Use
125 // release() to correctly release, mark inactive and release dependencies.
126 start_explicit = false;
127 release(false);
128 }
129 else if (required_by == 0) {
130 // This can only be the case if we didn't have start_explicit, since required_by would
131 // otherwise by non-zero. Since our release(s) above were with state != STOPPED, we now
132 // must mark inactive (i.e. it won't have been done as part of the release).
133 services->service_inactive(this);
134 }
135 }
136
137 // Start failure will have been logged already, only log if we are stopped for other reasons:
138 if (!start_failed) {
139 log_service_stopped(service_name);
140
141 // If this service chains to another, start the chained service now, if:
142 // - this service self-terminated (rather than being stopped),
143 // - ... successfully (i.e. exit code 0)
144 // - this service won't restart, and
145 // - a shutdown isn't in progress
146 if ((onstart_flags.always_chain || (did_finish(stop_reason)
147 && get_exit_status().did_exit_clean() && !will_restart))
148 && !start_on_completion.empty() && !services->is_shutting_down()) {
149 try {
150 auto chain_to = services->load_service(start_on_completion.c_str());
151 chain_to->start();
152 }
153 catch (service_load_exc &sle) {
154 log(loglevel_t::ERROR, "Couldn't chain to service ", start_on_completion, ": ",
155 "couldn't load ", sle.service_name, ": ", sle.exc_description);
156 }
157 catch (std::bad_alloc &bae) {
158 log(loglevel_t::ERROR, "Couldn't chain to service ", start_on_completion,
159 ": Out of memory");
160 }
161 }
162 }
163 notify_listeners(service_event_t::STOPPED);
164}

178void service_record::release(bool issue_stop) noexcept
179{
180 if (--required_by == 0) {
181 if (service_state == service_state_t::STOPPING) {
182 // If we are stopping but would have restarted, we now need to notify that the restart
183 // has been cancelled. Other start-cancelled cases are handled by do_stop() (called
184 // below).
185 if (desired_state == service_state_t::STARTED && !is_start_pinned()) {
186 notify_listeners(service_event_t::STARTCANCELLED);
187 }
188 }
189 desired_state = service_state_t::STOPPED;
190
191 if (is_start_pinned()) return;
192
193 // Can stop, and can release dependencies now. We don't need to issue a release if
194 // a require was pending though:
195 prop_release = !prop_require;
196 prop_require = false;
197 if (prop_release) {
198 services->add_prop_queue(this);
199 }
200
201 if (service_state != service_state_t::STOPPED && service_state != service_state_t::STOPPING
202 && issue_stop) {
203 stop_reason = stopped_reason_t::NORMAL;
204 do_stop();
205 }
206 }

256 dep.get_to()->require();

273 }

335void service_record::do_start() noexcept
336{
337 bool was_active = service_state != service_state_t::STOPPED;
338
339 desired_state = service_state_t::STARTED;
340
341 if (pinned_stopped) {
342 if (!was_active) {
343 failed_to_start(false, false);
344 }
345 return;
346 }
347
348 // re-attach any soft dependents, now that we are starting again
349 if (!was_active) {
350 for (auto dept : dependents) {
351 if (!dept->is_hard()) {
352 service_state_t dept_state = dept->get_from()->service_state;
353 if (!dept->holding_acq
354 && (dept_state == service_state_t::STARTED || dept_state == service_state_t::STARTING)) {
355 dept->holding_acq = true;
356 ++required_by;
357 }
358 }
359 }
360 }
361
362 if (was_active) {
363 // We're already starting/started, or we are stopping and need to wait for
364 // that the complete.
365 if (service_state != service_state_t::STOPPING) {
366 return;
367 }
368
369 if (! can_interrupt_stop()) {
370 return;
371 }
372
373 // We're STOPPING, and that can be interrupted. Our dependencies might be STOPPING,
374 // but if so they are waiting (for us), so they too can be instantly returned to
375 // STARTING state.
376 notify_listeners(service_event_t::STOPCANCELLED);
377 }
378 else { // !was_active
379 services->service_active(this);
380 prop_require = !prop_release;
381 prop_release = false;
382 if (prop_require) {
383 services->add_prop_queue(this);
384 }
385 }
386
387 initiate_start();
388}

400bool service_record::start_check_dependencies() noexcept
401{
402 bool all_deps_started = true;
403
404 for (auto & dep : depends_on) {
405 service_record * to = dep.get_to();
406 if (dep.is_only_ordering()
407 && to->service_state != service_state_t::STARTING) continue;
408 if (to->service_state != service_state_t::STARTED) {
409 dep.waiting_on = true;
410 all_deps_started = false;
411 }
412 }
413
414 for (auto * dept : dependents) {
415 if (!dept->waiting_on && dept->is_only_ordering()) {
416 service_record *from = dept->get_from();
417 if (from->get_state() == service_state_t::STARTING) {
418 dept->waiting_on = true;
419 }
420 }
421 }
422
423 return all_deps_started;

504void service_record::failed_to_start(bool depfailed, bool immediate_stop) noexcept
505{
506 desired_state = service_state_t::STOPPED;
507
508 if (waiting_for_console) {
509 services->unqueue_console(this);
510 waiting_for_console = false;
511 }
512
513 if (start_explicit) {
514 start_explicit = false;
515 release(false);
516 }
517
518 // Cancel start of dependents:
519 for (auto & dept : dependents) {
520 switch (dept->dep_type) {
521 case dependency_type::REGULAR:
522 case dependency_type::MILESTONE:
523 // If REGULAR and STARTED, we can't have failed to start i.e. we must be started, so
524 // we don't worry about that case. If MILESTONE and started the dependency is already
525 // satisfied so again we don't need to do anything.
526 if (dept->get_from()->service_state == service_state_t::STARTING) {
527 dept->get_from()->prop_failure = true;
528 services->add_prop_queue(dept->get_from());
529 }
530 break;
531 case dependency_type::WAITS_FOR:
532 case dependency_type::SOFT:
533 case dependency_type::BEFORE:
534 case dependency_type::AFTER:
535 if (dept->waiting_on) {
536 dept->waiting_on = false;
537 dept->get_from()->dependency_started();
538 }
539 }
540
541 // Always release now, so that our desired state will be STOPPED before we call
542 // stopped() below (if we do so). Otherwise it may decide to restart us.
543 if (dept->holding_acq) {
544 dept->holding_acq = false;
545 release(false);
546 }
547 }
548
549 start_failed = true;
550 log_service_failed(get_name(), depfailed);
551 notify_listeners(service_event_t::FAILEDSTART);
552 pinned_started = false;
553
554 if (immediate_stop) {
555 stopped();
556 }
557}

591void service_record::stop(bool bring_down) noexcept
592{
593 // Stop; remove activation, and don't self-restart.
594
595 if (start_explicit) {
596 start_explicit = false;
597 required_by--;
598 }
599
600 if (bring_down || required_by == 0) {
601 // Set desired state to STOPPED, this will inhibit automatic restart (and will be
602 // propagated to dependents)
603 desired_state = service_state_t::STOPPED;
604 }
605
606 if (is_start_pinned()) {
607 return;
608 }
609
610 // If our required_by count is 0, we should treat this as a full manual stop regardless
611 if (required_by == 0) {
612 bring_down = true;
613 prop_release = !prop_require;
614 if (prop_release) {
615 services->add_prop_queue(this);
616 }
617 }
618
619 if (bring_down && service_state != service_state_t::STOPPED) {
620 // Note even if we are already STOPPING we should call do_stop for the case where we need
621 // to interrupt any currently ongoing restart of dependents.
622 stop_reason = stopped_reason_t::NORMAL;
623 do_stop();
624 }

643void service_record::do_stop(bool with_restart) noexcept
644{
645 // Called when we should definitely stop. We may need to restart afterwards, but we
646 // won't know that for sure until the execution transition.
647 // Note: to inhibit automatic restart, including restart due to dependent still requiring this
648 // service, caller must first set desired_state to STOPPED
649
650 if (is_start_pinned()) return;
651
652 in_auto_restart = false;
653 in_user_restart = false;
654
655 // Will we restart? desired state of STOPPED inhibits auto-restart
656 bool do_auto_restart = false;
657 if (auto_restart == auto_restart_mode::ALWAYS) {
658 do_auto_restart = true;
659 }
660 else if (auto_restart == auto_restart_mode::ON_FAILURE) {
661 if (get_type() == service_type_t::PROCESS || get_type() == service_type_t::BGPROCESS) {
662 auto exit_status = get_exit_status();
663 if (exit_status.was_signalled()) {
664 if (!value(exit_status.get_signal()).is_in(SIGHUP, SIGINT, SIGUSR1, SIGUSR2, SIGTERM)) {
665 do_auto_restart = true;
666 }
667 }
668 else {
669 if (exit_status.get_exit_status() != 0) {
670 do_auto_restart = true;
671 }
672 }
673 }
674 }
675 bool for_restart = with_restart || (do_auto_restart && desired_state == service_state_t::STARTED);
676 bool restart_deps = with_restart;
677
678 if (!with_restart && for_restart) {
679 // auto restart - check for restarting too often
680 for_restart = check_restart();
681 in_auto_restart = for_restart;
682 }
683
684 // If we won't restart, release explicit activation:
685 if (!for_restart) {
686 if (start_explicit) {
687 start_explicit = false;
688 release(false);
689 }
690 }
691
692 bool all_deps_stopped = stop_dependents(for_restart, restart_deps);
693
694 if (service_state != service_state_t::STARTED) {
695 if (service_state == service_state_t::STARTING) {
696 // If waiting for a dependency, or waiting for the console, we can interrupt start. Otherwise,
697 // we need to delegate to can_interrupt_start() (which can be overridden).
698 if (!waiting_for_deps && !waiting_for_console) {
699 if (!can_interrupt_start()) {
700 // Well this is awkward: we're going to have to continue starting. We can stop once
701 // we've reached the started state.
702 return;
703 }
704
705 if (!interrupt_start()) {
706 // Now wait for service startup to actually end; we don't need to handle it here.
707 notify_listeners(service_event_t::STARTCANCELLED);
708 return;
709 }
710 }
711 else if (waiting_for_console) {
712 services->unqueue_console(this);
713 waiting_for_console = false;
714 }
715
716 // We must have had desired_state == STARTED.
717 notify_listeners(service_event_t::STARTCANCELLED);
718
719 // Reaching this point, we are starting interruptibly - so we
720 // stop now (by falling through to below).
721 }
722 else {
723 // If we're starting we need to wait for that to complete.
724 // If we're already stopping/stopped there's nothing to do.
725 return;
726 }
727 }
728
729 service_state = service_state_t::STOPPING;
730 waiting_for_deps = !all_deps_stopped;
731 if (all_deps_stopped) {
732 services->add_transition_queue(this);
733 }
734}

736bool service_record::stop_check_dependents() noexcept
737{
738 bool all_deps_stopped = true;
739 for (auto dept : dependents) {
740 // Note if the dependent is waiting on us, it must be restarting (since the
741 // waiting_on flag gets cleared when we stop, and would only be set if the
742 // service tries to restart). We can treat that as "stopped" for purposes of
743 // checking whether we can transition to stopped state.
744 if (dept->is_hard() && dept->holding_acq && !dept->waiting_on) {
745 all_deps_stopped = false;

753bool service_record::stop_dependents(bool for_restart, bool restart_deps) noexcept
754{
755 // We are in either STARTED or STARTING states.
756 bool all_deps_stopped = true;
757 for (auto dept : dependents) {
758 if (dept->is_hard()) {
759 service_record *dep_from = dept->get_from();
760
761 if (!dep_from->is_fundamentally_stopped()) {
762 // Note we check *first* since if the dependent service is not stopped,
763 // 1. We will issue a stop to it shortly and
764 // 2. It will notify us when stopped, at which point the stop_check_dependents()
765 // check is run anyway.
766 all_deps_stopped = false;
767 }
768
769 if (force_stop) {
770 // If this service is to be forcefully stopped, dependents must also be.
771 if (desired_state == service_state_t::STOPPED) {
772 // If our target state was forced to STOPPED, this is a failure
773 dep_from->stop_reason = stopped_reason_t::DEPFAILED;
774 dep_from->unrecoverable_stop();
775 }
776 else {
777 dep_from->forced_stop();
778 }
779 }
780
781 if (dep_from->get_state() != service_state_t::STOPPED) {
782 if (desired_state == service_state_t::STOPPED) {
783 if (dep_from->desired_state != service_state_t::STOPPED) {
784 // if we don't want to restart, don't restart dependent
785 dep_from->desired_state = service_state_t::STOPPED;
786 if (dep_from->start_explicit) {
787 dep_from->start_explicit = false;
788 dep_from->release(true);
789 }
790 dep_from->prop_stop = true;
791 services->add_prop_queue(dep_from);
792 }
793 }
794 else if (restart_deps && dep_from->get_state() != service_state_t::STOPPING) {
795 // restart dependent and propagate restart to all their hard dependents.
796 dep_from->stop_reason = stopped_reason_t::DEPRESTART;
797 dep_from->in_user_restart = true;
798 dep_from->prop_stop = true;
799 services->add_prop_queue(dep_from);
800 }
801 }
802 }
803 // Note that soft dependencies are retained if restarting, but otherwise
804 // they are broken.
805 else if (!for_restart) {
806 if (dept->waiting_on) {
807 // Note, milestone which is still waiting is considered a hard dependency and
808 // is handled above. This is therefore a true soft dependency, and we can just
809 // break the dependency link.
810 dept->waiting_on = false;
811 dept->get_from()->dependency_started();
812 }
813 if (dept->holding_acq) {
814 dept->holding_acq = false;
815 release(false);
816 }
817 }
818 }
819
820 return all_deps_stopped;
821}

823// All dependents have stopped; we can stop now, too. Only called when STOPPING.
824void service_record::bring_down() noexcept
825{
826 stopped();
827}

829void service_record::pin_start() noexcept
830{
831 if (!pinned_started) {
832 if (!dept_pinned_started) {
833 for (auto &dep : depends_on) {
834 if (dep.is_hard() && !dep.get_to()->dept_pinned_started) {
835 dep.get_to()->prop_pin_dpt = true;
836 services->add_prop_queue(dep.get_to());
837 }
838 }
839 }
840 pinned_started = true;
841 }
842}
843
844void service_record::unpin() noexcept
845{
846 if (pinned_started) {
847 pinned_started = false;
848
849 if (dept_pinned_started) return;
850
851 // unpin dependencies
852 for (auto &dep : depends_on) {
853 if (dep.is_hard() && dep.get_to()->dept_pinned_started) {
854 dep.get_to()->prop_pin_dpt = true;
855 services->add_prop_queue(dep.get_to());
856 }
857 }
858
859 // We only need special handling here if service was in STARTED state
860 if (service_state == service_state_t::STARTED) {
861 // If any dependents are stopping, then force_stop should already be set.
862
863 // If we reached required_by 0, we need to propagate release now (since it wasn't
864 // propagated as it normally would be when we hit 0, due to the pin)
865 if (required_by == 0) {
866 prop_release = true;
867 services->add_prop_queue(this);
868 }
869
870 if (desired_state == service_state_t::STOPPED || force_stop) {
871 do_stop();
872 services->process_queues();
873 }
874 }
875 }
876 if (pinned_stopped) {
877 pinned_stopped = false;
878 // We don't need to check state. If we're pinned stopped we can't be required and so desired
879 // state should always be stopped.
880 }
881}
