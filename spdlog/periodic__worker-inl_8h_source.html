13SPDLOG_INLINE periodic_worker::periodic_worker(const std::function<void()> &callback_fun, std::chrono::seconds interval)
14{
15 active_ = (interval > std::chrono::seconds::zero());

17 {
18 return;
19 }
20

30 }
31 });
32}
