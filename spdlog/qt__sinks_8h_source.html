22namespace spdlog {
23namespace sinks {
24template <typename Mutex> class qt_sink : public base_sink<Mutex> {
25public:
26 qt_sink(QObject *qt_object = nullptr, const std::string &meta_method = "") {
27 qt_object_ = qt_object;
28 meta_method_ = meta_method;
29 }
30
31 ~qt_sink() { flush_(); }
32
33protected:
34 void sink_it_(const details::log_msg &msg) override {
35 memory_buf_t formatted;
36 base_sink<Mutex>::formatter_->format(msg, formatted);
37 string_view_t str = string_view_t(formatted.data(), formatted.size());
38 QMetaObject::invokeMethod(qt_object_, meta_method_.c_str(), Qt::AutoConnection,
39 Q_ARG(QString, QString::fromUtf8(str.data(), static_cast<int>(str.size())).trimmed()));
40 }
41
42 void flush_() override {}
43
44private:
45 QObject *qt_object_ = nullptr;
46 std::string meta_method_;
47};
